```python
    """OpenAPI schema must be valid OpenAPI 3.0+"""
    from openapi_spec_validator import validate_spec
    validate_spec(app.openapi())

def test_all_endpoints_documented(api_schema):
    """All routes must be documented in OpenAPI spec"""
    documented_paths = set(api_schema.paths.keys())
    actual_routes = get_all_routes(app)
    
    missing = actual_routes - documented_paths
    assert not missing, f"Undocumented routes: {missing}"

def test_security_schemes_defined(api_schema):
    """Security schemes must be defined for protected endpoints"""
    assert "bearerAuth" in api_schema.components.security_schemes
    
    for path, methods in api_schema.paths.items():
        if path.startswith("/api/") and path != "/api/health":
            for method in methods:
                assert methods[method].security, f"{method} {path} missing security"
```
