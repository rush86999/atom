```python
def test_readiness_db_connection_refused():
    """Database connection failure returns 503"""
    with patch("core.health_checks.SessionLocal") as mock_db:
        mock_db.side_effect = ConnectionError("Database unavailable")
        
        response = client.get("/health/ready")
        
        assert response.status_code == 503
        assert response.json()["status"] == "unhealthy"
        assert "database" in response.json()["checks"]
        assert response.json()["checks"]["database"]["status"] == "error"

def test_error_response_format():
    """All error responses follow consistent format"""
    response = client.get("/health/ready")
    
    if response.status_code != 200:
        error = response.json()
        assert "error_code" in error or "status" in error
        assert "message" in error or "detail" in error
```

```python
@pytest.mark.parametrize("action,complexity,student_blocked,intern_blocked", [
    ("present", 1, False, False),
    ("stream", 2, True, False),
    ("form_submit", 3, True, False),
    ("delete_data", 4, True, True),
])
def test_governance_action_complexity_enforcement(action, complexity, student_blocked, intern_blocked):
    """Action complexity properly enforced by maturity"""
    response = client.post(
        f"/api/agent/{action}",
        json={"agent_id": "test-agent", "data": {}},
        headers={"X-Agent-Maturity": "STUDENT" if student_blocked else "INTERN"}
    )
    
    if student_blocked:
        assert response.status_code == 403
    elif intern_blocked and complexity >= 4:
        assert response.status_code == 403

def test_governance_cache_invalidation():
    """Cache properly invalidates after permission changes"""
    # First call - cache miss
    with patch("core.governance_cache.get") as mock_get:
        mock_get.return_value = None
        
        response = client.post(
            "/api/agent/chat",
            headers={"X-Agent-Maturity": "STUDENT"}
        )
        
        assert mock_get.called
    
    # Second call - cache hit
    with patch("core.governance_cache.get") as mock_get:
        mock_get.return_value = {"allowed": True}
        
        response = client.post(
            "/api/agent/chat",
            headers={"X-Agent-Maturity": "STUDENT"}
        )
        
        assert mock_get.called
```
