        # Tests 1-5: Breakpoint Management
        def test_set_breakpoint(self, debugger):

            workflow_id = "wf-001"
            step_id = "step-001"

            result = debugger.set_breakpoint(workflow_id, step_id)

        def test_clear_breakpoint(self, debugger):
            """Test clearing a breakpoint."""
            workflow_id = "wf-001"
            step_id = "step-001"
            debugger.set_breakpoint(workflow_id, step_id)

            result = debugger.clear_breakpoint(workflow_id, step_id)

            assert result is True
            assert (workflow_id, step_id) not in debugger.breakpoints

            mock_engine_class.return_value = mock_engine
