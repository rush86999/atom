2. Create `backend/tests/e2e_ui/scripts/` directory

3. Create `backend/tests/e2e_ui/scripts/start-test-env.sh`:
```bash
#!/bin/bash
# Start E2E UI test environment

set -e

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
PROJECT_ROOT="$(cd "$SCRIPT_DIR/../.." && pwd)"

echo "=== Starting Atom E2E UI Test Environment ==="
echo "Project root: $PROJECT_ROOT"

cd "$PROJECT_ROOT"

# Start services
docker-compose -f docker-compose-e2e-ui.yml up -d

echo ""
echo "=== Waiting for services to be healthy ==="
# Wait for backend (max 60s)
for i in {1..12}; do
  if curl -sf http://localhost:8001/health/live > /dev/null; then
    echo "Backend is ready!"
    break
  fi
  echo "Waiting for backend... ($i/12)"
  sleep 5
done

# Wait for frontend (max 60s)
for i in {1..12}; do
  if curl -sf http://localhost:3001 > /dev/null; then
    echo "Frontend is ready!"
    break
  fi
  echo "Waiting for frontend... ($i/12)"
  sleep 5
done

echo ""
echo "=== Services Started ==="
echo "Backend:  http://localhost:8001"
echo "Frontend: http://localhost:3001"
echo "Postgres: localhost:5434"
echo ""
echo "Run tests with:"
echo "  PYTHONPATH=backend pytest backend/tests/e2e_ui/ -v"
```

4. Create `backend/tests/e2e_ui/scripts/stop-test-env.sh`:
```bash
#!/bin/bash
# Stop E2E UI test environment

set -e

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
PROJECT_ROOT="$(cd "$SCRIPT_DIR/../.." && pwd)"

echo "=== Stopping Atom E2E UI Test Environment ==="

cd "$PROJECT_ROOT"

docker-compose -f docker-compose-e2e-ui.yml down -v

echo "=== Environment Stopped ==="
echo "Volumes removed. Run start-test-env.sh to start again."
```

