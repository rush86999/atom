    ```python
    from hypothesis import given, settings
    from hypothesis.strategies import integers, floats

    @settings(max_examples=200)
    @given(
        days_old=integers(min_value=0, max_value=365)
    )
    def test_decay_formula_bounds(days_old):
        # Decay formula: max(0, 1 - days_old/180)
        decay = max(0, 1 - (days_old / 180))
        assert 0.0 <= decay <= 1.0, f"Decay {decay} out of bounds for {days_old} days"

    @settings(max_examples=200)
    @given(
        initial_importance=floats(min_value=0.0, max_value=1.0, allow_nan=False),
        access_count=integers(min_value=0, max_value=100)
    )
    def test_importance_bounds(initial_importance, access_count):
        # Importance with access boost
        access_boost = min(access_count / 100, 0.2)
        importance = min(initial_importance + access_boost, 1.0)
        assert 0.0 <= importance <= 1.0
    ```
