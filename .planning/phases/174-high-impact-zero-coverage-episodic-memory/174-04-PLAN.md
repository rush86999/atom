    Use Hypothesis strategies:
    ```python
    from hypothesis import given, settings
    from hypothesis.strategies import integers, floats, sampled_from

    @settings(max_examples=200)
    @given(
        episode_count=integers(min_value=0, max_value=100),
        intervention_count=integers(min_value=0, max_value=100)
    )
    def test_intervention_rate_bounds(episode_count, intervention_count):
        if episode_count == 0:
            return  # Skip division by zero
        intervention_rate = intervention_count / episode_count
        assert 0.0 <= intervention_rate <= 1.0, \
            f"Intervention rate {intervention_rate} out of bounds"

    @settings(max_examples=200)
    @given(
        episode_score=floats(min_value=0, max_value=40, allow_nan=False),
        intervention_score=floats(min_value=0, max_value=30, allow_nan=False),
        compliance_score=floats(min_value=0, max_value=30, allow_nan=False)
    )
    def test_score_components_sum_to_total(episode_score, intervention_score, compliance_score):
        total_score = (episode_score + intervention_score + compliance_score) / 100
        assert 0.0 <= total_score <= 1.0, f"Total score {total_score} out of bounds"
    ```
