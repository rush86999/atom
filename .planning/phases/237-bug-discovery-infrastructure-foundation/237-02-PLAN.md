@pytest.mark.browser
@pytest.mark.slow
def test_broken_links_on_[page](authenticated_page: Page):
    """Discover broken links on [page]."""
    links = authenticated_page.locator("a").all()
    broken_links = []

    for link in links:
        href = link.get_attribute("href")
        response = authenticated_page.request.get(href)
        if response.status == 404:
            broken_links.append(href)

    assert len(broken_links) == 0, f"Broken links: {broken_links}"
