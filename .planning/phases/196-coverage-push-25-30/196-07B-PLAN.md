```python
def test_workflow_transaction_rollback(db: Session, mock_background_thread):
    """Test transaction rollback on workflow failure."""
    workflow = WorkflowFactory.create(
        definition={
            "nodes": [
                {"id": "1", "type": "task"},
                {"id": "2", "type": "task", "fail": True}  # This step fails
            ],
            "edges": [["1", "2"]]
        }
    )

    engine = WorkflowEngine(db)
    result = engine.execute(workflow.id)

    # Verify rollback
    execution = db.query(WorkflowExecution).filter_by(
        workflow_id=workflow.id
    ).first()

    assert execution.status == "failed"
    # Verify no partial data was committed
```
