    ```python
    import requests

    def make_platform_request(setup_test_user, endpoint: str, platform: str = "web"):
        """Make API request with platform header."""
        user_data = setup_test_user()
        token = user_data.get("access_token")

        headers = {
            "Authorization": f"Bearer {token}",
            "Content-Type": "application/json"
        }

        if platform != "web":
            headers["X-Platform"] = platform

        response = requests.get(
            f"http://localhost:8000{endpoint}",
            headers=headers
        )

        return response.json() if response.status_code == 200 else None

    def compare_schemas(response1: dict, response2: dict, required_fields: list):
        """Compare two response schemas have same required fields."""
        for field in required_fields:
            assert field in response1, f"Field {field} missing in response1"
            assert field in response2, f"Field {field} missing in response2"
            assert type(response1[field]) == type(response2[field]), \
                f"Field {field} type mismatch: {type(response1[field])} vs {type(response2[field])}"
    ```
