    ```python
        parts = token.split('.')
        payload = parts[1]
        # Add padding if needed
        padding = len(payload) % 4
        if padding != 0:
            payload += '=' * (4 - padding)
        decoded = base64.b64decode(payload)
        return json.loads(decoded)
    ```
