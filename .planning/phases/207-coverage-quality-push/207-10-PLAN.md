Generate comprehensive coverage report:

```bash
#!/bin/bash
# Phase 207 Coverage Aggregation Script

echo "=== Phase 207 Coverage Aggregation ===" > coverage_aggregation.txt
echo "Date: $(date)" >> coverage_aggregation.txt
echo "" >> coverage_aggregation.txt

# Run coverage for all new tests
echo "Running coverage for all Phase 207 tests..."
pytest \
  tests/unit/api/test_reports.py \
  tests/unit/api/test_websocket_routes.py \
  tests/unit/api/test_workflow_analytics_routes.py \
  tests/unit/api/test_time_travel_routes.py \
  tests/unit/api/test_onboarding_routes.py \
  tests/unit/api/test_sales_routes.py \
  tests/unit/core/test_lux_config.py \
  tests/unit/core/test_messaging_schemas.py \
  tests/unit/core/test_billing.py \
  tests/unit/core/test_llm_service.py \
  tests/unit/core/test_historical_learner.py \
  tests/unit/core/test_external_integration_service.py \
  tests/unit/tools/test_device_tool.py \
  tests/unit/tools/test_browser_tool.py \
  tests/unit/tools/test_canvas_tool.py \
  tests/unit/agent/test_agent_graduation_service_incremental.py \
  tests/unit/episodes/test_episode_retrieval_service_incremental.py \
  tests/unit/llm/test_byok_handler_incremental.py \
  --cov=api.reports \
  --cov=api.websocket_routes \
  --cov=api.workflow_analytics_routes \
  --cov=api.time_travel_routes \
  --cov=api.onboarding_routes \
  --cov=api.sales_routes \
  --cov=core.lux_config \
  --cov=core.messaging_schemas \
  --cov=core.billing \
  --cov=core.llm_service \
  --cov=core.historical_learner \
  --cov=core.external_integration_service \
  --cov=tools.device_tool \
  --cov=tools.browser_tool \
  --cov=tools.canvas_tool \
  --cov=core.agent_graduation_service \
  --cov=core.episode_retrieval_service \
  --cov=core.llm.byok_handler \
  --cov-branch \
  --cov-report=html \
  --cov-report=term \
  --cov-report=json \
  -v 2>&1 | tee coverage_run.log

# Extract metrics
echo "" >> coverage_aggregation.txt
echo "=== Coverage Summary ===" >> coverage_aggregation.txt
grep -A 20 "TOTAL" coverage_run.log >> coverage_aggregation.txt

# Count tests
echo "" >> coverage_aggregation.txt
echo "=== Test Count ===" >> coverage_aggregation.txt
grep -E "passed|failed" coverage_run.log | tail -5 >> coverage_aggregation.txt

# Branch coverage
echo "" >> coverage_aggregation.txt
echo "=== Branch Coverage ===" >> coverage_aggregation.txt
coverage report --show-missing | grep -E "branch|BRANCH" >> coverage_aggregation.txt || echo "Branch coverage not available" >> coverage_aggregation.txt

# File-level quality (75%+ files)
echo "" >> coverage_aggregation.txt
echo "=== File-Level Quality (75%+ files) ===" >> coverage_aggregation.txt
coverage report --sort=cover | awk '$4 >= "75.0%" {print $1, $4}' | wc -l | xargs -I {} echo "Files at 75%+: {}" >> coverage_aggregation.txt

# Collection errors
echo "" >> coverage_aggregation.txt
echo "=== Collection Errors ===" >> coverage_aggregation.txt
pytest --collect-only 2>&1 | grep -i error || echo "No collection errors ✓" >> coverage_aggregation.txt
```
