class TestCreateInvoice:
    """Tests for invoice creation."""

    def test_create_invoice_success(self, db_session: Session):
        """Test create_invoice creates invoice with line items."""
        service = BillingService(db_session)

        # Act: Create invoice
        invoice = service.create_invoice(
            user_id="user-123",
            items=[
                {"description": "Service A", "amount": 100.00},
                {"description": "Service B", "amount": 50.00}
            ]
        )

        # Assert: Invoice created
        assert invoice.user_id == "user-123"
        assert invoice.subtotal == 150.00
        assert invoice.total_amount == 150.00
        assert len(invoice.line_items) == 2

    def test_create_invoice_with_tax(self, db_session: Session):
        """Test create_invoice calculates tax correctly."""
        service = BillingService(db_session)

        # Act: Create invoice with tax
        invoice = service.create_invoice(
            user_id="user-123",
            items=[{"description": "Service", "amount": 100.00}],
            tax_rate=0.10
        )

        # Assert: Tax calculated
        assert invoice.subtotal == 100.00
        assert invoice.tax_amount == 10.00
        assert invoice.total_amount == 110.00

class TestStreamResponse:
    """Tests for streaming responses."""

    @patch('core.llm_service.openai.ChatCompletion.create')
    @pytest.mark.asyncio
    async def test_stream_openai(self, mock_create):
        """Test stream_response with OpenAI."""
        # Mock streaming response
        async def mock_stream():
            chunks = ["Hello", " world", "!"]
            for chunk in chunks:
                yield {"choices": [{"delta": {"content": chunk}}]}

        mock_create.return_value = mock_stream()

        service = LLMService()

        # Act: Stream response
        chunks = []
        async for chunk in service.stream_response(
            provider=LLMProvider.OPENAI,
            prompt="Test"
        ):
            chunks.append(chunk)

        # Assert: Chunks received
        assert len(chunks) == 3
        assert chunks[0].content == "Hello"
