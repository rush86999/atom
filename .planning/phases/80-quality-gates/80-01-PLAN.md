    2. PRESERVE existing fixtures (screenshot_page, video_page, authenticated_page, etc.)
    3. Add pytest_runtest_makereport hook extension:

    ```python
    @pytest.hookimpl(tryfirst=True, hookwrapper=True)
    def pytest_runtest_makereport(item, call):
        """Extended hook to capture screenshots on test failure."""
        outcome = yield
        rep = outcome.get_result()

        # Store test outcome for fixtures to access
        setattr(item, "rep_" + rep.when, rep)

        # Capture screenshot on test failure
        if rep.when == "call" and rep.failed:
            # Get page fixture if available
            page = getattr(item, "_page", None)
            if page is None:
                # Try to get page from function args
                if hasattr(item, "funcargs"):
                    page = item.funcargs.get("page") or item.funcargs.get("authenticated_page")

            if page is not None:
                import os
                from datetime import datetime

                # Create screenshots directory if not exists
                screenshot_dir = "backend/tests/e2e_ui/artifacts/screenshots"
                os.makedirs(screenshot_dir, exist_ok=True)

                # Generate descriptive filename
                timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
                test_name = item.name.replace("::", "_").replace("/", "_")[:100]
                screenshot_path = f"{screenshot_dir}/{timestamp}_{test_name}.png"

                # Capture full page screenshot
                page.screenshot(path=screenshot_path, full_page=True)
                print(f"\nScreenshot saved: {screenshot_path}")
    ```

    4. Add autouse fixture to track page object:
    ```python
    @pytest.fixture(autouse=True)
    def track_page_for_screenshots(request, page):
        """Track page object for screenshot capture on failure."""
        if hasattr(request, "_funcargs"):
            request.node._page = page
        yield
    ```

    DO NOT delete or modify the existing pytest_runtest_makereport hook - extend it.
