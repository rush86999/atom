**Pattern for cascade tests:**
```python
def test_example_cascade(self, db_session):
    parent = ParentFactory(_session=db_session)
    child1 = ChildFactory(parent_id=parent.id, _session=db_session)
    child2 = ChildFactory(parent_id=parent.id, _session=db_session)
    db_session.commit()

    # Delete parent
    db_session.delete(parent)
    db_session.commit()

    # Verify children are deleted
    remaining = db_session.query(Child).filter(
        Child.parent_id == parent.id
    ).all()
    assert len(remaining) == 0
```
