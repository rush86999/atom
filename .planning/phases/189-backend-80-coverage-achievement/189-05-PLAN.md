```bash
cd /Users/rushiparikh/projects/atom/backend
python -c "
import json
from pathlib import Path

# Load coverage data
with open('coverage.json', 'r') as f:
    data = json.load(f)

# Calculate overall coverage
total_statements = 0
total_covered = 0
files_analyzed = 0

# Zero coverage files
zero_coverage = []
# Partial coverage files
partial_coverage = []
# 80%+ coverage files
high_coverage = []

for file_path, file_data in data['files'].items():
    stmts = file_data['summary']['num_statements']
    if stmts == 0:
        continue

    files_analyzed += 1
    total_statements += stmts
    covered = file_data['summary']['covered_lines']
    total_covered += covered
    pct = file_data['summary']['percent_covered']

    if pct == 0.0:
        zero_coverage.append((file_path, stmts))
    elif pct >= 80.0:
        high_coverage.append((file_path, pct, stmts))
    else:
        partial_coverage.append((file_path, pct, stmts, covered, stmts - covered))

overall_pct = (total_covered / total_statements * 100) if total_statements > 0 else 0

print(f'Phase 189 Final Coverage Report')
print(f'='*50)
print(f'Overall Coverage: {overall_pct:.2f}% ({total_covered}/{total_statements} statements)')
print(f'Files Analyzed: {files_analyzed}')
print(f'High Coverage (80%+): {len(high_coverage)} files')
print(f'Partial Coverage: {len(partial_coverage)} files')
print(f'Zero Coverage: {len(zero_coverage)} files')
print()
print('Top 20 Zero-Coverage Files by Statement Count:')
zero_coverage.sort(key=lambda x: x[1], reverse=True)
for i, (path, stmts) in enumerate(zero_coverage[:20], 1):
    print(f'{i:2}. {path}: {stmts} statements')
"
```
