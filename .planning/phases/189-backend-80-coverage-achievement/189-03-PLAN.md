class TestErrorHandling:
    """Test error handling and retry logic."""

        (1, 2, True),
        (2, 2, True),
        (3, 2, False),
    ])
    def test_retry_on_failure(self, failure_count, max_retries, should_succeed):
        """Cover retry logic for failed agent executions."""
        meta_agent = AtomMetaAgent(max_retries=max_retries)

        agent = MagicMock(id="agent-1", capabilities=["task"])

        # Fail first N times, succeed on next
        call_count = 0

        async def failing_execute(*args, **kwargs):
            nonlocal call_count
            call_count += 1
            if call_count <= failure_count:
                raise Exception("Agent failure")
            return TaskResult(task_id="task-1", success=True, output="Done")

        agent.execute = failing_execute
        meta_agent.register_agent(agent)

        task = AgentTask(id="task-1", type="task", input_data={})
        result = meta_agent.delegate_task(task)

        if should_succeed:
            assert result.success is True
        else:
            assert result.success is False

    def test_handle_agent_timeout(self):
        """Cover timeout handling for long-running agents."""
        meta_agent = AtomMetaAgent(task_timeout=1)  # 1 second timeout

        agent = MagicMock(id="agent-1", capabilities=["task"])

        async def slow_execute(*args, **kwargs):
            import asyncio
            await asyncio.sleep(2)  # Exceeds timeout
            return TaskResult(task_id="task-1", success=True, output="Done")

        agent.execute = slow_execute
        meta_agent.register_agent(agent)

        task = AgentTask(id="task-1", type="task", input_data={})
        result = meta_agent.delegate_task(task)

        assert result.success is False
        assert "timeout" in result.error.lower()
