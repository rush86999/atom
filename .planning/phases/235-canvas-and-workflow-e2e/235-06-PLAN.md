    def execute_workflow_via_ui(page: Page, workflow_id: str) -> None:
        """Execute workflow via UI."""
        page.goto(f"http://localhost:3001/workflows/{workflow_id}")
        page.click('[data-testid="workflow-execute-button"]')
        page.wait_for_selector('[data-testid="workflow-executing"]', timeout=5000)

    def wait_for_workflow_completion(page: Page, timeout: int = 30000) -> None:
        """Wait for workflow execution to complete."""
        page.wait_for_selector('[data-testid="workflow-execution-complete"]', timeout=timeout)

    def get_execution_progress(page: Page) -> int:
        """Get workflow execution progress percentage."""
        progress_text = page.locator('[data-testid="workflow-progress"]').text_content()
        return int(progress_text.replace('%', ''))

    def verify_execution_order(db_session, workflow_id: str, expected_order: list) -> bool:
        """Verify skills executed in expected order."""
        from core.models import WorkflowExecution

        executions = db_session.query(WorkflowExecution).filter_by(
            workflow_id=workflow_id
        ).order_by(WorkflowExecution.started_at).all()

        actual_order = [e.skill_id for e in executions]
        return actual_order == expected_order

    def create_test_workflow_with_skills(db_session, skill_count: int = 3) -> tuple:
        """Create test workflow with skills."""
        from core.models import Workflow, WorkflowSkill, WorkflowConnection
        import uuid

        workflow_id = str(uuid.uuid4())
        workflow = Workflow(
            id=workflow_id,
            name=f"Test Workflow {str(uuid.uuid4())[:8]}",
            user_id="test-user",
            status="active",
            created_at=datetime.utcnow()
        )
        db_session.add(workflow)

        skill_ids = []
        for i in range(skill_count):
            skill_id = f"test-skill-{i}-{str(uuid.uuid4())[:8]}"
            skill_ids.append(skill_id)

            workflow_skill = WorkflowSkill(
                id=str(uuid.uuid4()),
                workflow_id=workflow_id,
                skill_id=skill_id,
                position_x=100 * (i + 1),
                position_y=100,
                order=i
            )
            db_session.add(workflow_skill)

        # Create connections (linear chain)
        for i in range(skill_count - 1):
            connection = WorkflowConnection(
                id=str(uuid.uuid4()),
                workflow_id=workflow_id,
                from_skill=skill_ids[i],
                to_skill=skill_ids[i + 1]
            )
            db_session.add(connection)

        db_session.commit()
        return workflow_id, skill_ids
    ```

    from freezegun import freeze_time
    from datetime import datetime
    import requests

    def create_scheduled_trigger(db_session, workflow_id: str, cron_expression: str) -> str:
        """Create scheduled trigger for workflow."""
        from core.models import WorkflowTrigger
        import uuid

        trigger_id = str(uuid.uuid4())
        trigger = WorkflowTrigger(
            id=trigger_id,
            workflow_id=workflow_id,
            trigger_type="scheduled",
            config={"cron": cron_expression},
            enabled=True,
            created_at=datetime.utcnow()
        )
        db_session.add(trigger)
        db_session.commit()
        return trigger_id

    def create_webhook_trigger(db_session, workflow_id: str, event_type: str, filters: dict = None) -> tuple:
        """Create webhook trigger and return (trigger_id, webhook_url)."""
        from core.models import WorkflowTrigger
        import uuid

        trigger_id = str(uuid.uuid4())
        webhook_token = str(uuid.uuid4())
        webhook_url = f"http://localhost:8000/api/webhooks/{webhook_token}"

        config = {
            "event_type": event_type,
            "webhook_token": webhook_token,
            "filters": filters or {}
        }

        trigger = WorkflowTrigger(
            id=trigger_id,
            workflow_id=workflow_id,
            trigger_type="webhook",
            config=config,
            enabled=True,
            created_at=datetime.utcnow()
        )
        db_session.add(trigger)
        db_session.commit()
        return trigger_id, webhook_url

    def fire_scheduler_tick() -> None:
        """Trigger workflow scheduler to check for due workflows."""
        # Call scheduler endpoint or service method
        response = requests.post("http://localhost:8000/api/v1/workflows/scheduler/tick")
        assert response.status_code == 200

    def send_webhook_event(webhook_url: str, payload: dict) -> requests.Response:
        """Send webhook event to trigger workflow."""
        return requests.post(webhook_url, json=payload)

    def verify_workflow_execution_from_trigger(db_session, workflow_id: str, trigger_type: str) -> bool:
        """Verify workflow executed from specific trigger type."""
        from core.models import WorkflowExecution

        execution = db_session.query(WorkflowExecution).filter_by(
            workflow_id=workflow_id
        ).order_by(WorkflowExecution.started_at.desc()).first()

        return execution and execution.trigger_type == trigger_type
    ```
