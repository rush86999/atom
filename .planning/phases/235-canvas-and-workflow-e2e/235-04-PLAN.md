    def execute_skill_via_ui(page: Page, skill_id: str, parameters: dict) -> None:

        page.click(f'[data-testid="skill-{skill_id}-execute"]')
        page.wait_for_selector('[data-testid="skill-execute-modal"]', timeout=5000)

        # Fill parameters
        for param_name, param_value in parameters.items():
            page.fill(f'[data-testid="skill-param-{param_name}"]', str(param_value))

        # Submit
        page.click('[data-testid="skill-execute-confirm"]')
