    ```python
    def navigate_to_workflow_composer(page: Page):
        """Navigate to workflow composer page."""
        page.goto("http://localhost:3001/workflows/create")
        page.wait_for_load_state("networkidle")
        assert page.locator('[data-testid="workflow-composer"]').is_visible()

    def add_skill_to_workflow(page: Page, skill_id: str) -> None:
        """Add skill to workflow composer."""
        page.click('[data-testid="workflow-add-skill"]')
        page.wait_for_selector('[data-testid="skill-select-dropdown"]', timeout=5000)
        page.select_option('[data-testid="skill-select-dropdown"]', skill_id)
        page.click('[data-testid="skill-add-confirm"]')
        page.wait_for_selector(f'[data-testid="workflow-skill-{skill_id}"]', timeout=5000)

    def connect_skills(page: Page, from_skill: str, to_skill: str) -> None:
        """Connect two skills in workflow composer."""
        from_element = page.locator(f'[data-testid="workflow-skill-{from_skill}"] [data-testid="output-port"]')
        to_element = page.locator(f'[data-testid="workflow-skill-{to_skill}"] [data-testid="input-port"]')

        # Drag from output port to input port
        from_element.drag_to(to_element)

        # Wait for connection line to appear
        page.wait_for_selector('[data-testid^="workflow-connection-"]', timeout=3000)

    def save_workflow(page: Page, name: str, description: str = "") -> None:
        """Save workflow."""
        page.fill('[data-testid="workflow-name-input"]', name)
        if description:
            page.fill('[data-testid="workflow-description-input"]', description)
        page.click('[data-testid="workflow-save-button"]')
        page.wait_for_selector('[data-testid="workflow-saved"]', timeout=10000)

    def create_test_skills(db_session, count: int = 3) -> list:
        """Create test skills for workflow composition."""
        from core.models import SkillExecution
        import uuid

        skill_ids = []
        for i in range(count):
            skill_id = f"test-workflow-skill-{i}-{str(uuid.uuid4())[:8]}"
            skill = SkillExecution(
                id=str(uuid.uuid4()),
                skill_id=skill_id,
                agent_id="system",
                status="Active",
                capability=f"Test skill {i}",
                skill_body="# Test skill",
                created_at=datetime.utcnow()
            )
            db_session.add(skill)
            skill_ids.append(skill_id)

        db_session.commit()
        return skill_ids
    ```

    ```python
    def create_workflow_with_connections(page: Page, skills: list, connections: list) -> str:
        """Create workflow with specific skill connections."""
        # Add skills
        for skill_id in skills:

    def verify_dag_in_database(db_session, workflow_id: str) -> bool:
        """Verify workflow DAG stored correctly in database."""
        from core.models import Workflow, WorkflowConnection
        import networkx as nx

        workflow = db_session.query(Workflow).filter_by(id=workflow_id).first()
        connections = db_session.query(WorkflowConnection).filter_by(workflow_id=workflow_id).all()

        G = nx.DiGraph()
        G.add_nodes_from([s.skill_id for s in workflow.skills])
        G.add_edges_from([(c.from_skill, c.to_skill) for c in connections])

        return nx.is_directed_acyclic_graph(G)
    ```
