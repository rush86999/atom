    import requests
    import time

    def create_workflow_via_mobile_api(token: str, workflow_data: dict) -> dict:
        """Create workflow via mobile API."""
        response = requests.post(
            "http://localhost:8000/api/workflows",
            json=workflow_data,
            headers={"Authorization": f"Bearer {token}", "X-Platform": "mobile"}
        )
        assert response.status_code == 201
        return response.json()

    def execute_workflow_via_mobile_api(token: str, workflow_id: str) -> str:
        """Execute workflow via mobile API."""
        response = requests.post(
            f"http://localhost:8000/api/workflows/{workflow_id}/execute",
            headers={"Authorization": f"Bearer {token}", "X-Platform": "mobile"}
        )
        assert response.status_code == 200
        return response.json()['execution_id']

    def poll_workflow_execution(token: str, execution_id: str, timeout: int = 30) -> dict:
        """Poll workflow execution until complete."""
        start_time = time.time()
        while time.time() - start_time < timeout:
            response = requests.get(
                f"http://localhost:8000/api/workflows/executions/{execution_id}",
                headers={"Authorization": f"Bearer {token}", "X-Platform": "mobile"}
            )
            assert response.status_code == 200
            execution = response.json()
            if execution['status'] in ['complete', 'failed']:
                return execution
            time.sleep(1)
        raise TimeoutError(f"Workflow execution did not complete within {timeout}s")

    def create_test_workflow_data() -> dict:
        """Create test workflow data."""
        return {
            "name": f"Mobile Test Workflow {str(uuid.uuid4())[:8]}",

            ],
            "connections": [],

        }
