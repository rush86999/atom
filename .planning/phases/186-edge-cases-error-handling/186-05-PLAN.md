    ```bash
    # Combined coverage
    pytest tests/error_paths/ tests/boundary_conditions/ tests/failure_modes/ --cov=core --cov=api --cov-report=term-missing --cov-report=json:coverage_combined.json --cov-report=html:htmlcov_combined --override-ini="addopts=" -q > coverage_combined.txt 2>&1
    ```

    **Parse coverage results:**
    ```python
    import json
    import sys

    # Parse combined coverage
    with open('coverage_combined.json') as f:
        data = json.load(f)

    overall_coverage = data['totals']['percent_covered']
    files = data['files']

    # Find files below 75% target
    below_target = []
    for filename, file_data in files.items():
        if 'error_paths' in filename or 'boundary' in filename or 'failure' in filename:
            coverage = file_data['summary']['percent_covered']
            if coverage < 75:
                below_target.append((filename, coverage))

    print(f"Overall coverage: {overall_coverage:.2f}%")
    print(f"Files below 75% target: {len(below_target)}")
    for filename, coverage in below_target:
        print(f"  {filename}: {coverage:.2f}%")
    ```
