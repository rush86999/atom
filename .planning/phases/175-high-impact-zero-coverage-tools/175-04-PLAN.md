    with patch('api.canvas_routes.ws_manager') as mock_ws:
        mock_ws.broadcast = AsyncMock()

        response = client.post("/api/canvas/submit", json={
            "canvas_id": "canvas-123",
            "form_data": {
                "field1": "value1",
                "field2": "value2"
            }
        })

        assert response.status_code == 200
        data = response.json()
        assert data["success"] is True
        assert "submission_id" in data["data"]

        # Verify WebSocket broadcast
        mock_ws.broadcast.assert_called_once()
        call_args = mock_ws.broadcast.call_args
        assert "canvas:form_submitted" in call_args[1]

        # Verify CanvasAudit created
        audit = db_session.query(CanvasAudit).filter(
            CanvasAudit.canvas_id == "canvas-123"
        ).first()
        assert audit is not None
        assert audit.action == "submit"
        assert audit.agent_id is None  # No agent
