```python
def test_session_persists_across_refresh(authenticated_page):
    # Setup: authenticated_page has token in localStorage
    dashboard = DashboardPage(authenticated_page)
    dashboard.navigate()

    # Verify: Dashboard loads

    # Action: Refresh page

    # Verify: Still authenticated
    assert dashboard.is_loaded()
```
