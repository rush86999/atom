    ```python
    import json
    from datetime import datetime, timezone

    # Load coverage data
    with open('tests/coverage_reports/metrics/coverage_251.json') as f:
        coverage_data = json.load(f)

    # Business impact tiers
    TIER_SCORES = {"Critical": 10, "High": 7, "Medium": 5, "Low": 3}

    # High-impact files (>200 lines)
    HIGH_IMPACT_FILES = {
        "core/models.py": "Critical",
        "core/workflow_engine.py": "Critical",
        "core/agent_world_model.py": "Critical",
        "core/atom_agent_endpoints.py": "High",
        "core/episode_service.py": "Critical",
        "core/atom_meta_agent.py": "Critical",
        "core/llm/byok_handler.py": "Critical",
        "core/workflow_analytics_engine.py": "High",
        "core/agent_social_layer.py": "High",
        "core/episode_segmentation_service.py": "Critical",
        # Add more from analysis
    }

    gaps = []
    for file_path, file_data in coverage_data.get('files', {}).items():
        # Filter to backend/core files only
        if not file_path.startswith('backend/core/') and not file_path.startswith('core/'):
            continue

        summary = file_data.get('summary', {})
        if not summary:
            continue

        coverage_pct = summary.get('percent_covered', 0.0)
        if coverage_pct >= 70.0:
            continue

        total_lines = summary.get('num_statements', 0)
        covered_lines = summary.get('covered_lines', 0)
        uncovered_lines = total_lines - covered_lines

        # Determine tier
        filename = file_path.split('/')[-1]
        tier = HIGH_IMPACT_FILES.get(filename, "Medium")

        # Calculate priority
        tier_score = TIER_SCORES[tier]
        priority_score = (uncovered_lines * tier_score) / (coverage_pct + 1)

        gaps.append({
            "file": file_path,
            "coverage_pct": round(coverage_pct, 2),
            "total_lines": total_lines,
            "covered_lines": covered_lines,
            "uncovered_lines": uncovered_lines,
            "business_impact": tier,
            "tier_score": tier_score,
            "priority_score": round(priority_score, 2),
            "gap_to_target": round(70.0 - coverage_pct, 2)
        })

    # Sort by priority
    gaps.sort(key=lambda x: x['priority_score'], reverse=True)

    # Save gap analysis
    report = {
        "generated_at": datetime.now(timezone.utc).isoformat(timespec='seconds') + 'Z',
        "baseline_coverage": coverage_data['totals']['percent_covered'],
        "target_coverage": 70.0,
        "total_files_analyzed": len(gaps),
        "total_missing_lines": sum(g['uncovered_lines'] for g in gaps),
        "high_impact_gaps": [g for g in gaps if g['business_impact'] in ['Critical', 'High']][:50],
        "all_gaps": gaps
    }

    with open('tests/coverage_reports/backend_251_gap_analysis.json', 'w') as f:
        json.dump(report, f, indent=2)
    ```
