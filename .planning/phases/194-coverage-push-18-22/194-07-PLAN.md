   def test_submit_form_with_validation_errors(self, client: TestClient, db_session: Session):
       """Cover form submission with validation errors."""
       form_data = {
           "canvas_id": "test-form",
           "submission": {
               "email": "invalid-email",  # Invalid email format
               "message": ""

       assert response.status_code == 400 or response.status_code == 422

   @pytest.mark.parametrize("field_type,value,valid", [
       ("email", "test@example.com", True),
       ("email", "invalid", False),
       ("text", "Some text", True),
       ("number", "123", True),
       ("number", "abc", False),
   ])
   def test_form_field_validation(self, client: TestClient, db_session: Session, field_type, value, valid):
       """Cover validation for various field types."""
       form_data = {
           "canvas_id": "test-form",
           "submission": {f"field_{field_type}": value}
       }

       response = client.post("/api/canvas/form/submit", json=form_data)

       if valid:
           assert response.status_code == 200 or response.status_code == 202
       else:
           assert response.status_code in [400, 422]

   def test_submit_form_for_nonexistent_canvas(self, client: TestClient, db_session: Session):
