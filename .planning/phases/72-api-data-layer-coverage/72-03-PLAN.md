**Relationship Testing Pattern:**
```python
def test_agent_execution_relationship(self, db: Session):
    """Test AgentRegistry -> AgentExecution one-to-many."""
    agent = AgentFactory(_session=db)
    execution = AgentExecutionFactory(
        _session=db,
        agent_id=agent.id,
        status="running"
    )

    # Forward relationship
    loaded_execution = db.query(AgentExecution).filter_by(id=execution.id).first()
    assert loaded_execution.agent.id == agent.id

    # Backward relationship (backref)
    loaded_agent = db.query(AgentRegistry).filter_by(id=agent.id).first()
    assert len(loaded_agent.executions) == 1
```

**Constraint Testing Pattern:**
```python
def test_user_email_unique(self, db: Session):
    """Test email uniqueness constraint."""
    user1 = UserFactory(_session=db, email="test@example.com")

        user2 = UserFactory(_session=db, email="test@example.com")
        db.flush()

    db.rollback()
```
