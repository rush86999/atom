**Circular Dependency Detection Pattern:**
```python
def test_no_circular_dependencies():
    """Verify no circular dependencies in migration graph."""
    config = Config("alembic.ini")
    script = ScriptDirectory.from_config(config)

    migrations = list(script.walk_revisions(base="base", head="head"))
    graph = {rev.revision: rev.down_revision for rev in migrations}

    # Detect cycles using DFS
    visited = set()
    rec_stack = set()

    def dfs(node):
        visited.add(node)
        rec_stack.add(node)

        # Get parents (migrations that depend on this one)
        parents = [rev for rev, down in graph.items() if down == node]

        for parent in parents:
            if parent not in visited:
                if dfs(parent):
                    return True
            elif parent in rec_stack:
                return True

        rec_stack.remove(node)
        return False

    for node in graph:
        if node not in visited:
            assert not dfs(node), f"Circular dependency detected involving {node}"
```

from alembic import command
from sqlalchemy import create_engine, inspect

@pytest.mark.e2e
@pytest.mark.requires_docker
def test_single_migration_rollback(fresh_database):
    """Test single migration upgrade and downgrade."""
    # Get a specific recent migration
    config = Config("alembic.ini")
    config.set_main_option("sqlalchemy.url", DATABASE_URL)

    # Get current head
    from alembic.script import ScriptDirectory
    script = ScriptDirectory.from_config(config)
    head = script.get_current_head()

    # Get previous revision
    migrations = list(script.walk_revisions(base="base", head="head"))
    if len(migrations) >= 2:
        target = migrations[1].revision  # Second-to-last

        # Downgrade to target
        command.downgrade(config, target)

        # Verify schema
        inspector = inspect(fresh_database)
        tables_before = set(inspector.get_table_names())

        # Upgrade back to head
        command.upgrade(config, "head")

        # Verify schema matches
        inspector = inspect(fresh_database)
        tables_after = set(inspector.get_table_names())

        assert tables_before != tables_after, "Schema should change"
        # After upgrade, should have head schema
        assert "agent_registry" in tables_after
