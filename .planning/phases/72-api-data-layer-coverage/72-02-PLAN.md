**Time Mocking Pattern (using freezegun):**
```python
from freezegun import freeze_time

def test_access_token_expiration(self, client):
    with freeze_time("2026-01-01 12:00:00"):
        response = client.post("/api/auth/login", json={...})
        token = response.json()["access_token"]

    with freeze_time("2026-01-01 13:00:00"):  # 1 hour later
        # Token should still be valid
        response = client.get("/api/protected", headers={"Authorization": f"Bearer {token}"})
        assert response.status_code == 200

    with freeze_time("2026-01-01 14:00:00"):  # 2 hours later (expired)
        response = client.get("/api/protected", headers={"Authorization": f"Bearer {token}"})
        assert response.status_code == 401
```

    ws.send_text = AsyncMock()

    # Verify cleanup happened
