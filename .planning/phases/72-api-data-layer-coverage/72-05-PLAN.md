```python
def test_transaction_commit_success(self, db: Session):
    """Test data persists after commit."""
    agent = AgentFactory(_session=db, name="commit_test")
    db.commit()

    # Create new session to verify persistence
    from core.database import SessionLocal
    new_session = SessionLocal()
    try:
        retrieved = new_session.query(AgentRegistry).filter_by(
            name="commit_test"
        ).first()
        assert retrieved is not None
        assert retrieved.name == "commit_test"
    finally:
        new_session.close()

def test_transaction_rollback_on_error(self, db: Session):

        name="rollback_test"
    ).first()
    assert retrieved is None
```

```python
import threading
import time

def test_concurrent_create_same_agent_fails(self, db: Session):
    """Test concurrent creation of same agent fails."""
    agent_id = "concurrent_test_agent"
    results = []
    errors = []

    def create_agent():
        try:
            from core.database import SessionLocal
            local_db = SessionLocal()
            try:
                agent = AgentRegistry(
                    id=agent_id,
                    name="Concurrent Test",
                    status="student"
                )
                local_db.add(agent)
                local_db.commit()
                results.append("success")
            except IntegrityError as e:
                errors.append(str(e))
                local_db.rollback()
            finally:
                local_db.close()
        except Exception as e:
            errors.append(str(e))

    # Create two agents concurrently
    t1 = threading.Thread(target=create_agent)
    t2 = threading.Thread(target=create_agent)
    t1.start()
    t2.start()
    t1.join()
    t2.join()

    # Only one should succeed
    assert len(results) == 1, f"Expected 1 success, got {len(results)}"
    assert len(errors) == 1, f"Expected 1 error, got {len(errors)}"
```

```python
def test_generate_constraint_coverage_report():
    """Generate coverage report for constraints."""
    from sqlalchemy import inspect

    # Get all constraints from database
    inspector = inspect(engine)

    # Document foreign keys
    fk_report = []
    for table in Base.metadata.tables.keys():
        fks = inspector.get_foreign_keys(table)
        for fk in fks:
            fk_report.append({
                "table": table,
                "constrained_columns": fk["constrained_columns"],
                "referred_table": fk["referred_table"],
                "referred_columns": fk["referred_columns"],
                "on_delete": fk.get("on_delete", "NO ACTION")
            })

    # Print report
    print("\n=== Foreign Key Constraint Coverage ===")
    for fk in fk_report:
```
