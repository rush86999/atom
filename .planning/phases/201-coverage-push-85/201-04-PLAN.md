    async def test_get_location_intern_agent(self, device_tool, intern_agent):
        """Test INTERN agent can access location."""
        with patch.object(device_tool.governance_service, 'check_action_permission') as mock_gov:
            mock_gov.return_value = Mock(allowed=True)

            with patch('tools.device_tool.geocoder') as mock_geocoder:
                mock_geocoder.ip.return_value = Mock(
                    latitude=37.7749,

                    city="San Francisco"

                    agent_id=intern_agent.id,
