```bash
cd /Users/rushiparikh/projects/atom/backend

# Run full test suite with coverage
python3 -m pytest --cov=backend --cov-branch --cov-report=json --cov-report=term-missing --cov-report=html -q

# Save final report
cp coverage.json coverage_final.json

# Generate summary stats
python3 << 'EOF'
import json

with open('coverage_final.json') as f:
    data = json.load(f)

totals = data['totals']
print(f"\n=== FINAL COVERAGE ===")
print(f"Overall: {totals['percent_covered']:.2f}%")
print(f"Lines: {totals['covered_lines']:,} / {totals['num_statements']:,}")
print(f"Branch: {totals['covered_branches']:,} / {totals['num_branches']:,}")
print(f"Missing: {totals['missing_lines']:,} lines")

# Module breakdown
modules = {}
for path, info in data['files'].items():
    if '/core/' in path:
        module = 'core'
    elif '/api/' in path:
        module = 'api'
    elif '/tools/' in path:
        module = 'tools'
    elif '/cli/' in path:
        module = 'cli'
    else:
        continue

    if module not in modules:
        modules[module] = {'files': 0, 'covered': 0, 'total': 0}

    modules[module]['files'] += 1
    modules[module]['covered'] += info['summary']['covered_lines']
    modules[module]['total'] += info['summary']['num_statements']

print(f"\n=== MODULE BREAKDOWN ===")
for module, stats in sorted(modules.items()):
    pct = (stats['covered'] / stats['total'] * 100) if stats['total'] > 0 else 0
    print(f"{module:10s}: {pct:5.1f}% ({stats['covered']:,}/{stats['total']:,}, {stats['files']} files)")
EOF
```
