import pytest
from datetime import datetime

class TestValidateAgentId:
    """Test agent ID validation."""

    def test_valid_agent_id(self):
        """Test valid agent ID formats."""
        valid_ids = [
            "agent-123",
            "test_agent_v1",
            "my-agent-2023",
            "a" * 50,  # Max length
        ]
        for agent_id in valid_ids:
            result = validate_agent_id(agent_id)
            assert result["valid"] is True

    def test_invalid_agent_id(self):
        """Test invalid agent ID formats."""
        invalid_ids = [
            "",  # Empty
            "agent spaces",  # Spaces
            "agent@123",  # Special chars
            "a" * 51,  # Too long
            None,  # None
        ]
        for agent_id in invalid_ids:
            result = validate_agent_id(agent_id)
            assert result["valid"] is False
            assert "error" in result

    @pytest.mark.parametrize("agent_id,expected", [
        ("valid-id", True),
        ("", False),
        ("invalid@id", False),
        (None, False),
    ])
    def test_validate_agent_id_parametrized(self, agent_id, expected):
        """Parametrized test for agent ID validation."""
        result = validate_agent_id(agent_id)
        assert result["valid"] == expected


class TestCalculateConfidence:
    """Test confidence score calculation."""

    def test_calculate_confidence_from_history(self):
        """Test confidence calculation from execution history."""
        history = [
            {"success": True, "accuracy": 0.9},
            {"success": True, "accuracy": 0.85},
            {"success": True, "accuracy": 0.95},
        ]
        result = calculate_confidence(history)
        assert result["confidence"] > 0.85
        assert result["sample_size"] == 3

    def test_calculate_confidence_empty_history(self):
        """Test confidence with no history."""
        result = calculate_confidence([])
        assert result["confidence"] == 0.5  # Default
        assert result["sample_size"] == 0

    def test_calculate_confidence_with_failures(self):
        """Test confidence calculation with mixed success/failure."""
        history = [
            {"success": True, "accuracy": 0.9},
            {"success": False, "error": "timeout"},
            {"success": True, "accuracy": 0.8},
        ]
        result = calculate_confidence(history)
        assert 0.4 < result["confidence"] < 0.7  # Reduced by failure

    @pytest.mark.parametrize("success_rate,expected_confidence", [
        (1.0, 0.9),  # All success
        (0.5, 0.5),  # Half success
        (0.0, 0.1),  # All failure
    ])
    def test_confidence_by_success_rate(self, success_rate, expected_confidence):
        """Test confidence correlates with success rate."""
        history = []
        for i in range(10):
            history.append({
                "success": i < (10 * success_rate),
                "accuracy": 0.8 if i < (10 * success_rate) else 0.0
            })
        result = calculate_confidence(history)
        assert abs(result["confidence"] - expected_confidence) < 0.2


class TestFormatAgentResponse:
    """Test agent response formatting."""

    def test_format_basic_response(self):
        """Test basic response formatting."""
        response = format_agent_response(
            agent_id="test-agent",
            content="Hello, world!",
            metadata={"source": "test"}
        )
        assert response["agent_id"] == "test-agent"
        assert response["content"] == "Hello, world!"
        assert response["metadata"]["source"] == "test"

    def test_format_response_with_timestamp(self):
        """Test response formatting with timestamp."""
        before = datetime.now()
        response = format_agent_response(
            agent_id="test-agent",
            content="Test",
            include_timestamp=True
        )
        after = datetime.now()

        assert "timestamp" in response
        assert before <= datetime.fromisoformat(response["timestamp"]) <= after

    def test_format_response_sanitizes_content(self):
        """Test that response content is sanitized."""
        response = format_agent_response(
            agent_id="test-agent",
            content="<script>alert('xss')</script>Hello"
        )
        assert "<script>" not in response["content"]

    @pytest.mark.parametrize("content,expected_safe", [
        ("Plain text", True),
        ("<b>Bold</b>", True),
        ("<script>alert('xss')</script>", False),
        ("javascript:alert('xss')", False),
    ])
    def test_content_sanitization(self, content, expected_safe):
        """Test content sanitization for XSS."""
        response = format_agent_response(
            agent_id="test-agent",
            content=content
        )
        if expected_safe:
            assert content in response["content"] or "<b>" in response["content"]
        else:
            assert "<script>" not in response.get("content", "")


class TestSanitizeAgentInput:
    """Test input sanitization utilities."""

    def test_sanitize_string_input(self):
        """Test sanitizing string input."""
        result = sanitize_agent_input("Hello <script>alert('xss')</script> World")
        assert "<script>" not in result
        assert "Hello" in result
        assert "World" in result

    def test_sanitize_dict_input(self):
        """Test sanitizing dictionary input."""
        input_data = {
            "message": "Hello <script>alert('xss')</script>",
            "count": 5,
            "nested": {"value": "<img src=x onerror=alert('xss')>"}
        }
        result = sanitize_agent_input(input_data)
        assert "<script>" not in result["message"]
        assert "onerror" not in str(result["nested"])

    def test_sanitize_preserves_safe_content(self):
        """Test that safe content is preserved."""
        result = sanitize_agent_input("Hello, world! 123")
        assert result == "Hello, world! 123"

        ("autonomous", AgentStatus.AUTONOMOUS),

    ])

        """Test parsing None as maturity level."""
