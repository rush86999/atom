Create detailed analysis comparing Wave 2 results to baseline:

```python
# Create analysis script
import json

# Load baseline
with open('backend/final_coverage.json') as f:
    baseline = json.load(f)

# Load wave 2
with open('backend/coverage_wave_2.json') as f:
    wave2 = json.load(f)

# Calculate changes
baseline_pct = baseline['totals']['percent_covered']
wave2_pct = wave2['totals']['percent_covered']
improvement = wave2_pct - baseline_pct

print(f"Baseline Coverage: {baseline_pct:.2f}%")
print(f"Wave 2 Coverage: {wave2_pct:.2f}%")
print(f"Improvement: +{improvement:.2f} percentage points")

# Module breakdown
modules = {}
for path, info in wave2['files'].items():
    if '/core/' in path:
        module = 'core'
    elif '/api/' in path:
        module = 'api'
    elif '/tools/' in path:
        module = 'tools'
    elif '/cli/' in path:
        module = 'cli'
    else:
        continue

    if module not in modules:
        modules[module] = {'covered': 0, 'total': 0}

    modules[module]['covered'] += info['summary']['covered_lines']
    modules[module]['total'] += info['summary']['num_statements']

print("\nModule Coverage:")
for module, stats in sorted(modules.items()):
    pct = (stats['covered'] / stats['total'] * 100) if stats['total'] > 0 else 0
    print(f"  {module}: {pct:.1f}% ({stats['covered']:,}/{stats['total']:,})")
```
