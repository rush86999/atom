    # backend/tests/scripts/compare_models_coverage.py
    import json

    # Load baseline (from phase_127_baseline.json)
    with open("tests/coverage_reports/metrics/phase_127_baseline.json") as f:
        baseline = json.load(f)

    # Load new models coverage
    with open("tests/coverage_reports/metrics/phase_127_models_coverage.json") as f:
        new_coverage = json.load(f)

    # Extract models.py coverage from baseline
    baseline_models = baseline["files"].get("backend/core/models.py", {})
    new_models = new_coverage["files"].get("backend/core/models.py", {})

    baseline_pct = baseline_models.get("summary", {}).get("percent_covered", 0)
    new_pct = new_models.get("summary", {}).get("percent_covered", 0)

    improvement = new_pct - baseline_pct

    result = {
        "file": "core/models.py",
        "baseline_coverage": baseline_pct,
        "new_coverage": new_pct,
        "improvement": round(improvement, 2),
        "lines_covered_before": baseline_models.get("summary", {}).get("covered_lines", 0),
        "lines_covered_after": new_models.get("summary", {}).get("covered_lines", 0),
        "lines_added": new_models.get("summary", {}).get("covered_lines", 0) - baseline_models.get("summary", {}).get("covered_lines", 0),
    }

    # Write result
    with open("tests/coverage_reports/metrics/phase_127_models_improvement.json", "w") as f:
        json.dump(result, f, indent=2)

    print(f"core/models.py coverage: {baseline_pct:.2f}% → {new_pct:.2f}% ({improvement:+.2f} pp)")
    print(f"Lines covered: {result['lines_covered_before']} → {result['lines_covered_after']} ({result['lines_added']:+d})")
