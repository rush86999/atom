    """
    Property-based coverage tests for core/workflow_engine.py.

    These tests use Hypothesis to validate workflow engine invariants:
    - DAGs remain acyclic
    - Topological sort produces valid ordering
    - Execution order respects dependencies
    - Parallel steps are independent
    """
    import pytest
    from hypothesis import given, strategies as st, assume, settings
    from hypothesis.stateful import RuleBasedStateMachine, rule, invariant
    from typing import List, Dict, Set, Optional
    from collections import defaultdict

    # Try to import workflow_engine components
    try:
        from core.workflow_engine import (
            WorkflowEngine,
            WorkflowDefinition,
            WorkflowStep,
            WorkflowExecution,
            WorkflowDAG,
            ValidationError,
            ExecutionError
        )
        WORKFLOW_AVAILABLE = True
    except ImportError:
        WORKFLOW_AVAILABLE = False
        pytest.skip("workflow_engine not available", allow_module_level=True)

    # Helper strategies
    @st.composite
    def workflow_steps(draw):
        """Generate valid workflow steps."""
        step_names = draw(st.lists(
            st.text(min_size=1, max_size=10).filter(lambda x: x.isalnum()),
            min_size=1,
            max_size=10,
            unique=True
        ))

        steps = []
        for name in step_names:
            step = WorkflowStep(
                id=name,
                action="test_action",
                inputs={"param": draw(st.integers(min_value=0, max_value=100))},
                dependencies=draw(st.lists(
                    st.sampled_from(step_names),
                    min_size=0,
                    max_size=3,
                    unique=True
                ).filter(lambda deps: name not in deps))  # No self-dependencies
            )
            steps.append(step)

        return steps

    @st.composite
    def valid_dag(draw):
        """Generate a valid DAG (no cycles)."""
        num_nodes = draw(st.integers(min_value=1, max_value=10))
        nodes = [f"step_{i}" for i in range(num_nodes)]

        # Create edges that don't form cycles
        edges = set()
        for i in range(num_nodes):
            for j in range(i + 1, num_nodes):
                if draw(st.floats(min_value=0, max_value=1) < 0.3):  # 30% edge probability
                    edges.add((nodes[i], nodes[j]))

        return {"nodes": nodes, "edges": list(edges)}

    class TestWorkflowDAGCoverage:
        """Property tests for WorkflowDAG."""

        @given(valid_dag())
        @settings(max_examples=50)
        def test_dag_is_acyclic(self, dag_data):
            """Property: Generated DAG should be acyclic."""
            if not WORKFLOW_AVAILABLE:
                pytest.skip("workflow_engine not available")

            dag = WorkflowDAG()
            for node in dag_data["nodes"]:
                dag.add_node(node)

            for src, dst in dag_data["edges"]:
                dag.add_edge(src, dst)

            # Verify acyclic
            assert dag.is_acyclic(), f"DAG should be acyclic: {dag_data}"

        @given(valid_dag())
        @settings(max_examples=50)
        def test_topological_sort_is_valid(self, dag_data):
            """Property: Topological sort produces valid ordering."""
            if not WORKFLOW_AVAILABLE:
                pytest.skip("workflow_engine not available")

            dag = WorkflowDAG()
            for node in dag_data["nodes"]:
                dag.add_node(node)

            for src, dst in dag_data["edges"]:
                dag.add_edge(src, dst)

            sorted_nodes = dag.topological_sort()

            # All nodes should be in result
            assert set(sorted_nodes) == set(dag_data["nodes"])

            # Dependencies should come before dependents
            node_index = {node: i for i, node in enumerate(sorted_nodes)}
            for src, dst in dag_data["edges"]:
                assert node_index[src] < node_index[dst], \
                    f"{src} should come before {dst}"

        @given(valid_dag())
        @settings(max_examples=30)
        def test_dag_cycle_detection(self, dag_data):
            """Property: Adding cycle edge should raise error."""
            if not WORKFLOW_AVAILABLE:
                pytest.skip("workflow_engine not available")

            dag = WorkflowDAG()
            for node in dag_data["nodes"]:
                dag.add_node(node)

            for src, dst in dag_data["edges"]:
                dag.add_edge(src, dst)

            # Try to add a cycle (reverse an existing edge)
            if dag_data["edges"]:
                # Reverse first edge to create cycle
                src, dst = dag_data["edges"][0]
                with pytest.raises(ValidationError):
                    dag.add_edge(dst, src)

    class TestWorkflowExecutionCoverage:
        """Property tests for workflow execution."""

        @given(workflow_steps())
        @settings(max_examples=30)
        def test_execution_order_respects_dependencies(self, steps):
            """Property: Execution order respects step dependencies."""
            if not WORKFLOW_AVAILABLE:
                pytest.skip("workflow_engine not available")

            workflow = WorkflowDefinition(
                id="test_workflow",
                name="Test Workflow",
                steps=steps
            )

            engine = WorkflowEngine()
            execution = engine.execute(workflow)

            # Verify execution order
            executed_order = execution.get_execution_order()

            # Create step lookup
            step_map = {s.id: s for s in steps}

            # Each step should come after its dependencies
            step_position = {step_id: i for i, step_id in enumerate(executed_order)}

            for step in steps:
                for dep in step.dependencies:
                    if dep in step_position and step.id in step_position:
                        assert step_position[dep] < step_position[step.id], \
                            f"Dependency {dep} should execute before {step.id}"

        @given(workflow_steps())
        @settings(max_examples=30)
        def test_parallel_steps_are_independent(self, steps):
            """Property: Parallel steps have no dependencies between them."""
            if not WORKFLOW_AVAILABLE:
                pytest.skip("workflow_engine not available")

            # Find steps with no mutual dependencies
            parallel_groups = []
            for i, step1 in enumerate(steps):
                for j, step2 in enumerate(steps[i+1:], i+1):
                    # Check if independent
                    step1_deps = set(step1.dependencies)
                    step2_deps = set(step2.dependencies)

                    if (step2.id not in step1_deps and
                        step1.id not in step2_deps):
                        parallel_groups.append((step1.id, step2.id))

            # These should be executable in parallel
            workflow = WorkflowDefinition(
                id="test_workflow",
                name="Test Workflow",
                steps=steps
            )

            engine = WorkflowEngine()
            execution = engine.execute(workflow)

            # Verify parallel steps were executed
            executed = set(execution.get_execution_order())
            for step1_id, step2_id in parallel_groups:
                assert step1_id in executed
                assert step2_id in executed
