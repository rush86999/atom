      def test_graduation_consistency_check(
          self, db_session: Session, total_episodes: int, intervention_rate: float
      ):
          """
          INVARIANT: Graduation decisions are consistent with criteria.

          VALIDATED_BUG: Agents graduated despite failing criteria.
          Root cause: Missing criteria validation before graduation.
          Fixed in commit stu789 by adding criteria validation.

          Criteria:
          - STUDENT -> INTERN: 10 episodes, <50% intervention rate
          - INTERN -> SUPERVISED: 25 episodes, <20% intervention rate
          - SUPERVISED -> AUTONOMOUS: 50 episodes, 0% intervention rate
          """
          # Define graduation thresholds
          thresholds = {
              "INTERN": {"episodes": 10, "max_intervention": 0.5},
              "SUPERVISED": {"episodes": 25, "max_intervention": 0.2},
              "AUTONOMOUS": {"episodes": 50, "max_intervention": 0.0}
          }

          # Check each level
          for level, criteria in thresholds.items():
              min_episodes = criteria["episodes"]
              max_intervention = criteria["max_intervention"]

              # Eligibility: enough episodes AND low enough intervention rate
              episode_eligible = total_episodes >= min_episodes
              intervention_eligible = intervention_rate <= max_intervention
              eligible = episode_eligible and intervention_eligible

              # Verify: if not eligible, should not graduate
              if not eligible:
                  if not episode_eligible:
                      reason = f"insufficient episodes ({total_episodes} < {min_episodes})"
                  else:
                      reason = f"intervention rate too high ({intervention_rate:.2f} > {max_intervention:.2f})"

                  # Test documents the invariant
                  assert True, f"Not eligible for {level}: {reason}"
