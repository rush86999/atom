    Script structure:
    ```python
    #!/usr/bin/env python3
    """
    Backend Coverage Gate Enforcement

    Enforces actual line coverage (from pytest --cov-branch) against progressive
    thresholds (70% → 75% → 80%). NOT service-level estimates.

    Usage:
        python backend_coverage_gate.py [--phase phase_1] [--format text]
    """
    import argparse
    import json
    import os
    import sys
    from pathlib import Path

    COVERAGE_JSON = Path(__file__).parent.parent / "coverage_reports" / "metrics" / "coverage.json"
    THRESHOLDS = {"phase_1": 70.0, "phase_2": 75.0, "phase_3": 80.0}

    def load_coverage():
        with open(COVERAGE_JSON) as f:
            return json.load(f)

    def check_gate(threshold: float, bypass: bool = False) -> tuple[bool, float]:
        data = load_coverage()
        coverage = data["totals"]["percent_covered"]
        passed = coverage >= threshold or bypass
        return passed, coverage

    def main():
        parser = argparse.ArgumentParser()
        parser.add_argument("--phase", default=os.getenv("COVERAGE_PHASE", "phase_1"))
        parser.add_argument("--format", choices=["text", "json"], default="text")
        parser.add_argument("--bypass", action="store_true", help="Emergency bypass")
        args = parser.parse_args()

        threshold = THRESHOLDS[args.phase]
        bypass = args.bypass or os.getenv("EMERGENCY_COVERAGE_BYPASS", "false").lower() == "true"
        passed, coverage = check_gate(threshold, bypass)

        gap = threshold - coverage
        if args.format == "text":
            print(f"Backend Coverage: {coverage:.2f}%")
            print(f"Threshold ({args.phase}): {threshold:.2f}%")
            print(f"Gap: {gap:+.2f}%")
            if bypass:
                print("⚠️  BYPASS ACTIVE (emergency mode)")
            print(f"Status: {'✓ PASS' if passed else '✗ FAIL'}")

        sys.exit(0 if passed else 1)

    if __name__ == "__main__":
        main()
    ```
