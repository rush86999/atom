    Add the submit endpoint to backend/api/canvas_routes.py (after StartRecordingRequest, before context management section):

    ```python
    # ============================================================================
    # Canvas Submission
    # ============================================================================

    @router.post("/submit")
    async def submit_canvas(
        request: CanvasSubmitRequest,
        db: Session = Depends(get_db),
        current_user: User = Depends(get_current_user)
    ) -> Dict[str, Any]:
        """
        Submit form data for a canvas.

        Validates authentication, required fields, and governance permissions.
        """
        # Governance check if agent_id provided
        if request.agent_id:
            governance = AgentGovernanceService(db)
            check = governance.can_perform_action(
                agent_id=request.agent_id,
                action_type="canvas_submit",
                action_complexity=3  # Form submission is complexity 3
            )

            if not check.get("allowed", True):
                return router.error_response(
                    error_code="GOVERNANCE_DENIED",
                    message=check.get("reason", "Permission denied"),
                    status_code=403
                )

        # TODO: Process form submission, save to database, etc.
        # For now, return success
        return router.success_response(
            data={
                "canvas_id": request.canvas_id,
                "submitted": True,
                "timestamp": datetime.now(timezone.utc).isoformat()
            }
        )
    ```

    This implements:
    - Authentication via get_current_user dependency
    - Request validation via CanvasSubmitRequest DTO
    - Governance check for agent operations
    - Error responses for auth/governance failures
    - Success response on valid submission
