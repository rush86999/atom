       @pytest.mark.asyncio
       async def test_postgres_concurrent_writes(postgres_engine):
           """
           POSTGRES: True concurrent writes (SQLite serializes these).
           Verifies PostgreSQL allows multiple writers simultaneously.
           """
           from sqlalchemy.ext.asyncio import create_async_engine, AsyncSession
           from core.models import AgentRegistry
           import uuid
           import asyncio

           # Async engine for true concurrency
           async_engine = create_async_engine(
               os.getenv("POSTGRES_TEST_URL").replace("postgresql://", "postgresql+asyncpg://")
           )

           async def create_agent(session, idx):
               async with AsyncSession(async_engine) as sess:
                   agent = AgentRegistry(
                       id=str(uuid.uuid4()),
                       name=f"Agent{idx}",
                       category="test",
                       module_path="test",
                       class_name="Test",
                       status=AgentStatus.STUDENT
                   )
                   sess.add(agent)
                   await sess.commit()

           # Launch 10 concurrent writes
           tasks = [create_agent(None, i) for i in range(10)]
           await asyncio.gather(*tasks)

           # Verify all 10 agents created
           async with AsyncSession(async_engine) as sess:
               result = await sess.execute(select(AgentRegistry))
               agents = result.scalars().all()
               assert len(agents) == 10
       ```

       def test_postgres_deadlock_detection(two_postgres_sessions):
           """
           POSTGRES: Deadlock detection with automatic rollback.
           PostgreSQL detects deadlocks and rolls back one transaction.
           """
           from core.models import AgentRegistry
           import uuid
           import time

           s1, s2 = two_postgres_sessions

           # Create two agents
           agent1 = AgentRegistry(
               id=str(uuid.uuid4()), name="Agent1", category="test",
               module_path="test", class_name="Test",
               status=AgentStatus.STUDENT, confidence_score=0.5
           )
           agent2 = AgentRegistry(
               id=str(uuid.uuid4()), name="Agent2", category="test",
               module_path="test", class_name="Test",
               status=AgentStatus.STUDENT, confidence_score=0.5
           )
           s1.add(agent1)
           s1.add(agent2)
           s1.commit()

           # Create deadlock: update in opposite order
           s1.query(AgentRegistry).filter_by(name="Agent1").update({"confidence_score": 0.7})
           time.sleep(0.01)  # Ensure locking happens

           s2.query(AgentRegistry).filter_by(name="Agent2").update({"confidence_score": 0.8})
           time.sleep(0.01)

           # Try to complete the deadlock cycle
           s1.query(AgentRegistry).filter_by(name="Agent2").update({"confidence_score": 0.7})

           try:
               s1.commit()
           except Exception as e:
               if "deadlock" in str(e).lower():
                   # Expected: PostgreSQL detected deadlock
                   s1.rollback()

           # Verify one transaction succeeded
           s2.commit()
       ```
