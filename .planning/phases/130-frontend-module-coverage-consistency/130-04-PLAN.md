# Plan 130-04: Core Feature Component Tests

**Objective:** Create comprehensive test suites for core feature components (Agents, Voice, Workflow, Calendar, etc.), achieving 80%+ coverage using React Testing Library and property-based testing.

**Context from Research (130-RESEARCH.md):**
- Core feature components (Agents, Voice, Workflow, Calendar) need comprehensive testing
- fast-check 4.5.3 already installed for property-based testing (Phase 108)
- React Testing Library patterns: test user behavior, not implementation
- Target threshold: 80-85% for core features

**Input from 130-02:**
- Gap analysis identifying core feature components below threshold
- Test inventory with effort estimates (2-3 test suites per component)
- Prioritized test plan for core modules

## Execution Context

**Primary Output:** Core feature test suites in component `__tests__` directories and `tests/property/`

**Key Decisions:**
1. Test user interactions (clicks, typing, form submissions) using `@testing-library/user-event`
2. Use property-based testing for state invariants (fast-check)
3. Test state management (Redux, React Context) with render patterns
4. Parallel execution with 130-03 (integration tests) - no dependencies

**Test Coverage Strategy:**
- **User Interactions:** Click events, form inputs, keyboard navigation
- **State Management:** Redux actions, Context updates, local state changes
- **Lifecycle:** Mount, update, unmount behavior
- **Property-Based Tests:** Invariants for state transformations (idempotency, commutativity)
- **Accessibility:** ARIA roles, keyboard navigation, screen reader support

// tests/property/agent-state-machine.test.ts
import fc from 'fast-check';
import { describe, it, expect } from '@jest/globals';
import { agentStateMachine, AgentState, AgentEvent } from '../../components/Agents/agentStateMachine';

describe('Agent State Machine - Property-Based Tests', () => {
  describe('State Transitions', () => {
    it('should always reach terminal state from any state', () => {
      fc.assert(
        fc.property(
          fc.array(fc.oneof(
            fc.constant('START'),
            fc.constant('STOP'),
            fc.constant('PAUSE'),
            fc.constant('ERROR'),
            fc.constant('COMPLETE')
          ), { minLength: 1, maxLength: 20 }),
          (events) => {
            let state: AgentState = { status: 'idle' };

            for (const event of events) {
              state = agentStateMachine(state, { type: event });
            }

            // Should be able to handle any sequence of events
            expect(state).toBeDefined();
            expect(typeof state.status).toBe('string');
          }
        )
      );
    });

    it('should be idempotent for pause events', () => {
      fc.assert(
        fc.property(
          fc.constantFrom('running', 'paused'),
          (initialStatus) => {
            const initialState: AgentState = { status: initialStatus };
            const once = agentStateMachine(initialState, { type: 'PAUSE' });
            const twice = agentStateMachine(once, { type: 'PAUSE' });

            expect(once.status).toBe(twice.status);
          }
        )
      );
    });

    it('should preserve state data during transitions', () => {
      fc.assert(
        fc.property(
          fc.record({
            id: fc.string(),
            name: fc.string(),
            config: fc.record({
              maxIterations: fc.nat({ max: 1000 }),
              timeout: fc.nat({ max: 300000 }),
            }),
          }),
          (agentData) => {
            let state: AgentState = {
              status: 'idle',
              data: agentData,
            };

            state = agentStateMachine(state, { type: 'START' });
            state = agentStateMachine(state, { type: 'STOP' });

            expect(state.data).toEqual(agentData);
          }
        )
      );
    });
  });

  describe('Execution Context', () => {
    it('should maintain execution count invariant', () => {
      fc.assert(
        fc.property(
          fc.array(fc.oneof(
            fc.constant('START'),
            fc.constant('COMPLETE')
          ), { minLength: 2, maxLength: 20 }),
          (events) => {
            let state: AgentState = { status: 'idle', executionCount: 0 };
            let startCount = 0;
            let completeCount = 0;

            for (const event of events) {
              state = agentStateMachine(state, { type: event });
              if (event === 'START') startCount++;
              if (event === 'COMPLETE') completeCount++;
            }

            // Execution count should equal START events
            expect(state.executionCount).toBe(startCount);
            // Execution count should be >= complete count
            expect(state.executionCount).toBeGreaterThanOrEqual(completeCount);
          }
        )
      );
    });
  });
});

            results: [[{ transcript: 'create new agent' }]],

          callback({ error: 'no-speech' });

      await user.click(micButton);

        expect(screen.getByText(/no speech detected/i)).toBeInTheDocument();

      const user = userEvent.setup();

      const micButton = screen.getByRole('button', { name: /start listening/i });
