# Plan 130-03: Integration Component Tests

**Objective:** Create comprehensive test suites for 36 integration components currently at 0% coverage, achieving 70%+ coverage for the integrations module.

**Context from Research (130-RESEARCH.md):**
- 36 integration components at 0% coverage (highest priority gap from 130-02)
- MSW (Mock Service Worker) already configured at `tests/mocks/handlers.ts`
- React Testing Library patterns for component testing (test behavior, not implementation)
- Target threshold: 70% for integrations (graduated rollout, ramp to 80% in Phase 131+)

**Input from 130-02:**
- Prioritized gap list identifying all 36 integration components
- Test inventory with effort estimates (3-5 test suites per component)
- Gap analysis report (130-02-GAPS.md) with CRITICAL priority classification

## Execution Context

**Primary Output:** Integration component test suites in `components/integrations/**/__tests__/`

**Key Decisions:**
1. Co-locate tests with components using `__tests__` directories
2. Use MSW for API mocking (network-level mocking, more realistic than component mocks)
3. Test user flows: connection, authentication, data sync, error handling
4. Parallel execution with 130-04 (core feature tests) - no dependencies between plans

**Test Coverage Strategy:**
- **OAuth Flows:** Test connection initiation, redirect handling, token storage
- **API Integration:** Test data fetching, loading states, error handling
- **User Actions:** Test form submissions, configuration updates, disconnect flows
- **Edge Cases:** Test network failures, invalid credentials, timeout scenarios

## Technical Approach

### 1. Test Suite Template
Base template for integration component tests:

```typescript
// components/integrations/__tests__/IntegrationComponent.test.tsx
import { render, screen, waitFor } from '@testing-library/react';
import userEvent from '@testing-library/user-event';
import { server } from '../../../tests/mocks/handlers';
import { rest } from 'msw';
import { IntegrationComponent } from '../IntegrationComponent';

describe('Integration Component', () => {
  beforeEach(() => {
    // Reset MSW handlers before each test
    server.resetHandlers();
  });

  afterEach(() => {
    // Clean up after each test
    jest.clearAllMocks();
  });

  describe('Connection Form', () => {
    it('renders connection form when not authenticated', () => {
      render(<IntegrationComponent />);
      expect(screen.getByLabelText(/instance url/i)).toBeInTheDocument();
      expect(screen.getByRole('button', { name: /connect/i })).toBeInTheDocument();
    });

    it('validates URL input format', async () => {
      const user = userEvent.setup();
      render(<IntegrationComponent />);

      const urlInput = screen.getByLabelText(/instance url/i);
      const connectButton = screen.getByRole('button', { name: /connect/i });

      // Test invalid URL
      await user.type(urlInput, 'not-a-valid-url');
      await user.click(connectButton);

      await waitFor(() => {
        expect(screen.getByText(/please enter a valid url/i)).toBeInTheDocument();
      });
    });

    it('initiates OAuth connection on valid submit', async () => {
      const user = userEvent.setup();
      const mockOnConnect = jest.fn();

      // MSW mock for successful connection initiation
      server.use(
        rest.post('/api/integrations/example/connect', (req, res, ctx) => {
          return res(ctx.status(200), ctx.json({ authUrl: 'https://example.com/oauth' }));
        })
      );

      render(<IntegrationComponent onConnect={mockOnConnect} />);

      const urlInput = screen.getByLabelText(/instance url/i);
      const connectButton = screen.getByRole('button', { name: /connect/i });

      await user.type(urlInput, 'https://example.atlassian.net');
      await user.click(connectButton);

      await waitFor(() => {
        expect(mockOnConnect).toHaveBeenCalledWith(expect.objectContaining({
          url: 'https://example.atlassian.net'
        }));
      });
    });
  });

  describe('OAuth Flow', () => {
    it('handles successful OAuth callback', async () => {
      // MSW mock for OAuth callback
      server.use(
        rest.get('/api/integrations/example/callback', (req, res, ctx) => {
          return res(ctx.status(200), ctx.json({
            status: 'connected',
            workspace: 'Test Workspace'
          }));
        })
      );

      render(<IntegrationComponent />);

      // Simulate OAuth callback
      window.dispatchEvent(new CustomEvent('oauth-callback', {
        detail: { code: 'test-code', state: 'test-state' }
      }));

      await waitFor(() => {
        expect(screen.getByText(/successfully connected/i)).toBeInTheDocument();
      });
    });

    it('handles OAuth authorization denied', async () => {
      server.use(
        rest.get('/api/integrations/example/callback', (req, res, ctx) => {
          return res(ctx.status(401), ctx.json({
            error: 'access_denied'
          }));
        })
      );

      render(<IntegrationComponent />);

      window.dispatchEvent(new CustomEvent('oauth-callback', {
        detail: { error: 'access_denied' }
      }));

      await waitFor(() => {
        expect(screen.getByText(/authorization was denied/i)).toBeInTheDocument();
      });
    });
  });

  describe('Error Handling', () => {
    it('displays error message on network failure', async () => {
      const user = userEvent.setup();

      // MSW mock for network error
      server.use(
        rest.post('/api/integrations/example/connect', (req, res) => {
          return res.networkError('Failed to connect');
        })
      );

      render(<IntegrationComponent />);

      const connectButton = screen.getByRole('button', { name: /connect/i });
      await user.click(connectButton);

      await waitFor(() => {
        expect(screen.getByText(/network error occurred/i)).toBeInTheDocument();
      });
    });

    it('displays error message on invalid credentials', async () => {
      const user = userEvent.setup();

      server.use(
        rest.post('/api/integrations/example/connect', (req, res, ctx) => {
          return res(ctx.status(401), ctx.json({
            error: 'Invalid credentials'
          }));
        })
      );

      render(<IntegrationComponent />);

      const connectButton = screen.getByRole('button', { name: /connect/i });
      await user.click(connectButton);

      await waitFor(() => {
        expect(screen.getByText(/invalid credentials/i)).toBeInTheDocument();
      });
    });

    it('displays error message on timeout', async () => {
      const user = userEvent.setup();

      server.use(
        rest.post('/api/integrations/example/connect', async (req, res) => {
          // Simulate timeout by never responding
          await new Promise(resolve => setTimeout(resolve, 10000));
          return res(ctx.status(200));
        })
      );

      render(<IntegrationComponent />);

      const connectButton = screen.getByRole('button', { name: /connect/i });
      await user.click(connectButton);

      await waitFor(() => {
        expect(screen.getByText(/request timed out/i)).toBeInTheDocument();
      }, { timeout: 5000 });
    });
  });

  describe('Loading States', () => {
    it('shows loading indicator during connection', async () => {
      const user = userEvent.setup();

      // MSW mock with delay
      server.use(
        rest.post('/api/integrations/example/connect', async (req, res, ctx) => {
          await new Promise(resolve => setTimeout(resolve, 100));
          return res(ctx.status(200), ctx.json({ authUrl: 'https://example.com/oauth' }));
        })
      );

      render(<IntegrationComponent />);

      const connectButton = screen.getByRole('button', { name: /connect/i });
      await user.click(connectButton);

      // Check for loading state
      expect(screen.getByTestId('loading-spinner') || screen.getByRole('button', { name: /connecting/i })).toBeInTheDocument();
    });
  });

  describe('Data Sync', () => {
    it('fetches and displays integration data after connection', async () => {
      server.use(
        rest.get('/api/integrations/example/data', (req, res, ctx) => {
          return res(ctx.status(200), ctx.json({
            items: [
              { id: '1', name: 'Item 1' },
              { id: '2', name: 'Item 2' },
            ]
          }));
        })
      );

      render(<IntegrationComponent isConnected={true} />);

      await waitFor(() => {
        expect(screen.getByText('Item 1')).toBeInTheDocument();
        expect(screen.getByText('Item 2')).toBeInTheDocument();
      });
    });
  });
});
```

// tests/mocks/handlers.ts - Integration API handlers

// Jira Integration
export const jiraHandlers = [
  rest.post('/api/integrations/jira/connect', (req, res, ctx) => {
    return res(ctx.status(200), ctx.json({
      authUrl: 'https://auth.atlassian.com/authorize',
      state: expect.any(String)
    }));
  }),
  rest.get('/api/integrations/jira/callback', (req, res, ctx) => {
    const { code, state } = req.url.searchParams;
    if (!code) {
      return res(ctx.status(400), ctx.json({ error: 'Missing code' }));
    }
    return res(ctx.status(200), ctx.json({
      status: 'connected',
      workspace: 'Test Workspace'
    }));
  }),
  rest.get('/api/integrations/jira/issues', (req, res, ctx) => {
    return res(ctx.status(200), ctx.json({
      issues: [
        { id: 'JIRA-1', summary: 'Test Issue 1', status: 'Open' },
        { id: 'JIRA-2', summary: 'Test Issue 2', status: 'In Progress' },
      ]
    }));
  }),
];

// Export all handlers for server.use()
