### Task 2: Parse Coverage Report
**Type:** auto
**Target:** coverage_264_final.json

**Parse Script:**
```python
import json
import sys

def parse_coverage_report(json_path):
    """Parse coverage JSON and generate summary statistics."""
    with open(json_path) as f:
        data = json.load(f)

    files = data.get('files', {})

    # Overall statistics
    total_lines = sum(f['summary']['num_statements'] for f in files.values())
    covered_lines = sum(f['summary']['covered_lines'] for f in files.values())
    overall_coverage = (covered_lines / total_lines * 100) if total_lines > 0 else 0

    # Per-module statistics
    modules = {}
    for file_path, file_data in files.items():
        # Extract module name (e.g., 'core', 'api', 'tools')
        module = file_path.split('/')[1] if '/' in file_path else 'other'

        if module not in modules:
            modules[module] = {'total': 0, 'covered': 0, 'files': []}

        modules[module]['total'] += file_data['summary']['num_statements']
        modules[module]['covered'] += file_data['summary']['covered_lines']
        modules[module]['files'].append({
            'path': file_path,
            'coverage': file_data['summary']['percent_covered'],
            'missing': file_data['summary']['missing_lines']
        })

    # Sort files by coverage (ascending) within each module
    for module in modules.values():
        module['files'].sort(key=lambda x: x['coverage'])

    return {
        'overall': {
            'total_lines': total_lines,
            'covered_lines': covered_lines,
            'coverage_pct': round(overall_coverage, 2)
        },
        'modules': modules
    }

if __name__ == '__main__':
    stats = parse_coverage_report('tests/coverage_reports/metrics/coverage_264_final.json')

    print(f"Overall Coverage: {stats['overall']['coverage_pct']}%")
    print(f"  Total Lines: {stats['overall']['total_lines']}")
    print(f"  Covered Lines: {stats['overall']['covered_lines']}")
    print()
    print("Per-Module Coverage:")
    for module, data in sorted(stats['modules'].items()):
        coverage = (data['covered'] / data['total'] * 100) if data['total'] > 0 else 0
        print(f"  {module}: {coverage:.2f}% ({data['covered']}/{data['total']} lines)")
```

```python
def gap_analysis_to_80(stats):
    """Calculate what's needed to reach 80% coverage."""
    current_coverage = stats['overall']['coverage_pct']
    target_coverage = 80.0
    gap = target_coverage - current_coverage

    total_lines = stats['overall']['total_lines']
    covered_lines = stats['overall']['covered_lines']

    # Lines needed to reach 80%
    target_covered = int(target_coverage / 100 * total_lines)
    lines_needed = target_covered - covered_lines

    # Find files with most uncovered lines
    low_coverage_files = []
    for module, data in stats['modules'].items():
        for file in data['files']:
            uncovered = file['missing']
            if uncovered > 0:
                low_coverage_files.append({
                    'module': module,
                    'path': file['path'],
                    'uncovered': uncovered,
                    'coverage': file['coverage']
                })

    # Sort by uncovered lines (descending)
    low_coverage_files.sort(key=lambda x: x['uncovered'], reverse=True)

    return {
        'current_coverage': current_coverage,
        'target_coverage': target_coverage,
        'gap_percentage': round(gap, 2),
        'lines_needed': lines_needed,
        'high_impact_files': low_coverage_files[:30]  # Top 30 files
    }
```

**Steps:**
1. Calculate gap percentage (80 - current)
2. Calculate lines needed to reach 80%
3. Identify high-impact files (most uncovered lines)
4. Categorize gaps:
   - **Easy wins:** <50 uncovered lines
   - **Medium effort:** 50-200 uncovered lines
   - **Large files:** >200 uncovered lines
5. Generate recommendations

**Expected Outcome:**

- Lines needed to reach 80%

- Effort categorization (easy/medium/large)
