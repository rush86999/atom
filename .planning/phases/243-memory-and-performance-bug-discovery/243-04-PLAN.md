    Pattern extending BugFilingService:
    ```python
    from backend.tests.bug_discovery.bug_filing_service import BugFilingService

    class MemoryPerformanceFilingService(BugFilingService):
        def file_memory_leak(self, test_name, memory_increase_mb, iterations, **kwargs):
            metadata = {
                "test_type": "memory",
                "memory_increase_mb": memory_increase_mb,
                "iterations": iterations,
                "platform": "backend-python"
            }
            metadata.update(kwargs)
            return self.file_bug(
                test_name=test_name,
                error_message=f"Memory leak detected: {memory_increase_mb:.2f} MB over {iterations} iterations",
                metadata=metadata
            )

        def _determine_severity(self, test_type, metadata):
            # Memory leaks are high severity
            if test_type == "memory":
                if metadata.get("memory_increase_mb", 0) > 50:
                    return "critical"
                return "high"
            # Performance regressions are medium
            if test_type == "performance":
                return "medium"
            return super()._determine_severity(test_type, metadata)
    ```
