    Pattern from research doc:
    ```python
    def check_regression(current_file, baseline_file, threshold=0.2):
        with open(current_file) as f:
            current = json.load(f)
        with open(baseline_file) as f:
            baseline = json.load(f)

        # Check performance score
        current_score = current['categories']['performance']['score']
        baseline_score = baseline['categories']['performance']['score']

        if current_score < baseline_score * (1 - threshold):
            print(f"REGRESSION: Performance score {current_score:.2f} < baseline {baseline_score:.2f}")
            return True

        # Check Core Web Vitals
        for metric in ['first-contentful-paint', 'largest-contentful-paint', 'total-blocking-time', 'cumulative-layout-shift']:
            current_value = current['audits'][metric]['numericValue']
            baseline_value = baseline['audits'][metric]['numericValue']
            if current_value > baseline_value * (1 + threshold):
                print(f"REGRESSION: {metric} {current_value:.0f}ms > baseline {baseline_value:.0f}ms")
                return True

        return False
    ```
