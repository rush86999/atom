   ```typescript
   import { by, element, expect } from 'detox';
   import { describe, it, beforeAll } from '@jest/globals';

   describe('Feature Parity (Mobile)', () => {
     beforeAll(async () => {
       await device.launchApp();
     });

     // Mirrors web: TestFeatureParity from test_feature_parity.py
     const AGENT_CHAT_FEATURES = [
       'streaming', 'history', 'feedback', 'canvas_presentations', 'skill_execution'
     ];

     const CANVAS_TYPES = [
       'generic', 'docs', 'email', 'sheets', 'orchestration', 'terminal', 'coding'
     ];

     it('should support all agent chat features', async () => {
       await element(by.id('agent-chat-screen')).tap();

       for (const feature of AGENT_CHAT_FEATURES) {
         const featureElement = element(by.id(`feature-${feature}`));
         await expect(featureElement).toBeVisible();
       }
     });

     it('should support all canvas types', async () => {
       await element(by.id('canvas-screen')).tap();

       for (const canvasType of CANVAS_TYPES) {
         await element(by.id(`present-${canvasType}-button`)).tap();
         await expect(element(by.id(`canvas-${canvasType}`))).toBeVisible();
         await element(by.id('close-canvas-button')).tap();
       }
     });

     it('should support skill execution', async () => {
       await element(by.id('skills-screen')).tap();
       await element(by.id('skill-list')).toBeVisible();

       // Install a test skill
       await element(by.id('install-test-skill-button')).tap();
       await expect(element(by.text('Skill installed'))).toBeVisible();

       // Execute the skill
       await element(by.id('execute-test-skill-button')).tap();
       await expect(element(by.id('skill-output'))).toBeVisible();
     });
   });
   ```

2. Create `.github/workflows/e2e-mobile.yml`:
   ```yaml
   name: E2E Tests (Mobile)

   on:
     push:
       branches: [main]
     workflow_dispatch:

   jobs:
     e2e-mobile:
       runs-on: macos-latest  # Required for iOS simulation
       timeout-minutes: 45

       steps:
         - name: Checkout code
           uses: actions/checkout@v4

         - name: Setup Node.js
           uses: actions/setup-node@v4
           with:
             node-version: '20'

         - name: Setup Expo
           uses: expo/expo-github-action@v8
           with:
             expo-version: latest
             eas-version: latest

         - name: Install dependencies
           working-directory: ./mobile
           run: npm ci

         - name: Build app for testing
           working-directory: ./mobile
           run: |
             # Use Expo dev client build if Detox requires it
             npx expo prebuild --clean || echo "Prebuild skipped"

         - name: Run Detox tests with JSON output
           working-directory: ./mobile
           run: |
             npm run e2e:build:ios
             # IMPORTANT: Use --reporter=json for E2E aggregator compatibility
             npm run e2e:test:ios -- --reporter=json --outputPath=e2e/results.json
           # If Detox doesn't support JSON directly, convert from JUnit to JSON
           # See: https://wix.github.io/Detox/docs/api/cli

         - name: Convert test results to JSON if needed
           working-directory: ./mobile
           run: |
             # If Detox outputs JUnit XML, convert to JSON for aggregation
             if [ -f e2e/junit.xml ] && [ ! -f e2e/results.json ]; then
               # Simple XML to JSON conversion (full implementation in Plan 07)
               node -e "
                 const fs = require('fs');
                 const xml = fs.readFileSync('e2e/junit.xml', 'utf8');
                 // Parse and convert to JSON format expected by e2e_aggregator.py
                 const results = { stats: { total: 0, passed: 0, failed: 0, duration: 0 } };
                 fs.writeFileSync('e2e/results.json', JSON.stringify(results, null, 2));
               "
             fi

         - name: Upload test artifacts
           if: always()
           uses: actions/upload-artifact@v4
           with:
             name: e2e-mobile-artifacts
             path: mobile/e2e/artifacts/
             retention-days: 7
             if-no-files-found: ignore

         - name: Upload test results for aggregation
           if: always()
           uses: actions/upload-artifact@v4
           with:
             name: e2e-mobile-report
             path: mobile/e2e/results.json
             retention-days: 7
             if-no-files-found: warn
   ```
