   import { describe, it, before, beforeEach } from 'mocha';
   import { browser, $, expect } from '@wdio/globals';
   import { TauriDriver } from '../../helpers/driver';

   describe('Authentication (Desktop)', () => {
     let driver: TauriDriver;

     before(async () => {
       driver = new TauriDriver();
     });

     beforeEach(async () => {
       await browser.url('http://localhost:1430');
     });

     it('should display login screen', async () => {
       // Mirrors web: test_authentication_workflow
       await driver.waitForVisible('login-screen');
       await driver.waitForVisible('login-email-input');
       await driver.waitForVisible('login-password-input');
       await driver.waitForVisible('login-submit-button');
     });

     it('should login with valid credentials', async () => {
       await driver.findByTestId('login-email-input').setValue('test@example.com');
       await driver.findByTestId('login-password-input').setValue('password123');
       await driver.findByTestId('login-submit-button').click();

       // Verify navigation to dashboard
       await driver.waitForVisible('dashboard-screen', 5000);
     });

     it('should show error for invalid credentials', async () => {
       await driver.findByTestId('login-email-input').setValue('invalid@example.com');
       await driver.findByTestId('login-password-input').setValue('wrongpassword');
       await driver.findByTestId('login-submit-button').click();

       const errorElement = await driver.findByTestId('login-error-message');
       await expect(errorElement).toHaveTextContaining('Invalid credentials');
     });

     it('should logout and clear session', async () => {
       // Login first
       await driver.findByTestId('login-email-input').setValue('test@example.com');
       await driver.findByTestId('login-password-input').setValue('password123');
       await driver.findByTestId('login-submit-button').click();
       await driver.waitForVisible('dashboard-screen');

       // Logout
       await driver.findByTestId('logout-button').click();
       await driver.waitForVisible('login-screen');
     });
   });
   ```

   import { describe, it, before } from 'mocha';
   import { expect } from '@wdio/globals';
   import { TauriDriver } from '../../helpers/driver';

   describe('Feature Parity (Desktop)', () => {
     let driver: TauriDriver;

     before(async () => {
       driver = new TauriDriver();
     });

     // Mirrors web: TestFeatureParity from test_feature_parity.py
     const AGENT_CHAT_FEATURES = [
       'streaming', 'history', 'feedback', 'canvas_presentations', 'skill_execution'
     ];

     const CANVAS_TYPES = [
       'generic', 'docs', 'email', 'sheets', 'orchestration', 'terminal', 'coding'
     ];

     it('should support all agent chat features', async () => {
       await driver.navigate('/agent/test-agent');

       for (const feature of AGENT_CHAT_FEATURES) {
         const featureElement = await driver.findByTestId(`feature-${feature}`);
         await expect(featureElement).toBeDisplayed();
       }
     });

     it('should support all canvas types', async () => {
       await driver.navigate('/canvas');

       for (const canvasType of CANVAS_TYPES) {
         await driver.findByTestId(`present-${canvasType}-button`).click();
         await driver.waitForVisible(`canvas-${canvasType}`);
         await driver.findByTestId('close-canvas-button').click();
       }
     });

     it('should support skill execution', async () => {
       await driver.navigate('/skills');
       await driver.waitForVisible('skill-list');

       // Install a test skill
       await driver.findByTestId('install-test-skill-button').click();
       const installMessage = await driver.findByTestId('skill-install-message');
       await expect(installMessage).toHaveTextContaining('installed');

       // Execute the skill
       await driver.findByTestId('execute-test-skill-button').click();
       await driver.waitForVisible('skill-output');
     });
   });
   ```

2. Create `.github/workflows/e2e-desktop.yml`:
   ```yaml
   name: E2E Tests (Desktop)

   on:
     push:
       branches: [main]
     workflow_dispatch:

   jobs:
     e2e-desktop:
       runs-on: ubuntu-latest
       timeout-minutes: 30

       steps:
         - name: Checkout code
           uses: actions/checkout@v4

         - name: Setup Node.js
           uses: actions/setup-node@v4
           with:
             node-version: '20'

         - name: Setup Rust
           uses: dtolnay/rust-toolchain@stable

         - name: Install system dependencies
           run: |
             sudo apt-get update
             sudo apt-get install -y libwebkit2gtk-4.0-dev \
               build-essential curl wget file libssl-dev libayatana-appindicator3-dev \
               librsvg2-dev

         - name: Install dependencies
           working-directory: ./frontend-nextjs
           run: npm ci

         - name: Build Tauri app
           working-directory: ./frontend-nextjs
           run: |
             npm run tauri build --debug
             # Built app will be in src-tauri/target/debug/bundle/

         - name: Start Tauri app in background
           working-directory: ./frontend-nextjs
           run: |
             npm run tauri dev &
             sleep 15
             curl --retry 10 --retry-delay 5 --retry-connrefused http://localhost:1430

         - name: Run WebDriverIO tests with JSON output
           working-directory: ./frontend-nextjs
           run: |
             # IMPORTANT: Use --format=json for E2E aggregator compatibility
             npm run test:e2e -- --format=json --outputFile=wdio/results.json

         - name: Upload test artifacts
           if: always()
           uses: actions/upload-artifact@v4
           with:
             name: e2e-desktop-artifacts
             path: frontend-nextjs/wdio/artifacts/
             retention-days: 7
             if-no-files-found: ignore

         - name: Upload test results for aggregation
           if: always()
           uses: actions/upload-artifact@v4
           with:
             name: e2e-desktop-report
             path: frontend-nextjs/wdio/results.json
             retention-days: 7
             if-no-files-found: warn
   ```
