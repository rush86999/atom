3. Create `wdio/helpers/driver.ts`:
   ```typescript
   /**
    * Tauri WebDriver helper for cross-platform test abstraction.
    * Provides unified interface for desktop E2E tests.
    */
   import { browser, $, $$ } from '@wdio/globals';

   export class TauriDriver {
     /**
      * Navigate to a route within the Tauri app.
      */
     async navigate(route: string): Promise<void> {
       await browser.url(`http://localhost:1430${route}`);
     }

     /**
      * Find element by test ID (resilient selector).
      */
     async findByTestId(testId: string) {
       return $(`[data-testid="${testId}"]`);
     }

     /**
      * Wait for element to be visible.
      */
     async waitForVisible(testId: string, timeout = 5000): Promise<void> {
       await this.findByTestId(testId).waitForDisplayed({ timeout });
     }

     /**
      * Send message to agent chat.
      */
     async sendAgentMessage(message: string): Promise<void> {
       const input = await this.findByTestId('agent-chat-input');
       await input.setValue(message);
       const sendBtn = await this.findByTestId('send-message-button');
       await sendBtn.click();
     }

     /**
      * Check if element is visible.
      */
     async isVisible(testId: string): Promise<boolean> {
       return await this.findByTestId(testId).isDisplayed();
     }
   }
   ```
