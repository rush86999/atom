class TestCoverageModuleBreakdown:
    """Test module-level coverage breakdown."""

    def test_module_coverage_totals(self):
        """Test that module coverage totals are calculated correctly."""
        result = run(
            ["python3", "-m", "pytest", "--cov=backend", "--cov-report=json", "-o", "addopts=", "-q"],
            cwd="/Users/rushiparikh/projects/atom/backend",
            capture_output=True,
            text=True
        )

        with open("backend/coverage.json") as f:
            data = json.load(f)

        # Calculate module totals
        modules = {}
        for path, info in data['files'].items():
            if '/core/' in path:
                module = 'core'
            elif '/api/' in path:
                module = 'api'
            elif '/tools/' in path:
                module = 'tools'
            elif '/cli/' in path:
                module = 'cli'
            else:
                continue

            if module not in modules:
                modules[module] = {'covered': 0, 'total': 0}
            modules[module]['covered'] += info['summary']['covered_lines']
            modules[module]['total'] += info['summary']['num_statements']

        # Verify totals are calculated
        assert 'core' in modules
        assert 'api' in modules
        assert modules['core']['total'] > 0

    def test_zero_coverage_files_identified(self):
        """Test that zero-coverage files are correctly identified."""
        result = run(
            ["python3", "-m", "pytest", "--cov=backend", "--cov-report=json", "-o", "addopts=", "-q"],
            cwd="/Users/rushiparikh/projects/atom/backend",
            capture_output=True,
            text=True
        )

        with open("backend/coverage.json") as f:
            data = json.load(f)

        # Count zero-coverage files >100 lines
        zero_coverage = []
        for path, info in data['files'].items():
            stmts = info['summary']['num_statements']
            covered = info['summary']['covered_lines']
            if stmts > 100 and covered == 0:
                zero_coverage.append(path)

        # Should have identified zero-coverage files

import json

    if '/core/' in path:
        module = 'core'
    elif '/api/' in path:
        module = 'api'
    elif '/tools/' in path:
        module = 'tools'
    elif '/cli/' in path:
        module = 'cli'
    elif '/integrations/' in path:
        module = 'integrations'

        module = 'skills'

    modules[module]['total'] += info['summary']['num_statements']
    modules[module]['files'] += 1

for module, stats in sorted(modules.items()):
    pct = (stats['covered'] / stats['total'] * 100) if stats['total'] > 0 else 0
    print(f"\n{module.upper()}:")
    print(f"  Coverage: {pct:.2f}% ({stats['covered']:,}/{stats['total']:,}) in {stats['files']} files")

# Top improved files
print("\n" + "=" * 60)
print("TOP IMPROVED FILES (Phase 203)")
print("=" * 60)

improved_files = []
for path, info in data['files'].items():
    fname = path.split('/')[-1]
    stmts = info['summary']['num_statements']
    covered = info['summary']['covered_lines']
    if stmts > 100:
        improved_files.append({
            'name': fname,
            'coverage': info['summary']['percent_covered'],
            'statements': stmts
        })

improved_files.sort(key=lambda x: x['coverage'], reverse=True)
for i, f in enumerate(improved_files[:10], 1):
    print(f"{i:2d}. {f['name']:50s} {f['coverage']:5.2f}% ({f['statements']:4d} stmts)")

zero_coverage = [f for f in improved_files if f['coverage'] == 0]
zero_coverage.sort(key=lambda x: x['statements'], reverse=True)

print(f"Total: {len(zero_coverage)} files")
for i, f in enumerate(zero_coverage[:15], 1):
    print(f"{i:2d}. {f['name']:50s} ({f['statements']:4d} stmts)")

if len(zero_coverage) > 15:
    print(f"... and {len(zero_coverage) - 15} more")

print("\n" + "=" * 60)
EOF
