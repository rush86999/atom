```python
class DebugEvent(Base):
    """
    Debug event captured during agent execution.

    Tracks anomalies, errors, and notable events for debugging
    and system health monitoring.
    """
    __tablename__ = "debug_events"

    id = Column(String, primary_key=True, default=lambda: f"debug_evt_{uuid4().hex[:12]}")
    agent_id = Column(String, ForeignKey("agents.id"), nullable=True, index=True)
    episode_id = Column(String, ForeignKey("episodes.id"), nullable=True, index=True)
    event_type = Column(String(50), nullable=False, index=True)  # error, warning, anomaly, performance
    severity = Column(String(20), nullable=False, default="info")  # critical, high, medium, low, info
    message = Column(Text, nullable=False)
    metadata = Column(JSON, nullable=True)
    status = Column(String(20), default="open")  # open, investigating, resolved, ignored
    created_at = Column(DateTime(timezone=True), default=lambda: datetime.now(timezone.utc))
    updated_at = Column(DateTime(timezone=True), onupdate=lambda: datetime.now(timezone.utc))
    resolved_at = Column(DateTime(timezone=True), nullable=True)

    # Relationship to insights
    insights = relationship("DebugInsight", back_populates="event", cascade="all, delete-orphan")

    def __repr__(self):
        return f"<DebugEvent {self.id}: {self.event_type} - {self.severity}>"


class DebugInsight(Base):
    """
    AI-generated insight about a debug event.

    Provides analysis, root cause hypotheses, and suggested actions
    for debug events using AI-powered diagnostics.
    """
    __tablename__ = "debug_insights"

    id = Column(String, primary_key=True, default=lambda: f"debug_ins_{uuid4().hex[:12]}")
    event_id = Column(String, ForeignKey("debug_events.id"), nullable=False, index=True)
    insight_type = Column(String(50), nullable=False)  # root_cause, suggestion, pattern, correlation
    confidence = Column(Float, default=0.5)  # 0.0 to 1.0
    title = Column(String(200), nullable=True)
    description = Column(Text, nullable=True)
    suggested_action = Column(Text, nullable=True)
    metadata = Column(JSON, nullable=True)
    status = Column(String(20), default="pending")  # pending, reviewed, applied, dismissed
    created_at = Column(DateTime(timezone=True), default=lambda: datetime.now(timezone.utc))
    updated_at = Column(DateTime(timezone=True), onupdate=lambda: datetime.now(timezone.utc))

    # Relationship to event
    event = relationship("DebugEvent", back_populates="insights")

    def __repr__(self):
        return f"<DebugInsight {self.id}: {self.insight_type} (confidence: {self.confidence})>"
```

Key design decisions:
1. Uses String primary keys with UUID-based prefixes (consistent with other models)
2. Foreign keys to agents and episodes for traceability
3. JSON metadata field for flexible event/insight data
4. Status workflow fields for lifecycle tracking
5. Proper cascade delete for insights when event is deleted
6. Timezone-aware datetime fields (using datetime.now(timezone.utc))
