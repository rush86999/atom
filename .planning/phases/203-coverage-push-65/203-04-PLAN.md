        """Test state transition from CREATED to RUNNING."""
        engine = WorkflowEngine()

            await engine.execute_workflow(workflow.id)
            workflow = await engine.get_workflow(workflow.id)
            assert workflow.status in [WorkflowStatus.RUNNING, WorkflowStatus.COMPLETED]

    async def test_state_transition_running_to_completed(self):
        """Test state transition from RUNNING to COMPLETED."""
        engine = WorkflowEngine()
        workflow = await engine.create_workflow(
            name="test_workflow",
            steps=[{"id": "step1", "action": "test_action", "params": {}}]
        )

        with patch.object(engine, "_execute_step", return_value={"status": "success"}):
            result = await engine.execute_workflow(workflow.id)
            if result["status"] == "completed":
                workflow = await engine.get_workflow(workflow.id)
                assert workflow.status == WorkflowStatus.COMPLETED

    async def test_state_transition_to_failed_on_error(self):
        """Test state transition to FAILED when step fails."""
        engine = WorkflowEngine()
        workflow = await engine.create_workflow(
            name="test_workflow",
            steps=[{"id": "step1", "action": "test_action", "params": {}}]
        )

        with patch.object(engine, "_execute_step", return_value={"status": "error", "message": "Test error"}):
            result = await engine.execute_workflow(workflow.id)
            workflow = await engine.get_workflow(workflow.id)
            assert workflow.status == WorkflowStatus.FAILED

    async def test_invalid_state_transition_raises_error(self):

@pytest.mark.asyncio
class TestWorkflowEngineErrors:
    """Test workflow engine error handling."""

    async def test_step_timeout(self):
        """Test workflow step timeout handling."""
        engine = WorkflowEngine(config={"timeout_seconds": 1})
        workflow = await engine.create_workflow(
            name="test_workflow",
            steps=[{"id": "step1", "action": "slow_action", "params": {}}]
        )

        async def slow_step(*args, **kwargs):
            import asyncio
            await asyncio.sleep(2)
            return {"status": "success"}

        with patch.object(engine, "_execute_step", side_effect=slow_step):
            result = await engine.execute_workflow(workflow.id)
            # Should handle timeout gracefully

    async def test_step_exception_handling(self):
        """Test exception handling in step execution."""
        engine = WorkflowEngine()
        workflow = await engine.create_workflow(
            name="test_workflow",
            steps=[{"id": "step1", "action": "failing_action", "params": {}}]
        )

        with patch.object(engine, "_execute_step", side_effect=Exception("Step failed")):
            result = await engine.execute_workflow(workflow.id)
            workflow = await engine.get_workflow(workflow.id)
            assert workflow.status == WorkflowStatus.FAILED

    async def test_dependency_cycle_detection(self):
        """Test detection of circular dependencies in steps."""
        engine = WorkflowEngine()

        with pytest.raises(ValueError, match="cycle"):
            await engine.create_workflow(
                name="test_workflow",
                steps=[
                    {"id": "step1", "action": "action1", "params": {}, "depends_on": ["step3"]},
                    {"id": "step2", "action": "action2", "params": {}, "depends_on": ["step1"]},
                    {"id": "step3", "action": "action3", "params": {}, "depends_on": ["step2"]}
                ]
            )

    async def test_missing_dependency_error(self):
        """Test error when step depends on non-existent step."""
        engine = WorkflowEngine()

        with pytest.raises(ValueError, match="depends on"):
            await engine.create_workflow(
                name="test_workflow",
                steps=[
                    {"id": "step1", "action": "action1", "params": {}, "depends_on": ["nonexistent"]}
                ]
            )
