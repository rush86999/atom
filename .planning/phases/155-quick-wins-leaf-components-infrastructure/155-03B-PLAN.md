import { Agent, Canvas, DeviceInfo } from '../types';

interface MockAgentOptions {
  id?: string;
  name?: string;
  maturity?: 'STUDENT' | 'INTERN' | 'SUPERVISED' | 'AUTONOMOUS';
  status?: 'active' | 'inactive';
}

export const createMockAgent = (options: MockAgentOptions = {}): Agent => ({
  id: options.id || 'mock-agent-id',
  name: options.name || 'Mock Agent',
  maturity: options.maturity || 'INTERN',
  status: options.status || 'active',
  createdAt: new Date().toISOString(),
  updatedAt: new Date().toISOString(),
});

export const createMockAgentList = (count: number = 3): Agent[] =>
  Array.from({ length: count }, (_, i) => createMockAgent({
    id: `mock-agent-${i}`,
    name: `Mock Agent ${i}`,
  }));

interface MockCanvasOptions {
  id?: string;
  title?: string;
  type?: 'chart' | 'form' | 'markdown';
}

export const createMockCanvas = (options: MockCanvasOptions = {}): Canvas => ({
  id: options.id || 'mock-canvas-id',
  title: options.title || 'Mock Canvas',
  type: options.type || 'markdown',
  content: options.type === 'markdown' ? '# Mock Content' : {},
  createdAt: new Date().toISOString(),
});

export const createMockDevice = (): DeviceInfo => ({
  platform: Platform.OS,
  osVersion: Platform.Version as string,
  appVersion: '1.0.0',
  deviceId: 'mock-device-id',
});

// Export all for use in tests
export const mockAgents = createMockAgentList();
export const mockCanvas = createMockCanvas();
export const mockDevice = createMockDevice();
