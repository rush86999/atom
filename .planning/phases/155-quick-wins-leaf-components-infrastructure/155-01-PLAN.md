Pattern example:
```rust
#[cfg(test)]
mod tests {
    use super::*;
    use serde_json;

    #[test]
    fn test_agent_dto_serialization() {
        let agent = AgentDto {
            id: "test-agent".to_string(),
            name: "Test Agent".to_string(),
            maturity: MaturityLevel::Autonomous,
            status: "active".to_string(),
        };

        let json = serde_json::to_string(&agent).unwrap();
        assert!(json.contains("\"id\":\"test-agent\""));
        assert!(json.contains("\"maturity\":\"AUTONOMOUS\""));
    }

    #[test]
    fn test_agent_dto_deserialization() {
        let json = r#"{"id":"test-agent","name":"Test","maturity":"AUTONOMOUS","status":"active"}"#;
        let agent: AgentDto = serde_json::from_str(json).unwrap();
        assert_eq!(agent.id, "test-agent");
        assert_eq!(agent.maturity, MaturityLevel::Autonomous);
    }

    #[test]
    fn test_agent_dto_round_trip() {
        let original = AgentDto {
            id: "test-agent".to_string(),
            name: "Test Agent".to_string(),
            maturity: MaturityLevel::Supervised,
            status: "active".to_string(),
        };

        let json = serde_json::to_string(&original).unwrap();
        let deserialized: AgentDto = serde_json::from_str(&json).unwrap();

        assert_eq!(original.id, deserialized.id);
        assert_eq!(original.maturity, deserialized.maturity);
    }
}
```

Use serde_json for JSON serialization tests. Test both serialization (Rust -> JSON) and deserialization (JSON -> Rust).
