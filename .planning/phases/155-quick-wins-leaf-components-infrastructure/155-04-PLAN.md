```typescript
import { renderHook, act } from '@testing-library/react';
import { useFormatters, useLocalStorage } from '../hooks/useFormatters';

describe('useFormatters Hook', () => {
  it('returns formatter functions', () => {
    const { result } = renderHook(() => useFormatters());

    expect(result.current.formatDate).toBeInstanceOf(Function);
    expect(result.current.formatCurrency).toBeInstanceOf(Function);
  });

  it('formats dates correctly', () => {
    const { result } = renderHook(() => useFormatters());

    expect(result.current.formatDate('2026-03-08')).toBe('March 8, 2026');
  });

  it('memoizes formatter functions', () => {
    const { result, rerender } = renderHook(() => useFormatters());

    const firstFormatDate = result.current.formatDate;
    rerender();
    const secondFormatDate = result.current.formatDate;

    expect(firstFormatDate).toBe(secondFormatDate);
  });
});

describe('useLocalStorage Hook', () => {
  it('reads from localStorage', () => {
    localStorage.setItem('test-key', '"test-value"');
    const { result } = renderHook(() => useLocalStorage('test-key'));

    expect(result.current[0]).toBe('test-value');
  });

  it('writes to localStorage', () => {
    const { result } = renderHook(() => useLocalStorage('test-key'));

    act(() => {
      const setValue = result.current[1];
      setValue('new-value');
    });

    expect(localStorage.getItem('test-key')).toBe('"new-value"');
  });
});
```

```tsx
import { renderHook, act } from '@testing-library/react';
import { useAsyncStorage } from '../storage/asyncStorage';

describe('AsyncStorage', () => {
  beforeEach(async () => {
    await AsyncStorage.clear();
  });

  it('setItem and getItem', async () => {
    await AsyncStorage.setItem('test-key', 'test-value');
    const value = await AsyncStorage.getItem('test-key');
    expect(value).toBe('test-value');
  });

  it('stores and retrieves objects', async () => {
    const obj = { id: 1, name: 'Test' };
    await AsyncStorage.setItem('obj-key', JSON.stringify(obj));
    const value = await AsyncStorage.getItem('obj-key');
    expect(JSON.parse(value || '')).toEqual(obj);
  });

  it('multiSet and multiGet', async () => {
    const pairs = [['key1', 'value1'], ['key2', 'value2']];
    await AsyncStorage.multiSet(pairs);
    const values = await AsyncStorage.multiGet(['key1', 'key2']);
    expect(values).toEqual(pairs);
  });

  it('getAllKeys', async () => {
    await AsyncStorage.setItem('key1', 'value1');
    await AsyncStorage.setItem('key2', 'value2');
    const keys = await AsyncStorage.getAllKeys();
    expect(keys).toContain('key1');
    expect(keys).toContain('key2');
  });
});
```
