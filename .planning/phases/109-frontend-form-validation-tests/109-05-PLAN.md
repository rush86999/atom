export const formSubmissionHandlers = [
  // Submit form data successfully
  rest.post('/api/forms/submit', (req, res, ctx) => {
    return res(
      ctx.status(200),
      ctx.json({
        success: true,
        submission_id: `sub-${Date.now()}`,
        message: 'Form submitted successfully',
        submitted_at: new Date().toISOString()
      })
    );
  }),

  // Form submission with validation error from server
  rest.post('/api/forms/submit', (req, res, ctx) => {
    const body = req.body as any;
    // Simulate server validation error for specific field
    if (body.email === 'invalid@example.com') {
      return res(
        ctx.status(400),
        ctx.json({
          success: false,
          error: 'Validation failed',
          field_errors: {
            email: 'This email is already registered'
          }
        })
      );
    }
    // Default success
    return res(
      ctx.status(200),
      ctx.json({ success: true, submission_id: 'sub-123' })
    );
  }),

  // Form submission with server error
  rest.post('/api/forms/error', (req, res, ctx) => {
    return res(
      ctx.status(500),
      ctx.json({
        success: false,
        error: 'Internal server error',
        timestamp: new Date().toISOString()
      })
    );
  }),

  // Form submission timeout simulation
  rest.post('/api/forms/timeout', (req, res, ctx) => {
    return res(
      ctx.delay(10000),  // 10 second delay
      ctx.status(200),
      ctx.json({ success: true })
    );
  }),

  // Form submission with network error simulation
  rest.post('/api/forms/network-error', (req, res) => {
    // Simulate network failure by not responding
    // This will trigger a network error in the client
    return res.networkError('Network connection failed');
  }),
];

// Export updated allHandlers
export const allHandlers = [
  ...commonHandlers,
  ...agentHandlers,
  ...canvasHandlers,
  ...deviceHandlers,
  ...formSubmissionHandlers,  // NEW
];
