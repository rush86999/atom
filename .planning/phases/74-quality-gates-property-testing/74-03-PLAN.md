       ```yaml
  # GATES-04: Local coverage check before push
  - repo: local
    hooks:
      - id: pytest-with-coverage
        name: Run pytest with coverage check (80% minimum)
        entry: bash -c "cd backend && pytest tests/ -q --cov=core --cov=api --cov=tools --cov-branch --cov-report=term-missing:skip-covered --cov-fail-under=80"
        language: system
        pass_filenames: false
        always_run: true
       ```

    2. This hook:
       - Runs from backend/ directory
       - Uses same flags as CI (cov-branch, cov-fail-under=80)
       - Shows missing lines in terminal for developer feedback
       - Runs on every commit (always_run: true)
