Create server configuration and utilities:

1. Create tests/mocks/server.ts:
   ```typescript
   import { setupServer, SetupServerApi } from 'msw/node';
   import { allHandlers } from './handlers';

   // Create MSW server with all handlers
   export const server = setupServer(...allHandlers);

   /**
    * Start the mock server with optional custom handlers
    */
   export function setupMockServer(customHandlers?: any[]) {
       if (customHandlers) {
           server.use(...customHandlers);
       }
       server.listen({ onUnhandledRequest: 'error' });
       return server;
   }

   /**
    * Reset all handlers to default
    */
   export function resetMockServer() {
       server.resetHandlers(...allHandlers);
   }

   /**
    * Override a specific handler for a test scenario
    */
   export function overrideHandler(handler: any) {
       server.use(handler);
   }

   /**
    * Override multiple handlers for a test scenario
    */
   export function overrideHandlers(...handlers: any[]) {
       server.use(...handlers);
   }

   export default server;
   ```
