  <action>
    Create FastCheck property tests for Redux/Zustand reducer invariants:

    ```typescript
    import fc from 'fast-check';
    import { renderHook, act } from '@testing-library/react';

    // Define a simple reducer for testing (replace with actual reducers)
    type State = { count: number; name: string };
    type Action = { type: 'INCREMENT' } | { type: 'DECREMENT' } | { type: 'SET_NAME'; payload: string };

    function reducer(state: State, action: Action): State {
      switch (action.type) {
        case 'INCREMENT': return { ...state, count: state.count + 1 };
        case 'DECREMENT': return { ...state, count: state.count - 1 };
        case 'SET_NAME': return { ...state, name: action.payload };
        default: return state;
      }
    }

    describe('Reducer Invariants Property Tests', () => {
      // INVARIANT: Reducer should not mutate input state
      fc.assert(
        fc.property(
          fc.integer(),
          fc.string(),
          (initialCount, initialName) => {
            const state: State = { count: initialCount, name: initialName };
            const stateCopy = JSON.stringify(state);
            reducer(state, { type: 'INCREMENT' });
            expect(JSON.stringify(state)).toBe(stateCopy);
          }
        ),
        { numRuns: 100 }
      );

      // INVARIANT: INCREMENT/DECREMENT should only affect count field
      fc.assert(
        fc.property(
          fc.integer(),
          fc.string(),
          (initialCount, initialName) => {
            let state: State = { count: initialCount, name: initialName };
            state = reducer(state, { type: 'INCREMENT' });
            expect(state.name).toBe(initialName);
            expect(state.count).toBe(initialCount + 1);
          }
        ),
        { numRuns: 100 }
      );

      // INVARIANT: INCREMENT then DECREMENT should return to original state
      fc.assert(
        fc.property(
          fc.integer(),
          (initialCount) => {
            let state: State = { count: initialCount, name: 'test' };
            state = reducer(state, { type: 'INCREMENT' });
            state = reducer(state, { type: 'DECREMENT' });
            expect(state.count).toBe(initialCount);
          }
        ),
        { numRuns: 100 }
      );

      // INVARIANT: Unknown action type should return unchanged state
      fc.assert(
        fc.property(
          fc.integer(),
          fc.string(),
          (initialCount, initialName) => {
            const state: State = { count: initialCount, name: initialName };
            const newState = reducer(state, { type: 'UNKNOWN' } as any);
            expect(newState).toBe(state);
          }
        ),
        { numRuns: 50 }
      );

      // INVARIANT: Multiple INCREMENTs should be additive
      fc.assert(
        fc.property(
          fc.integer(),
          fc.integer({ min: 1, max: 100 }),
          (initialCount, times) => {
            let state: State = { count: initialCount, name: 'test' };
            for (let i = 0; i < times; i++) {
              state = reducer(state, { type: 'INCREMENT' });
            }
            expect(state.count).toBe(initialCount + times);
          }
        ),
        { numRuns: 100 }
      );
    });
    ```
