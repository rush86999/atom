    ```python
    from typing import Dict, List, Optional, Deque
    from collections import deque
    from datetime import datetime, timedelta, timezone
    import logging

    logger = logging.getLogger(__name__)


    class ProviderHealthMonitor:
        """Track provider API health for routing decisions using EMA scoring"""

        def __init__(self, window_minutes: int = 5):
            self.window_minutes = window_minutes
            self.call_history: Dict[str, Deque[tuple]] = {}  # provider_id -> deque of (timestamp, success, latency_ms)
            self.health_scores: Dict[str, float] = {}  # provider_id -> 0.0-1.0 score

        def record_call(self, provider_id: str, success: bool, latency_ms: float):
            """Record API call outcome for health tracking

            Args:
                provider_id: Provider identifier (e.g., "openai", "anthropic")
                success: True if API call succeeded, False otherwise
                latency_ms: Request latency in milliseconds
            """
            timestamp = datetime.now(timezone.utc)

            if provider_id not in self.call_history:
                self.call_history[provider_id] = deque()

            history = self.call_history[provider_id]
            history.append((timestamp, success, latency_ms))

            # Trim old entries outside sliding window
            cutoff = timestamp - timedelta(minutes=self.window_minutes)
            while history and history[0][0] < cutoff:
                history.popleft()

            # Update health score using EMA
            self._update_health_score(provider_id)

        def get_health_score(self, provider_id: str) -> float:
            """Get health score for provider (0.0-1.0)

            Returns default 1.0 (healthy) if no history exists
            """
            return self.health_scores.get(provider_id, 1.0)

        def get_healthy_providers(self, min_score: float = 0.5) -> List[str]:
            """Get providers with health score above threshold"""
            return [
                provider_id
                for provider_id, score in self.health_scores.items()
                if score >= min_score
            ]

        def get_provider_stats(self, provider_id: str) -> Dict:
            """Get detailed stats for a provider"""
            history = self.call_history.get(provider_id, deque())

            if not history:
                return {
                    "provider_id": provider_id,
                    "health_score": 1.0,
                    "call_count": 0,
                    "success_rate": None,
                    "avg_latency_ms": None
                }

            success_count = sum(1 for _, success, _ in history if success)
            total_latency = sum(lat for _, _, lat in history)

            return {
                "provider_id": provider_id,
                "health_score": self.health_scores.get(provider_id, 1.0),
                "call_count": len(history),
                "success_rate": success_count / len(history) if history else None,
                "avg_latency_ms": total_latency / len(history) if history else None
            }

        def _update_health_score(self, provider_id: str):
            """Calculate health score from recent history (0.0-1.0)

            Uses weighted combination:
            - Success rate: 70% weight
            - Latency score: 30% weight (inverse of latency, 5s = 0 score)
            """
            history = self.call_history.get(provider_id, deque())

            if not history:
                self.health_scores[provider_id] = 1.0  # Default healthy
                return

            # Success rate (70% weight)
            success_count = sum(1 for _, success, _ in history if success)
            success_rate = success_count / len(history)

            # Latency score (30% weight, inverse of latency)
            avg_latency = sum(lat for _, _, lat in history) / len(history)
            # 5 second latency = 0 score, 0ms = 1.0 score
            latency_score = max(0, 1 - (avg_latency / 5000))

            # Combined score using EMA (exponential moving average)
            new_score = (success_rate * 0.7) + (latency_score * 0.3)

            # Apply EMA smoothing: 20% weight to new score, 80% to existing
            existing_score = self.health_scores.get(provider_id, 1.0)
            self.health_scores[provider_id] = (existing_score * 0.8) + (new_score * 0.2)


    _health_monitor_instance: Optional[ProviderHealthMonitor] = None


    def get_provider_health_monitor() -> ProviderHealthMonitor:
        """Get singleton instance of ProviderHealthMonitor"""
        global _health_monitor_instance
        if _health_monitor_instance is None:
            _health_monitor_instance = ProviderHealthMonitor()
        return _health_monitor_instance
    ```

    ```python
    import os
    import asyncio
    import logging
    from typing import Optional
    from apscheduler.schedulers.asyncio import AsyncIOScheduler

    from core.provider_auto_discovery import get_auto_discovery

    logger = logging.getLogger(__name__)

    # Environment variable for auto-sync toggle
    PROVIDER_AUTO_SYNC_ENABLED = os.getenv("PROVIDER_AUTO_SYNC_ENABLED", "true").lower() == "true"


    class ProviderScheduler:
        """Manages scheduled provider registry sync jobs using APScheduler"""

        def __init__(self):
            self.scheduler = AsyncIOScheduler()
            self.auto_discovery = get_auto_discovery()
            self._consecutive_failures = 0
            self._max_failures = 3

        def start(self):
            """Schedule auto-sync every 24 hours with error handling"""
            if not PROVIDER_AUTO_SYNC_ENABLED:
                logger.info("Provider auto-sync disabled via PROVIDER_AUTO_SYNC_ENABLED")
                return

            # Schedule sync job every 24 hours
            self.scheduler.add_job(
                self._sync_with_error_handling,
                'interval',
                hours=24,
                id='provider_auto_sync',
                max_instances=1,  # Prevent overlapping executions
                coalesce=True,    # Combine missed jobs
                misfire_grace_time=3600  # 1 hour grace
            )
            self.scheduler.start()
            logger.info("ProviderScheduler started (24-hour polling interval)")

        async def _sync_with_error_handling(self):
            """Wrapper with error handling and exponential backoff"""
            try:
                logger.info("Starting scheduled provider sync")
                result = await self.auto_discovery.sync_providers()
                logger.info(f"Auto-sync completed: {result}")

                # Reset failure counter on success
                self._consecutive_failures = 0

            except Exception as e:
                self._consecutive_failures += 1
                logger.error(f"Auto-sync failed (attempt {self._consecutive_failures}): {e}")

                # Implement exponential backoff after 3 consecutive failures
                if self._consecutive_failures >= self._max_failures:
                    logger.warning(f"Too many consecutive failures ({self._consecutive_failures}), enabling backoff")
                    # Backoff: double the interval temporarily
                    job = self.scheduler.get_job('provider_auto_sync')
                    if job:
                        job.reschedule('interval', hours=48)  # Double interval
                        logger.info("Sync interval increased to 48 hours due to repeated failures")

        async def trigger_sync_now(self) -> dict:
            """Manually trigger sync immediately (for testing and admin use)"""
            try:
                logger.info("Manual sync triggered")
                result = await self.auto_discovery.sync_providers()
                return {"success": True, "result": result}
            except Exception as e:
                logger.error(f"Manual sync failed: {e}")
                return {"success": False, "error": str(e)}

        def get_scheduler_status(self) -> dict:
            """Get scheduler status for health monitoring"""
            return {
                "running": self.scheduler.running,
                "auto_sync_enabled": PROVIDER_AUTO_SYNC_ENABLED,
                "consecutive_failures": self._consecutive_failures,
                "jobs": [
                    {
                        "id": job.id,
                        "next_run_time": job.next_run_time.isoformat() if job.next_run_time else None
                    }
                    for job in self.scheduler.get_jobs()
                ]
            }

        def stop(self):
            """Stop the scheduler"""
            if self.scheduler.running:
                self.scheduler.shutdown()
                logger.info("ProviderScheduler stopped")


    _scheduler_instance: Optional[ProviderScheduler] = None


    def get_provider_scheduler() -> ProviderScheduler:
        """Get singleton instance of ProviderScheduler"""
        global _scheduler_instance
        if _scheduler_instance is None:
            _scheduler_instance = ProviderScheduler()
        return _scheduler_instance
    ```

    ```python
    import pytest
    from unittest.mock import Mock, patch, AsyncMock
    import asyncio

    from core.provider_scheduler import ProviderScheduler, get_provider_scheduler, PROVIDER_AUTO_SYNC_ENABLED


    class TestProviderScheduler:
        """Test provider scheduler with APScheduler integration"""

        def test_singleton_pattern(self):
            """Verify singleton instance"""
            scheduler1 = get_provider_scheduler()
            scheduler2 = get_provider_scheduler()
            assert scheduler1 is scheduler2

        @patch('core.provider_scheduler.PROVIDER_AUTO_SYNC_ENABLED', False)
        def test_start_disabled_when_env_false(self):
            """Test scheduler doesn't start when auto-sync disabled"""
            scheduler = ProviderScheduler()
            scheduler.start()

            assert not scheduler.scheduler.running

        @patch('core.provider_scheduler.PROVIDER_AUTO_SYNC_ENABLED', True)
        def test_start_schedules_job(self):
            """Test start() schedules the auto-sync job"""
            scheduler = ProviderScheduler()
            scheduler.start()

            jobs = scheduler.scheduler.get_jobs()
            assert len(jobs) == 1
            assert jobs[0].id == 'provider_auto_sync'

            scheduler.stop()

        @patch('core.provider_scheduler.get_auto_discovery')
        async def test_sync_with_error_handling_success(self, mock_get_ad):
            """Test successful sync execution"""
            mock_ad = Mock()
            mock_ad.sync_providers = AsyncMock(return_value={"providers_synced": 5, "models_synced": 100})
            mock_get_ad.return_value = mock_ad

            scheduler = ProviderScheduler()
            await scheduler._sync_with_error_handling()

            assert scheduler._consecutive_failures == 0
            mock_ad.sync_providers.assert_called_once()

        @patch('core.provider_scheduler.get_auto_discovery')
        async def test_sync_with_error_handling_failure(self, mock_get_ad):
            """Test sync failure increments failure counter"""
            mock_ad = Mock()
            mock_ad.sync_providers = AsyncMock(side_effect=Exception("API error"))
            mock_get_ad.return_value = mock_ad

            scheduler = ProviderScheduler()
            await scheduler._sync_with_error_handling()

            assert scheduler._consecutive_failures == 1

        @patch('core.provider_scheduler.get_auto_discovery')
        async def test_trigger_sync_now(self, mock_get_ad):
            """Test manual sync trigger returns result"""
            mock_ad = Mock()
            mock_ad.sync_providers = AsyncMock(return_value={"providers_synced": 2})
            mock_get_ad.return_value = mock_ad

            scheduler = ProviderScheduler()
            result = await scheduler.trigger_sync_now()

            assert result["success"] is True
            assert "result" in result

        def test_get_scheduler_status(self):
            """Test scheduler status returns correct info"""
            scheduler = ProviderScheduler()

            status = scheduler.get_scheduler_status()

            assert "running" in status
            assert "auto_sync_enabled" in status
            assert "consecutive_failures" in status
            assert "jobs" in status

        @patch('core.provider_scheduler.PROVIDER_AUTO_SYNC_ENABLED', True)
        def test_stop_shutdowns_scheduler(self):
            """Test stop() shuts down the scheduler"""
            scheduler = ProviderScheduler()
            scheduler.start()
            assert scheduler.scheduler.running

            scheduler.stop()
            assert not scheduler.scheduler.running
    ```
