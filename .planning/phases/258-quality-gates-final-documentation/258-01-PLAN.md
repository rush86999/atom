## Objective

**CI/CD Quality Gates:** Set up automated quality gates in the CI/CD pipeline that enforce coverage thresholds (70% → 75% → 80%) and 100% test pass rate. Build gates should prevent merging if builds fail or quality thresholds are not met.

**Purpose:** Quality gates ensure that code quality standards are maintained automatically. Every pull request must pass coverage thresholds and achieve 100% test pass rate before merging. This prevents regression and maintains the quality improvements from Phases 251-257.

**Output:** Automated quality gate workflows that enforce coverage and pass rate requirements, with clear feedback to developers on what needs to be fixed.

### Quality Gate Strategy

**Progressive Enforcement:**
- Phase 1: Warn on coverage below 70%, block on test failures
- Phase 2: Block merges if coverage below 75%
- Phase 3: Block merges if coverage below 80%

**Coverage Targets:**
- Backend: 70% → 75% → 80% (from 4.60% baseline)
- Frontend: 70% → 75% → 80% (from 14.12% baseline)

```yaml
name: Quality Gate

on:
  pull_request:
    branches: [main, develop]
  push:
    branches: [main, develop]

jobs:
  quality-check:
    runs-on: ubuntu-latest
    timeout-minutes: 15

    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.11'
          cache: 'pip'

      - name: Install backend dependencies
        run: |
          cd backend
          pip install -e ".[test]"

      - name: Run backend tests with coverage
        run: |
          cd backend
          python3 -m pytest \
            --cov-config=.coverage-rc \
            --cov=core --cov=api --cov=tools \
            --cov-report=json:tests/coverage_reports/metrics/coverage_pr.json \
            --cov-report=term-missing \
            --cov-branch \
            --tb=short \
            --ignore=tests/database/test_migrations.py \
            --ignore=tests/e2e_ui \
            -v

      - name: Check backend coverage threshold
        run: |
          cd backend
          python3 -c "
          import json
          with open('tests/coverage_reports/metrics/coverage_pr.json') as f:
              data = json.load(f)
          coverage = data['totals']['percent_covered']
          print(f'Coverage: {coverage:.2f}%')
          if coverage < 70.0:
              print(f'❌ Coverage {coverage:.2f}% below threshold 70%')
              exit(1)
          print(f'✅ Coverage {coverage:.2f}% meets threshold 70%')
          "

      - name: Set up Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'
          cache-dependency-path: frontend-nextjs/package-lock.json

      - name: Install frontend dependencies
        run: |
          cd frontend-nextjs
          npm ci

      - name: Run frontend tests with coverage
        run: |
          cd frontend-nextjs
          npm run test:coverage -- --reporter=json --reporter=text

      - name: Check frontend coverage threshold
        run: |
          cd frontend-nextjs
          python3 -c "
          import json
          with open('coverage/coverage-summary.json') as f:
              data = json.load(f)
          coverage = data['total']['lines']['pct']
          print(f'Coverage: {coverage:.2f}%')
          if coverage < 70.0:
              print(f'❌ Coverage {coverage:.2f}% below threshold 70%')
              exit(1)
          print(f'✅ Coverage {coverage:.2f}% meets threshold 70%')
          "

      - name: Comment PR with coverage results
        if: github.event_name == 'pull_request'
        uses: actions/github-script@v7
        with:
          script: |
            const fs = require('fs');
            const backendData = JSON.parse(fs.readFileSync('backend/tests/coverage_reports/metrics/coverage_pr.json', 'utf8'));
            const frontendData = JSON.parse(fs.readFileSync('frontend-nextjs/coverage/coverage-summary.json', 'utf8'));

            const backendCoverage = backendData.totals.percent_covered.toFixed(2);
            const frontendCoverage = frontendData.total.lines.pct.toFixed(2);

            const body = `## 📊 Coverage Report

            | Component | Coverage | Threshold | Status |
            |-----------|----------|-----------|--------|
            | Backend | ${backendCoverage}% | 70% | ${backendCoverage >= 70 ? '✅' : '❌'} |
            | Frontend | ${frontendCoverage}% | 70% | ${frontendCoverage >= 70 ? '✅' : '❌'} |

            ${backendCoverage < 70 || frontendCoverage < 70 ? '⚠️ **Action Required**: Coverage below threshold. Please add tests to improve coverage.' : '✅ All coverage thresholds met.'}
            `;

            github.rest.issues.createComment({
              issue_number: context.issue.number,
              owner: context.repo.owner,
              repo: context.repo.repo,
              body: body
            });
```

**Create update script:** `.github/scripts/update-quality-threshold.py`

```python
#!/usr/bin/env python3
"""
Update quality gate thresholds based on current coverage.

This script automatically adjusts thresholds when coverage improves.
Usage: python3 .github/scripts/update-quality-threshold.py
"""

import yaml
import json
import sys
from datetime import datetime
from pathlib import Path

def load_config():
    """Load quality gate configuration."""
    config_path = Path(".github/quality-gate-config.yml")
    with open(config_path) as f:
        return yaml.safe_load(f)

def get_current_coverage(component):
    """Get current coverage from latest measurement."""
    if component == "backend":
        coverage_file = Path("backend/tests/coverage_reports/metrics/coverage_latest.json")
    else:
        coverage_file = Path("frontend-nextjs/coverage/coverage-summary.json")

    if not coverage_file.exists():
        print(f"⚠️ Coverage file not found: {coverage_file}")
        return None

    with open(coverage_file) as f:
        data = json.load(f)

    if component == "backend":
        return data["totals"]["percent_covered"]
    else:
        return data["total"]["lines"]["pct"]

def update_threshold_if_improved(config, component):
    """Update threshold if coverage has improved."""
    current_cov = get_current_coverage(component)
    if current_cov is None:
        return False

    component_config = config["quality_gates"][component]
    current_threshold = component_config["current_threshold"]

    # Auto-increment threshold if coverage exceeds by 5%
    if current_cov >= current_threshold + 5:
        new_threshold = min(current_cov, 80)  # Cap at 80%
        component_config["current_threshold"] = new_threshold
        print(f"✅ {component.capitalize()} threshold updated: {current_threshold}% → {new_threshold}%")
        return True

    print(f"ℹ️ {component.capitalize()} threshold: {current_threshold}% (coverage: {current_cov:.2f}%)")
    return False

def save_config(config):
    """Save updated configuration."""
    config_path = Path(".github/quality-gate-config.yml")
    with open(config_path, "w") as f:
        yaml.dump(config, f, default_flow_style=False)

def main():
    """Main execution."""
    config = load_config()

    print("🔍 Checking coverage for threshold updates...")
    print(f"Timestamp: {datetime.now().isoformat()}")

    backend_updated = update_threshold_if_improved(config, "backend")
    frontend_updated = update_threshold_if_improved(config, "frontend")

    if backend_updated or frontend_updated:
        save_config(config)
        print("\n✅ Configuration updated. Commit the changes to apply new thresholds.")
        return 0
    else:
        print("\nℹ️ No threshold updates needed.")
        return 0

if __name__ == "__main__":
    sys.exit(main())
```
