    ```python
    """GovernanceCache comprehensive coverage tests."""
    import pytest
    import time
    from unittest.mock import Mock
    from core.agent_governance_cache import GovernanceCache
    from core.models import MaturityLevel


    @pytest.fixture
    def cache():
        """GovernanceCache fixture."""
    cache = GovernanceCache()
    cache.clear()  # Start fresh
    return cache


    class TestCacheOperations:
        """Test basic cache CRUD operations."""

        def test_cache_hit_returns_cached_value(self, cache):
            """Cache hit returns previously cached value."""
            agent_data = {
                "agent_id": "agent-123",
                "maturity_level": MaturityLevel.AUTONOMOUS,
                "enabled": True
            }

            cache.set("agent-123", agent_data)
            result = cache.get("agent-123")

            assert result is not None
            assert result["agent_id"] == "agent-123"
            assert result["maturity_level"] == MaturityLevel.AUTONOMOUS

        def test_cache_miss_returns_none(self, cache):
            """Cache miss returns None."""
            result = cache.get("nonexistent")
            assert result is None

        def test_cache_set_with_ttl_expires(self, cache):
            """Cache entry expires after TTL."""
            cache.set("temp-agent", {"data": "value"}, ttl=0.1)  # 100ms TTL

            # Immediately available
            assert cache.get("temp-agent") is not None

            # Wait for expiration
            time.sleep(0.15)
            assert cache.get("temp-agent") is None

        def test_cache_delete_removes_entry(self, cache):
            """Delete removes entry from cache."""
            cache.set("agent-123", {"data": "value"})
            assert cache.get("agent-123") is not None

            cache.delete("agent-123")
            assert cache.get("agent-123") is None

        def test_cache_clear_removes_all_entries(self, cache):
            """Clear removes all cache entries."""
            cache.set("agent-1", {"data": "value1"})
            cache.set("agent-2", {"data": "value2"})

            cache.clear()

            assert cache.get("agent-1") is None
            assert cache.get("agent-2") is None


    class TestCachePerformance:
        """Test cache performance metrics."""

        def test_cache_lookup_performance_sub_millisecond(self, cache):
            """Cache lookup completes in < 1ms."""
            cache.set("agent-perf", {"data": "value"})

            start = time.perf_counter()
            for _ in range(1000):
                cache.get("agent-perf")
            end = time.perf_counter()

            avg_time_ms = (end - start) * 1000 / 1000
            assert avg_time_ms < 1.0, f"Cache lookup too slow: {avg_time_ms:.3f}ms"

        def test_cache_throughput_high(self, cache):
            """Cache handles high throughput operations."""
            start = time.perf_counter()

            for i in range(1000):
                cache.set(f"agent-{i}", {"data": f"value-{i}"})
                cache.get(f"agent-{i}")

            end = time.perf_counter()
            total_time = end - start

            assert total_time < 1.0, f"Throughput too low: {1000/total_time:.0f} ops/sec"


    class TestCacheStatistics:
        """Test cache statistics tracking."""

        def test_cache_hit_ratio_calculated(self, cache):
            """Hit ratio accurately calculated."""
            cache.set("agent-1", {"data": "value1"})

            # Generate hits and misses
            cache.get("agent-1")  # hit
            cache.get("agent-1")  # hit
            cache.get("nonexistent")  # miss

            stats = cache.get_statistics()
            assert stats["hits"] == 2
            assert stats["misses"] == 1
            assert stats["hit_ratio"] == 2/3

        def test_cache_size_tracked(self, cache):
            """Cache size accurately tracked."""
            cache.set("agent-1", {"data": "value1"})
            cache.set("agent-2", {"data": "value2"})

            stats = cache.get_statistics()
            assert stats["size"] == 2

        def test_cache_reset_statistics(self, cache):
            """Statistics can be reset."""
            cache.set("agent-1", {"data": "value1"})
            cache.get("agent-1")  # hit

            cache.reset_statistics()

            stats = cache.get_statistics()
            assert stats["hits"] == 0
            assert stats["misses"] == 0


    class TestCacheInvalidation:
        """Test cache invalidation strategies."""

        def test_invalidate_by_pattern(self, cache):
            """Invalidate cache entries matching pattern."""
            cache.set("agent-123", {"data": "value1"})
            cache.set("agent-456", {"data": "value2"})
            cache.set("workflow-789", {"data": "value3"})

            cache.invalidate_pattern("agent-*")

            assert cache.get("agent-123") is None
            assert cache.get("agent-456") is None
            assert cache.get("workflow-789") is not None

        def test_invalidate_by_maturity_level(self, cache):
            """Invalidate all agents of specific maturity level."""
            cache.set("agent-1", {"maturity_level": MaturityLevel.STUDENT})
            cache.set("agent-2", {"maturity_level": MaturityLevel.STUDENT})
            cache.set("agent-3", {"maturity_level": MaturityLevel.AUTONOMOUS})

            cache.invalidate_by_maturity(MaturityLevel.STUDENT)

            assert cache.get("agent-1") is None
            assert cache.get("agent-2") is None
            assert cache.get("agent-3") is not None


    class TestCacheConcurrency:
        """Test cache thread-safety."""

        def test_concurrent_set_operations(self, cache):
            """Concurrent set operations are thread-safe."""
            import threading

            def set_values(start, end):
                for i in range(start, end):
                    cache.set(f"agent-{i}", {"value": i})

            threads = [
                threading.Thread(target=set_values, args=(0, 100)),
                threading.Thread(target=set_values, args=(100, 200)),
                threading.Thread(target=set_values, args=(200, 300))
            ]

            for t in threads:
                t.start()
            for t in threads:
                t.join()

            stats = cache.get_statistics()
            assert stats["size"] == 300
    ```
