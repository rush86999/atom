  <files>.planning/phases/188-coverage-gap-closure/188-01-BASELINE.md</files>
  <action>
Parse coverage.json to identify and prioritize gaps:

```python
# Run from backend directory
python3 << 'EOF'
import json
from pathlib import Path

# Load coverage data
with open('coverage.json') as f:
    data = json.load(f)

# Analyze gaps
gaps = {
    'critical': [],   # <50% coverage, high business impact
    'moderate': [],   # 50-75% coverage
    'zero': [],       # 0% coverage
    'good': [],       # >75% coverage
}

for file_path, file_data in data['files'].items():
    # Filter to core/ only (skip tests/, __init__.py)
    if not file_path.startswith('core/'):
        continue
    if '__pycache__' in file_path or '__init__' in file_path:
        continue

    pct = file_data['summary']['percent_covered']
    missing = file_data['summary']['missing_lines']
    statements = file_data['summary']['num_statements']

    if statements < 20:  # Skip tiny files
        continue

    entry = {
        'file': file_path,
        'percent': round(pct, 1),
        'missing': missing,
        'statements': statements,
        'covered': statements - missing
    }

    if pct == 0:
        gaps['zero'].append(entry)
    elif pct < 50:
        gaps['critical'].append(entry)
    elif pct < 75:
        gaps['moderate'].append(entry)
    else:
        gaps['good'].append(entry)

# Sort by percent (lowest first)
for key in gaps:
    gaps[key].sort(key=lambda x: x['percent'])

# Print summary
print(f"=== Coverage Gap Analysis ===")
print(f"Critical (<50%, >20 stmts): {len(gaps['critical'])} files")
for f in gaps['critical'][:10]:
    print(f"  {f['file']}: {f['percent']}% ({f['missing']} missing / {f['statements']} total)")
print(f"\nZero coverage: {len(gaps['zero'])} files")
for f in gaps['zero'][:10]:
    print(f"  {f['file']}: 0%")

# Save to baseline doc
with open('../.planning/phases/188-coverage-gap-closure/188-01-BASELINE.md', 'w') as out:
    out.write(f"# Phase 188 Coverage Baseline\n\n")
    out.write(f"**Generated:** {Path.cwd().name} {datetime}\n\n")
    out.write(f"## Summary\n\n")
    total_files = len(gaps['critical']) + len(gaps['moderate']) + len(gaps['good']) + len(gaps['zero'])
    out.write(f"- Total analyzed: {total_files} files\n")
    out.write(f"- Critical gaps (<50%): {len(gaps['critical'])} files\n")
    out.write(f"- Zero coverage: {len(gaps['zero'])} files\n")
    out.write(f"- Moderate (50-75%): {len(gaps['moderate'])} files\n")
    out.write(f"- Good (>75%): {len(gaps['good'])} files\n\n")

    out.write(f"## Critical Gaps (Priority Order)\n\n")
    for f in gaps['critical']:
        out.write(f"- **{f['file']}**: {f['percent']}% ({f['missing']} missing / {f['statements']} total)\n")

    out.write(f"\n## Zero Coverage Files\n\n")
    for f in gaps['zero']:
        out.write(f"- **{f['file']}**: 0%\n")

print(f"\nBaseline saved to 188-01-BASELINE.md")
EOF
```
