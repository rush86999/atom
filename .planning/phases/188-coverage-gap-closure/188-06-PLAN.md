import json
from pathlib import Path

# Load final coverage data
with open('coverage.json') as f:
    data = json.load(f)

# Calculate overall coverage
total_statements = 0
total_covered = 0

files_by_coverage = {
    'zero': [],
    'low': [],      # <50%
    'medium': [],   # 50-75%
    'good': [],     # 75-90%
    'excellent': [] # >90%
}

for file_path, file_data in data['files'].items():
    if not file_path.startswith('core/'):
        continue
    if '__pycache__' in file_path or '__init__' in file_path:
        continue

    statements = file_data['summary']['num_statements']
    covered = file_data['summary']['covered_lines']
    percent = file_data['summary']['percent_covered']

    if statements < 20:
        continue

    total_statements += statements
    total_covered += covered

    if percent == 0:
        files_by_coverage['zero'].append(file_path)
    elif percent < 50:
        files_by_coverage['low'].append(file_path)
    elif percent < 75:
        files_by_coverage['medium'].append(file_path)
    elif percent < 90:
        files_by_coverage['good'].append(file_path)
    else:
        files_by_coverage['excellent'].append(file_path)

overall_pct = (total_covered / total_statements * 100) if total_statements > 0 else 0

print(f"=== FINAL COVERAGE REPORT ===")
print(f"Overall coverage: {overall_pct:.2f}%")
print(f"Total statements: {total_statements}")
print(f"Total covered: {total_covered}")
print()
print(f"Zero coverage: {len(files_by_coverage['zero'])} files")
print(f"Low (<50%): {len(files_by_coverage['low'])} files")
print(f"Medium (50-75%): {len(files_by_coverage['medium'])} files")
print(f"Good (75-90%): {len(files_by_coverage['good'])} files")
print(f"Excellent (>90%): {len(files_by_coverage['excellent'])} files")
print()

# Check specific files from baseline
target_files = [
    'core/agent_evolution_loop.py',
    'core/agent_graduation_service.py',
    'core/agent_promotion_service.py',
    'core/llm/cognitive_tier_system.py',
    'core/llm/cache_aware_router.py',
]

print(f"=== TARGET FILE COVERAGE ===")
for file_path in target_files:
    if file_path in data['files']:
        fd = data['files'][file_path]
        pct = fd['summary']['percent_covered']
        covered = fd['summary']['covered_lines']
        total = fd['summary']['num_statements']
        print(f"{file_path}: {pct:.1f}% ({covered}/{total} lines)")
    else:
        print(f"{file_path}: NOT FOUND")

# Save to verification doc
with open('../.planning/phases/188-coverage-gap-closure/188-06-COVERAGE-FINAL.md', 'w') as out:
    out.write(f"# Phase 188 Final Coverage Report\n\n")
    out.write(f"**Generated:** {Path.cwd().name} {datetime}\n\n")
    out.write(f"## Overall Coverage\n\n")
    out.write(f"**{overall_pct:.2f}%** - {total_covered}/{total_statements} lines covered\n\n")
    out.write(f"## Success Criteria Check\n\n")
    out.write(f"- [X] Overall coverage: {overall_pct:.2f}% {'(PASS 76%+)' if overall_pct >= 76 else '(FAIL <76%)'}\n")
    out.write(f"- [X] Zero coverage files: {len(files_by_coverage['zero'])} (target: 0)\n")
    out.write(f"- [X] Below-50% files: {len(files_by_coverage['low'])} (target: minimize)\n")
    out.write(f"\n## Target Files\n\n")
    for file_path in target_files:
        if file_path in data['files']:
            fd = data['files'][file_path]
            pct = fd['summary']['percent_covered']
            out.write(f"- **{file_path}**: {pct:.1f}%\n")

print(f"\nFinal report saved to 188-06-COVERAGE-FINAL.md")
EOF
