from sqlalchemy.orm import Session

from core.proposal_service import ProposalService, AgentProposal, ProposalStatus
from core.agent_governance_service import AgentGovernanceService
from core.models import AgentRegistry, AgentMaturity

class TestProposalServiceCoverage:
    """Coverage expansion for ProposalService."""

    @pytest.fixture
    def db_session(self):
        """Get test database session."""
        from core.database import SessionLocal
        session = SessionLocal()
        yield session
        session.rollback()
        session.close()

    @pytest.fixture

    def test_create_proposal_success(self, proposal_service, db_session):

            name="Intern Agent",
            maturity=AgentMaturity.INTERN

        proposal = proposal_service.create_proposal(

        proposal = proposal_service.create_proposal(
