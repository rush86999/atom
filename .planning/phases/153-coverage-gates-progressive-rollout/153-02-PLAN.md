   ```javascript
   get coverageThreshold() {
     const phase = process.env.COVERAGE_PHASE || 'phase_1';

     const thresholds = {
       phase_1: {
         branches: 70,
         functions: 70,
         lines: 70,
         statements: 70,
       },
       phase_2: {
         branches: 75,
         functions: 75,
         lines: 75,

       },
       phase_3: {
         branches: 80,
         functions: 80,
         lines: 80,
         statements: 80,
       },
     };

     return {
       global: thresholds[phase],
       // New code always requires 80% regardless of phase
       './src/**/*.{ts,tsx}': {
         branches: 80,
         functions: 80,
         lines: 80,
         statements: 80,
       },
     };
   }
   ```
