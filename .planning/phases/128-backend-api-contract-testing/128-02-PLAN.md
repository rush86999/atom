"""Contract tests for core API endpoints using Schemathesis."""
import pytest
from conftest import schema, app_client
from hypothesis import settings


class TestHealthEndpoints:
    """Contract tests for health check endpoints."""

    @schema.parametrize(endpoint="/health")
    @settings(max_examples=10, deadline=None)
    def test_health_endpoint_contracts(self, case, app_client):
        """Test /health endpoint conforms to OpenAPI spec."""
        response = case.call_and_validate()
        # Additional business logic assertions
        assert response.status_code in [200, 503]
        if response.status_code == 200:
            data = response.json()
            assert "status" in data or "memory_mb" in data

    @schema.parametrize(endpoint="/")
    @settings(max_examples=5, deadline=None)
    def test_root_endpoint_contracts(self, case, app_client):
        """Test root endpoint conforms to OpenAPI spec."""
        response = case.call_and_validate()
        assert response.status_code == 200
        data = response.json()
        assert "name" in data


class TestAgentEndpoints:
