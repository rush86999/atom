def test_knowledge_graph_relations(self, db_session):
    """Cover knowledge graph operations (lines 220-280)"""
    model = AgentWorldModel(db_session, agent_id="agent-1")

    # Store related facts
    fact1 = model.store_fact("Invoice approval requires VP", "policy.pdf", {})
    fact2 = model.store_fact("VP is Jane Doe", "org_chart.pdf", {})

    # Create relationship
    model.create_relation(
        from_fact_id=fact1.id,
        to_fact_id=fact2.id,
        relation_type="references"
    )

    # Get related facts
    related = model.get_related_facts(fact1.id)

    assert len(related) >= 1
    assert fact2.id in [r.id for r in related]
