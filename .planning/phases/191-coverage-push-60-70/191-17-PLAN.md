def test_generate_citations(self):
    """Cover citation generation (lines 150-200)"""
    extractor = PolicyFactExtractor()

    facts = [
        {"fact": "Expenses > $100 need approval", "line_number": 5},
        {"fact": "Submit within 30 days", "line_number": 8},
    ]

    citations = extractor.generate_citations(
        facts=facts,
        source="expense_policy.pdf",
        metadata={"page": 1}
    )

    assert len(citations) == 2
    assert citations[0]["source"] == "expense_policy.pdf"
    assert citations[0]["line_number"] == 5
