def test_social_graph_connections(self, db_session):
    """Cover social graph operations (lines 250-376)"""
    # Create multiple agents
    agents = [
        AgentRegistry(id=f"agent-{i}", name=f"Agent {i}", maturity_level="INTERN")
        for i in range(5)
    ]
    db_session.add_all(agents)
    db_session.commit()

    service = AgentSocialLayer(db_session)

    # Create connections
    service.create_connection("agent-0", "agent-1")
    service.create_connection("agent-0", "agent-2")
    service.create_connection("agent-1", "agent-3")

    # Get connections for agent-0
    connections = service.get_connections("agent-0")

    assert len(connections) == 2
    assert "agent-1" in [c.connected_agent_id for c in connections]
    assert "agent-2" in [c.connected_agent_id for c in connections]
