def test_thread_safe_cache_operations(self, db_session):
    """Cover thread safety (lines 200-278)"""
    import threading

    cache = GovernanceCache(db_session)
    results = []
    errors = []

    def cache_operation():
        try:
            result = cache.get_permission(
                AgentRegistry(id="agent", maturity_level="INTERN"),
                "stream_llm",
                complexity=2
            )
            results.append(result)
        except Exception as e:
            errors.append(e)

    threads = [threading.Thread(target=cache_operation) for _ in range(100)]
    for t in threads:
        t.start()
    for t in threads:
        t.join()

    assert len(errors) == 0, f"Thread safety violations: {errors}"
    assert len(results) == 100
