            bp = debugger.get_breakpoint(bp_id)

        def test_conditional_breakpoint_not_triggered(self, debugger, sample_workflow):
            \"\"\"Test conditional breakpoint doesn't trigger when condition is false.\"\"\"
            bp_id = debugger.set_breakpoint(
                workflow_id=sample_workflow.id,
                step_id="step-3",
                condition="status == 'failed'"
            )
            # Execute workflow where condition is false
            result = debugger.execute_until_breakpoint(sample_workflow.id, bp_id)
            assert result["triggered"] is False

        def test_breakpoint_with_hit_count(self, debugger, sample_workflow):
            \"\"\"Test breakpoint that triggers after N hits.\"\"\"
            bp_id = debugger.set_breakpoint(

                hit_count=3

            export_path = tmp_path / "debug_session.json"
