    class TestProblemAnalysis:
        """Test APAR problem analysis functionality."""

        def test_analyze_problem_basic(self, apar_engine, sample_problem):
            """Test basic problem analysis.\"\"\"
            analysis = apar_engine.analyze_problem(sample_problem)
            assert "problem_type" in analysis
            assert "root_cause" in analysis
            assert "severity" in analysis

        @pytest.mark.parametrize("problem_type,expected_category", [
            ("performance", "performance_degradation"),
            ("error", "runtime_error"),
            ("timeout", "timeout_issue"),
            ("resource", "resource_exhaustion"),
        ])
        def test_categorize_problem_by_type(self, apar_engine, problem_type, expected_category):
            """Test problem categorization by type.\"\"\"
            problem = Problem(id=f"prob-{problem_type}", type=problem_type, description="Test")
            category = apar_engine.categorize_problem(problem)
            assert category == expected_category

        def test_analyze_problem_with_context(self, apar_engine):
            """Test problem analysis with additional context.\"\"\"
            problem = Problem(
                id="prob-002",
                type="error",
                description="API timeout",
                context={"endpoint": "/api/v1/chat", "timeout_ms": 5000}
            )
            analysis = apar_engine.analyze_problem(problem)
            assert analysis["context"]["endpoint"] == "/api/v1/chat"

        def test_severity_assessment(self, apar_engine):
            """Test automatic severity assessment.\"\"\"
            critical_problem = Problem(id="prob-003", type="error", description="System crash")
            severity = apar_engine.assess_severity(critical_problem)
            assert severity in ["low", "medium", "high", "critical"]

        @pytest.mark.parametrize("severity,expected_impact", [
            ("low", {"user_impact": "minimal", "urgency": "low"}),
            ("medium", {"user_impact": "moderate", "urgency": "medium"}),
            ("high", {"user_impact": "significant", "urgency": "high"}),
            ("critical", {"user_impact": "severe", "urgency": "critical"}),
        ])
        def test_severity_impact_mapping(self, apar_engine, severity, expected_impact):
            """Test severity to impact mapping.\"\"\"
            impact = apar_engine.get_severity_impact(severity)
            assert impact["user_impact"] == expected_impact["user_impact"]

    class TestResolutionGeneration:
        """Test APAR resolution generation.\"\"\"

        def test_generate_resolution_strategy(self, apar_engine, sample_problem):
            """Test resolution strategy generation.\"\"\"
            strategy = apar_engine.generate_resolution(sample_problem)
            assert "resolution_type" in strategy
            assert "steps" in strategy
            assert len(strategy["steps"]) > 0

        @pytest.mark.parametrize("problem_type,resolution_type", [
            ("performance", "optimization"),
            ("error", "retry_with_backoff"),
            ("timeout", "increase_timeout"),
            ("resource", "scale_resources"),
        ])
        def test_resolution_type_by_problem(self, apar_engine, problem_type, resolution_type):
            """Test resolution type selection by problem type.\"\"\"
            problem = Problem(id=f"prob-{problem_type}", type=problem_type, description="Test")
            strategy = apar_engine.generate_resolution(problem)
            assert strategy["resolution_type"] == resolution_type

        def test_resolution_steps_ordering(self, apar_engine, sample_problem):
            """Test resolution steps are properly ordered.\"\"\"
            strategy = apar_engine.generate_resolution(sample_problem)
            steps = strategy["steps"]
            assert len(steps) >= 2
            # Verify steps have sequence numbers
            for i, step in enumerate(steps):
                assert step["sequence"] == i + 1

        def test_resolution_with_parameters(self, apar_engine):
            """Test resolution generation with parameters.\"\"\"
            problem = Problem(
                id="prob-004",
                type="timeout",
                description="Timeout issue",
                parameters={"current_timeout": 5000, "max_timeout": 30000}
            )
            strategy = apar_engine.generate_resolution(problem)
            assert "parameters" in strategy
            assert strategy["parameters"]["new_timeout"] > 5000

    class TestResolutionExecution:
        """Test APAR resolution execution.\"\"\"

        def test_execute_resolution_success(self, apar_engine, sample_problem):
            \"\"\"Test successful resolution execution.\"\"\"
            strategy = apar_engine.generate_resolution(sample_problem)
            result = apar_engine.execute_resolution(sample_problem.id, strategy)
            assert result["status"] == "success"
            assert "execution_time_ms" in result

        def test_execute_resolution_with_steps(self, apar_engine, sample_problem):
            \"\"\"Test executing resolution with multiple steps.\"\"\"
            strategy = apar_engine.generate_resolution(sample_problem)
            result = apar_engine.execute_resolution(sample_problem.id, strategy)
            assert result["steps_executed"] > 0
            assert result["steps_executed"] <= len(strategy["steps"])

        def test_execute_resolution_failure_handling(self, apar_engine):
            \"\"\"Test resolution execution failure handling.\"\"\"
            problem = Problem(id="prob-005", type="error", description="Unresolvable error")
            strategy = {"resolution_type": "manual_intervention", "steps": []}
            result = apar_engine.execute_resolution(problem.id, strategy)
            assert result["status"] in ["failed", "requires_manual_intervention"]

        def test_rollback_on_resolution_failure(self, apar_engine, sample_problem):
            \"\"\"Test rollback when resolution execution fails mid-way.\"\"\"
            strategy = {
                "resolution_type": "multi_step",
                "steps": [
                    {"sequence": 1, "action": "step1", "rollback": "rollback1"},
                    {"sequence": 2, "action": "step2", "rollback": "rollback2"},
                ]
            }
            # Mock step 2 failure
            with patch.object(apar_engine, '_execute_step', side_effect=[Mock(success=True), Exception("Step 2 failed")]):
                result = apar_engine.execute_resolution(sample_problem.id, strategy)
                assert result["rolled_back"] is True

    class TestLearningAndFeedback:
        """Test APAR learning and feedback mechanisms.\"\"\"

        def test_record_resolution_outcome(self, apar_engine, sample_problem):
            \"\"\"Test recording resolution outcome for learning.\"\"\"
            outcome = {
                "resolution_id": "res-001",
                "problem_id": sample_problem.id,
                "success": True,
                "execution_time_ms": 1500,
                "user_satisfaction": "high"
            }
            recorded = apar_engine.record_outcome(outcome)
            assert recorded["recorded"] is True

        @pytest.mark.parametrize("success,satisfaction,confidence_delta", [
            (True, "high", 0.1),
            (True, "medium", 0.05),
            (False, "low", -0.1),
            (False, "medium", -0.05),
        ])
        def test_confidence_update_from_outcome(self, apar_engine, success, satisfaction, confidence_delta):
            \"\"\"Test confidence score updates based on outcomes.\"\"\"
            initial_confidence = apar_engine.get_confidence("resolution_strategy_1")
            outcome = {
                "resolution_id": "res-002",
                "strategy_id": "resolution_strategy_1",
                "success": success,
                "user_satisfaction": satisfaction
            }
            apar_engine.record_outcome(outcome)
            new_confidence = apar_engine.get_confidence("resolution_strategy_1")
            assert abs(new_confidence - initial_confidence - confidence_delta) < 0.01

        def test_get_resolution_statistics(self, apar_engine):
            \"\"\"Test getting resolution statistics.\"\"\"
            stats = apar_engine.get_resolution_statistics()
            assert "total_resolutions" in stats
            assert "success_rate" in stats
            assert stats["success_rate"] >= 0.0
            assert stats["success_rate"] <= 1.0

    class TestAPErrorsAndEdgeCases:
        """Test APAR error handling and edge cases.\"\"\"

        def test_handle_unknown_problem_type(self, apar_engine):
            \"\"\"Test handling unknown problem type gracefully.\"\"\"
            problem = Problem(id="prob-006", type="unknown_type", description="Unknown")
            analysis = apar_engine.analyze_problem(problem)
            assert "error" in analysis or analysis["problem_type"] == "unknown"

        def test_handle_missing_context(self, apar_engine):
            \"\"\"Test handling problem with missing context.\"\"\"
            problem = Problem(id="prob-007", type="error", description="No context")
            analysis = apar_engine.analyze_problem(problem)
            assert "root_cause" in analysis

        def test_empty_resolution_steps(self, apar_engine, sample_problem):
            \"\"\"Test handling resolution with no steps.\"\"\"
            strategy = {"resolution_type": "none", "steps": []}
            result = apar_engine.execute_resolution(sample_problem.id, strategy)
            assert result["status"] in ["no_action", "skipped"]

        def test_concurrent_resolution_handling(self,apar_engine):
            \"\"\"Test handling multiple concurrent resolutions.\"\"\"
            problems = [Problem(id=f"prob-{i}", type="error", description=f"Error {i}") for i in range(5)]
            results = [apar_engine.generate_resolution(p) for p in problems]
            assert len(results) == 5
