    ```python
    @given(
        model=sampled_from(["gpt-4", "claude-3"]),
        finish_reason=sampled_from(["stop", "length"])
    )
    @settings(max_examples=20)
    def test_single_chunk_stream_eos(self, model, finish_reason):
        """INVARIANT: Single-chunk stream has EOS signaled"""
        chunk = {
            'index': 0,
            'content': "response",
            'finish_reason': finish_reason,
            'model': model,
            'provider': 'openai'
        }
        assert chunk['finish_reason'] == finish_reason
        assert chunk['index'] == 0
    ```

    b) Large stream handling (100+ chunks):
    ```python
    @given(chunk_count=integers(min_value=100, max_value=1000))
    @settings(max_examples=20)
    def test_large_stream_ordering_invariant(self, chunk_count):
        """INVARIANT: Ordering holds for large streams"""
        chunks = [{'index': i, 'content': f"token_{i}"} for i in range(chunk_count)]
        assert [c['index'] for c in chunks] == list(range(chunk_count))
    ```

    c) Unicode content handling:
    ```python
    @given(unicode_content=text(min_size=1, max_size=1000))
    @settings(max_examples=50)
    def test_unicode_chunk_integrity(self, unicode_content):
        """INVARIANT: Unicode content preserved across chunks"""
        chunk = {'content': unicode_content, 'index': 0}
        assert chunk['content'] == unicode_content
        assert len(chunk['content'].encode('utf-8')) >= len(unicode_content)
    ```
