    class TestAtomSaaSWebSocketCoverage:
        '''Coverage-driven tests for atom_saas_websocket.py (0% -> 75%+ target)'''

        @pytest.mark.parametrize("connection_id,client_id,expected_state", [
            ("conn-1", "client-1", "authenticated"),
            ("conn-2", "client-2", "authenticated"),
            ("conn-3", None, "unauthenticated"),
        ])
        def test_connection_establishment(self, connection_id, client_id, expected_state, mocker):
            '''Cover connection establishment (lines 50-150)'''
            # Mock WebSocket connection
            mock_websocket = mocker.patch('core.atom_saas_websocket.WebSocketConnection')
            mock_ws = Mock()
            mock_ws.id = connection_id
            mock_websocket.return_value = mock_ws

            # Mock authentication
            mock_auth = mocker.patch('core.atom_saas_websocket.authenticate_connection')
            mock_auth.return_value = Mock(client_id=client_id) if client_id else None

            from core.atom_saas_websocket import AtomSaaSWebSocket
            ws_manager = AtomSaaSWebSocket()

            result = ws_manager.connect(connection_id, mock_ws)
            assert result.state == expected_state

        @pytest.mark.parametrize("message_type,expected_handler", [
            ("chat", "handle_chat_message"),
            ("agent_execute", "handle_agent_execution"),
            ("canvas_update", "handle_canvas_update"),
            ("ping", "handle_ping"),
        ])
        def test_message_routing(self, message_type, expected_handler, mocker):
            '''Cover message routing (lines 150-250)'''
            # Mock handler
            mock_handler = mocker.patch(f'core.atom_saas_websocket.{expected_handler}')
            mock_handler.return_value = {"status": "processed"}

            from core.atom_saas_websocket import AtomSaaSWebSocket
            ws_manager = AtomSaaSWebSocket()

            message = {"type": message_type, "data": "test"}
            result = ws_manager.route_message("conn-1", message)

            assert result["status"] == "processed"
            mock_handler.assert_called_once()

        @pytest.mark.parametrize("broadcast_type,recipient_count", [
            ("all", 5),          # Broadcast to all clients
            ("room", 3),         # Broadcast to room
            ("client", 1),       # Broadcast to single client
        ])
        def test_broadcasting(self, broadcast_type, recipient_count, mocker):
            '''Cover broadcasting (lines 250-350)'''
            # Mock connections
            mock_connections = [
                Mock(id=f"conn-{i}", send=Mock()) for i in range(5)
            ]

            from core.atom_saas_websocket import AtomSaaSWebSocket
            ws_manager = AtomSaaSWebSocket()
            ws_manager.connections = {f"conn-{i}": conn for i, conn in enumerate(mock_connections)}

            message = {"type": "broadcast", "data": "test"}
            ws_manager.broadcast(message, broadcast_type=broadcast_type)

            if broadcast_type == "all":
                expected_calls = 5
            elif broadcast_type == "room":
                expected_calls = 3
            else:
                expected_calls = 1

            # Verify send called correct number of times
            total_calls = sum(conn.send.call_count for conn in mock_connections)
            assert total_calls >= expected_calls

        @pytest.mark.parametrize("room,action,member_count", [
            ("room-1", "join", 3),     # Join room
            ("room-1", "leave", 2),    # Leave room
            ("room-2", "create", 1),   # Create room
        ])
        def test_room_management(self, room, action, member_count, mocker):
            '''Cover room/channel management (lines 350-450)'''
            from core.atom_saas_websocket import AtomSaaSWebSocket
            ws_manager = AtomSaaSWebSocket()

            if action == "join":
                ws_manager.join_room("conn-1", room)
                ws_manager.join_room("conn-2", room)
                ws_manager.join_room("conn-3", room)
            elif action == "leave":
                ws_manager.join_room("conn-1", room)
                ws_manager.join_room("conn-2", room)
                ws_manager.join_room("conn-3", room)
                ws_manager.leave_room("conn-3", room)
            else:
                ws_manager.create_room(room)

            room_members = ws_manager.get_room_members(room)
            assert len(room_members) == member_count

        @pytest.mark.parametrize("token,should_auth", [
            ("valid_token", True),
            ("invalid_token", False),
            ("", False),
            (None, False),
        ])
        def test_authentication(self, token, should_auth, mocker):
            '''Cover authentication (lines 450-550)'''
            # Mock token validation
            mock_validate = mocker.patch('core.atom_saas_websocket.validate_token')
            mock_validate.return_value = Mock(client_id="client-1") if token == "valid_token" else None

            from core.atom_saas_websocket import AtomSaaSWebSocket
            ws_manager = AtomSaaSWebSocket()

            result = ws_manager.authenticate("conn-1", token)
            assert result.authenticated == should_auth

        @pytest.mark.parametrize("invalid_input,error_type", [
            (None, TypeError),           # None connection_id
            ("", ValueError),            # Empty connection_id
            ({"type": None}, ValueError),  # None message type
        ])
        def test_input_validation(self, invalid_input, error_type):
            '''Cover input validation (lines 100-150)'''
            from core.atom_saas_websocket import AtomSaaSWebSocket
            ws_manager = AtomSaaSWebSocket()

            with pytest.raises(error_type):
                if isinstance(invalid_input, str):
                    ws_manager.connect(invalid_input, Mock())
                else:
                    ws_manager.route_message("conn-1", invalid_input)
