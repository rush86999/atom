    class TestIntegrationDataMapperCoverage:
        '''Coverage-driven tests for integration_data_mapper.py (0% -> 75%+ target)'''

        @pytest.mark.parametrize("source_type,target_type,expected_result", [
            ("json", "dict", {"status": "transformed"}),
            ("csv", "list", [{"col1": "val1"}]),
            ("xml", "dict", {"root": {}}),
        ])
        def test_data_transformation(self, source_type, target_type, expected_result, mocker):
            '''Cover data transformation (lines 50-150)'''
            # Mock data source
            mock_source = mocker.patch('core.integration_data_mapper.DataSource')
            if source_type == "json":
                mock_data = '{"key": "value"}'
            elif source_type == "csv":
                mock_data = "col1,col2\nval1,val2"
            else:
                mock_data = "<root></root>"

            mock_source.return_value.read.return_value = mock_data

            from core.integration_data_mapper import IntegrationDataMapper
            mapper = IntegrationDataMapper()

            result = mapper.transform(source_type, target_type, mock_data)
            assert result is not None

        @pytest.mark.parametrize("mapping_config,field_count", [
            ({"rename": {"old_name": "new_name"}}, 1),
            ({"extract": ["field1", "field2"]}, 2),
            ({"combine": {"new_field": ["f1", "f2"]}}, 1),
        ])
        def test_schema_mapping(self, mapping_config, field_count, mocker):
            '''Cover schema mapping (lines 150-250)'''
            # Mock source data
            mock_data = {
                "old_name": "value",
                "field1": "v1",
                "field2": "v2",
                "f1": "a",
                "f2": "b"
            }

            from core.integration_data_mapper import IntegrationDataMapper
            mapper = IntegrationDataMapper()

            result = mapper.apply_mapping(mock_data, mapping_config)
            assert len(result) >= field_count

        @pytest.mark.parametrize("validation_rule,should_pass", [
            ("required_fields", True),    # All required fields present
            ("type_check", True),        # Types match schema
            ("value_range", True),       # Values in range
            ("no_required", False),      # Missing required fields
        ])
        def test_data_validation(self, validation_rule, should_pass):
            '''Cover data validation (lines 250-350)'''
            from core.integration_data_mapper import IntegrationDataMapper
            mapper = IntegrationDataMapper()

            if validation_rule == "required_fields":
                data = {"field1": "value1", "field2": "value2"}
                schema = {"required": ["field1", "field2"]}
            elif validation_rule == "type_check":
                data = {"field1": "string", "field2": 123}
                schema = {"types": {"field1": "str", "field2": "int"}}
            elif validation_rule == "value_range":
                data = {"field1": 50}
                schema = {"ranges": {"field1": (0, 100)}}
            else:
                data = {"field1": "value1"}
                schema = {"required": ["field1", "field2"]}

            result = mapper.validate(data, schema, rule=validation_rule)
            assert result.valid == should_pass

        @pytest.mark.parametrize("batch_size,record_count,expected_batches", [
            (10, 100, 10),
            (50, 150, 3),
            (100, 50, 1),
        ])
        def test_batch_processing(self, batch_size, record_count, expected_batches, mocker):
            '''Cover batch processing (lines 350-450)'''
            # Mock data source
            mock_data = [Mock(id=i, value=f"value_{i}") for i in range(record_count)]

            from core.integration_data_mapper import IntegrationDataMapper
            mapper = IntegrationDataMapper()

            batches = list(mapper.process_batch(mock_data, batch_size=batch_size))
            assert len(batches) == expected_batches

        @pytest.mark.parametrize("error_type,recovery_action", [
            ("invalid_data", "skip"),
            ("transform_error", "log_and_continue"),
            ("connection_error", "retry"),
        ])
        def test_error_handling(self, error_type, recovery_action, mocker):
            '''Cover error handling (lines 450-550)'''
            from core.integration_data_mapper import IntegrationDataMapper
            mapper = IntegrationDataMapper()

            if error_type == "invalid_data":
                data = {"invalid": "data"}
            elif error_type == "transform_error":
                data = Mock(transform=Mock(side_effect=Exception("Transform failed")))
            else:
                data = None

            result = mapper.handle_error(error_type, data)
            assert result.action == recovery_action

        @pytest.mark.parametrize("invalid_input,error_type", [
            (None, TypeError),           # None data
            ({}, ValueError),            # Empty data
            ("invalid", ValueError),     # Invalid format
        ])
        def test_input_validation(self, invalid_input, error_type):
            '''Cover input validation (lines 100-150)'''
            from core.integration_data_mapper import IntegrationDataMapper
            mapper = IntegrationDataMapper()

            with pytest.raises(error_type):
                mapper.validate_input(invalid_input)
