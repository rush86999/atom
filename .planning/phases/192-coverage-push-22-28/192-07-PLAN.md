    class TestAtomMetaAgentExtended:
        '''Extended coverage tests for atom_meta_agent.py

        Builds on existing coverage to achieve 75%+ target.

        may be partially skipped due to extensive mocking requirements.
        Focus on testable synchronous methods and simpler async paths.
        '''

        @pytest.mark.parametrize("maturity_level,expected_capabilities", [
            ("STUDENT", ["read", "observe"]),          # Read-only
            ("INTERN", ["read", "observe", "delegate"]),  # + delegation
            ("SUPERVISED", ["read", "observe", "delegate", "execute"]),  # + execution
            ("AUTONOMOUS", ["read", "observe", "delegate", "execute", "delete"]),  # Full
        ])
        def test_initialization_with_maturity(self, maturity_level, expected_capabilities, mocker, db_session):
            '''Cover initialization with maturity levels (lines 50-120)'''
            mock_governance = mocker.patch('core.atom_meta_agent.AgentGovernanceService')
            mock_governance.return_value.get_maturity.return_value = maturity_level

            from core.atom_meta_agent import AtomMetaAgent
            meta_agent = AtomMetaAgent(
                db_session=db_session,
                agent_id="meta-1",
                maturity_level=maturity_level
            )

            assert meta_agent.maturity_level == maturity_level
            assert all(cap in meta_agent.capabilities for cap in expected_capabilities)

        @pytest.mark.parametrize("task_type,routing_strategy", [
            ("simple", "direct"),
            ("complex", "decompose"),
            ("parallel", "split"),
            ("sequential", "chain"),
        ])
        def test_task_delegation(self, task_type, routing_strategy, mocker, db_session):
            '''Cover task delegation logic (lines 120-250)'''
            mock_sub_agent = mocker.patch('core.atom_meta_agent.Agent')
            mock_sub_agent.return_value.execute.return_value = Mock(status="completed")

            from core.atom_meta_agent import AtomMetaAgent
            meta_agent = AtomMetaAgent(db_session=db_session, agent_id="meta-1")

            result = meta_agent.delegate_task(
                task_type=task_type,
                task_data={"input": "test"}
            )

            assert result.routing_strategy == routing_strategy

        @pytest.mark.parametrize("agent_count,coordination_mode", [
            (1, "single"),
            (2, "pair"),
            (3, "group"),
            (5, "swarm"),
        ])
        def test_sub_agent_coordination(self, agent_count, coordination_mode, mocker, db_session):
            '''Cover sub-agent coordination (lines 250-400)'''
            mock_agents = [
                Mock(id=f"agent-{i}", status="ready", execute=AsyncMock(return_value=Mock(status="completed")))
                for i in range(agent_count)
            ]

            from core.atom_meta_agent import AtomMetaAgent
            meta_agent = AtomMetaAgent(db_session=db_session, agent_id="meta-1")

            # Note: This tests the coordination setup, not full async execution
            coordination = meta_agent.setup_coordination(
                agents=mock_agents,
                mode=coordination_mode
            )

            assert coordination.mode == coordination_mode
            assert len(coordination.agents) == agent_count

        @pytest.mark.parametrize("state,expected_action", [
            ("idle", "start"),
            ("running", "continue"),
            ("paused", "resume"),
            ("error", "recover"),
            ("completed", "finalize"),
        ])
        def test_state_management(self, state, expected_action, mocker, db_session):
            '''Cover state management (lines 400-500)'''
            from core.atom_meta_agent import AtomMetaAgent
            meta_agent = AtomMetaAgent(db_session=db_session, agent_id="meta-1")
            meta_agent.state = state

            action = meta_agent.get_next_action()
            assert action == expected_action

        @pytest.mark.parametrize("error_type,recovery_strategy", [
            ("timeout", "retry"),
            ("agent_failure", "reassign"),
            ("resource_exhausted", "scale_down"),
            ("validation_error", "abort"),
        ])
        def test_error_handling(self, error_type, recovery_strategy, mocker, db_session):
            '''Cover error handling and recovery (lines 500-600)'''
            mock_agent = mocker.patch('core.atom_meta_agent.Agent')
            mock_agent.return_value.execute.side_effect = Exception(error_type)

            from core.atom_meta_agent import AtomMetaAgent
            meta_agent = AtomMetaAgent(db_session=db_session, agent_id="meta-1")

            recovery = meta_agent.handle_error(error_type=error_type)
            assert recovery.strategy == recovery_strategy

        @pytest.mark.parametrize("invalid_task,error_type", [
            (None, TypeError),           # None task
            ({}, ValueError),            # Empty task
            ({"type": None}, ValueError),  # None task type
        ])
        def test_task_validation(self, invalid_task, error_type, db_session):
            '''Cover task validation (lines 100-150)'''
            from core.atom_meta_agent import AtomMetaAgent
            meta_agent = AtomMetaAgent(db_session=db_session, agent_id="meta-1")

            with pytest.raises(error_type):
                meta_agent.validate_task(invalid_task)
