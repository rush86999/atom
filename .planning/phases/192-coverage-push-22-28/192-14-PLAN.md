    ```python
    from fastapi.testclient import TestClient
    from unittest.mock import Mock, patch

    class TestBusinessFactsRoutesCoverage:
        '''Coverage-driven tests for business_facts_routes.py (0% -> 70%+ target)'''

        @pytest.fixture
        def test_client(self):
            '''FastAPI TestClient instance'''
            from api.admin.business_facts_routes import app
            return TestClient(app)

        @pytest.fixture
        def admin_headers(self):
            '''Admin authentication headers'''
            return {"Authorization": "Bearer admin_token"}

        @pytest.fixture
        def user_headers(self):
            '''Regular user authentication headers'''
            return {"Authorization": "Bearer user_token"}

        @pytest.mark.parametrize("endpoint,method,expected_status", [
            ("/api/v1/admin/facts", "GET", 200),
            ("/api/v1/admin/facts", "POST", 201),
            ("/api/v1/admin/facts/fact-1", "GET", 200),
            ("/api/v1/admin/facts/fact-1", "PUT", 200),
            ("/api/v1/admin/facts/fact-1", "DELETE", 200),
        ])
        def test_fact_crud_endpoints(self, endpoint, method, expected_status, test_client, admin_headers, mocker):
            '''Cover fact CRUD endpoints (lines 50-200)'''
            # Mock database queries
            mock_db = mocker.patch('api.admin.business_facts_routes.db')
            mock_db.query.return_value.filter.return_value.first.return_value = Mock(
                id="fact-1",
                fact="Invoices > $500 need VP approval",
                citation="policy.pdf:p4",
                verified=True
            )

            if method == "GET" and endpoint == "/api/v1/admin/facts":
                response = test_client.get(endpoint, headers=admin_headers)
            elif method == "POST":
                response = test_client.post(
                    endpoint,
                    json={
                        "fact": "Invoices > $500 need VP approval",
                        "citation": "policy.pdf:p4"
                    },
                    headers=admin_headers
                )
            elif method == "GET":
                response = test_client.get(endpoint, headers=admin_headers)
            elif method == "PUT":
                response = test_client.put(
                    endpoint,
                    json={"fact": "Updated fact"},
                    headers=admin_headers
                )
            else:  # DELETE
                response = test_client.delete(endpoint, headers=admin_headers)

            assert response.status_code in [expected_status, 404]

        @pytest.mark.parametrize("citation_status,expected_result", [
            ("valid", True),         # Valid citation
            ("invalid", False),      # Invalid citation
            ("missing_file", False), # Missing source file
        ])
        def test_citation_verification(self, citation_status, expected_result, test_client, admin_headers, mocker):
            '''Cover citation verification (lines 200-350)'''
            # Mock citation verifier
            mock_verifier = mocker.patch('api.admin.business_facts_routes.CitationVerifier')
            if citation_status == "valid":
                mock_verifier.return_value.verify.return_value = {"valid": True}
            elif citation_status == "invalid":
                mock_verifier.return_value.verify.return_value = {"valid": False, "error": "Not found"}
            else:
                mock_verifier.return_value.verify.side_effect = FileNotFoundError("File not found")

            endpoint = "/api/v1/admin/facts/fact-1/verify"
            response = test_client.post(endpoint, headers=admin_headers)

            assert response.status_code in [200, 404]
            if response.status_code == 200:
                assert response.json().get("verified") == expected_result

        @pytest.mark.parametrize("query,fact_count", [
            ("invoice", 2),       # Matches 2 facts
            ("approval", 1),      # Matches 1 fact
            ("nonexistent", 0),   # No matches
        ])
        def test_fact_retrieval(self, query, fact_count, test_client, admin_headers, mocker):
            '''Cover fact retrieval and search (lines 350-450)'''
            # Mock fact search
            mock_search = mocker.patch('api.admin.business_facts_routes.search_facts')
            mock_facts = [
                Mock(id="fact-1", fact="Invoice > $500 needs approval", citation="policy.pdf:p4"),
                Mock(id="fact-2", fact="Invoice approval process", citation="handbook.pdf:p2"),
                Mock(id="fact-3", fact="Expense policy", citation="policy.pdf:p10"),
            ]
            mock_search.return_value = [f for f in mock_facts if query.lower() in f.fact.lower()]

            endpoint = f"/api/v1/admin/facts/search?q={query}"
            response = test_client.get(endpoint, headers=admin_headers)

            assert response.status_code == 200
            assert len(response.json().get("facts", [])) == fact_count

        @pytest.mark.parametrize("user_role,endpoint,expected_status", [
            ("admin", "/api/v1/admin/facts", 200),      # Admin can access
            ("user", "/api/v1/admin/facts", 403),        # User forbidden
            ("admin", "/api/v1/admin/facts", 201),       # Admin can create
            ("user", "/api/v1/admin/facts", 403),        # User forbidden
        ])
        def test_access_control(self, user_role, endpoint, expected_status, test_client, mocker):
            '''Cover RBAC enforcement (lines 450-550)'''
            # Mock RBAC check
            mock_rbac = mocker.patch('api.admin.business_facts_routes.check_admin_access')
            mock_rbac.return_value = (user_role == "admin")

            headers = {"Authorization": f"Bearer {user_role}_token"}

            if "POST" in str(endpoint):
                response = test_client.post(
                    endpoint,
                    json={"fact": "Test", "citation": "test.pdf:p1"},
                    headers=headers
                )
            else:
                response = test_client.get(endpoint, headers=headers)

            assert response.status_code == expected_status

        @pytest.mark.parametrize("invalid_fact,expected_error", [
            ({}, 400),                  # Empty body
            ({"fact": ""}, 400),        # Empty fact
            ({"fact": "Test"}, 400),    # Missing citation
        ])
        def test_fact_validation(self, invalid_fact, expected_error, test_client, admin_headers):
            '''Cover fact validation (lines 150-200)'''
            endpoint = "/api/v1/admin/facts"

            response = test_client.post(endpoint, json=invalid_fact, headers=admin_headers)
            assert response.status_code == expected_error
    ```

    - Lines 450-550: RBAC enforcement
