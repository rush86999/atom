    class TestEpisodeSegmentationExtended:
        '''Extended coverage tests for episode_segmentation_service.py

        Builds on existing coverage to achieve 75%+ target.
        Focus areas: time gaps, topic changes, canvas context, consolidation.
        '''

        @pytest.mark.parametrize("gap_minutes,expected_segments", [
            (5, 3),     # 5-min gap: 3 segments
            (15, 2),    # 15-min gap: 2 segments
            (30, 1),    # 30-min gap: 1 segment
            (60, 1),    # 60-min gap: 1 segment
        ])
        def test_segment_by_time_gap(self, gap_minutes, expected_segments, mocker, db_session):
            '''Cover time-based segmentation (lines 100-200)'''
            # Mock LanceDB client
            mock_lancedb = mocker.patch('core.episode_segmentation_service.lancedb')
            mock_client = Mock()
            mock_table = Mock()
            mock_client.open_table.return_value = mock_table

            # Mock episode data with time gaps
            mock_df = pd.DataFrame({
                'episode_id': ['ep1'] * 4,
                'timestamp': [
                    datetime.now(timezone.utc) - timedelta(minutes=100),
                    datetime.now(timezone.utc) - timedelta(minutes=90),
                    datetime.now(timezone.utc) - timedelta(minutes=50),
                    datetime.now(timezone.utc) - timedelta(minutes=10),
                ],
                'content': ['msg1', 'msg2', 'msg3', 'msg4']
            })
            mock_table.search.return_value.to_pandas.return_value = mock_df
            mock_lancedb.connect.return_value = mock_client

            from core.episode_segmentation_service import EpisodeSegmentationService
            service = EpisodeSegmentationService(db_session)
            segments = service.segment_by_time_gap(episode_id="ep1", gap_minutes=gap_minutes)

            assert len(segments) == expected_segments

        @pytest.mark.parametrize("similarity_threshold,topic_changes", [
            (0.9, 0),    # High similarity: no topic change
            (0.7, 1),    # Medium similarity: 1 topic change
            (0.5, 2),    # Low similarity: 2 topic changes
        ])
        def test_segment_by_topic_change(self, similarity_threshold, topic_changes, mocker, db_session):
            '''Cover topic change detection (lines 200-350)'''
            # Mock embedding service
            mock_embeddings = mocker.patch('core.episode_segmentation_service.EmbeddingService')
            mock_embeddings.return_value.compute_similarity.side_effect = [
                0.95, 0.6, 0.4,  # Similarities between consecutive messages
            ]

            from core.episode_segmentation_service import EpisodeSegmentationService
            service = EpisodeSegmentationService(db_session)
            segments = service.segment_by_topic_change(
                episode_id="ep1",
                similarity_threshold=similarity_threshold
            )

            # Count topic boundaries
            boundaries = sum(1 for s in segments if s.get('is_topic_boundary', False))
            assert boundaries == topic_changes

        @pytest.mark.parametrize("canvas_type,creates_boundary", [
            ("markdown", True),     # Canvas presentation creates boundary
            ("chart", True),        # Chart creates boundary
            ("form", True),         # Form creates boundary
            ("sheet", True),        # Sheet creates boundary
        ])
        def test_segment_by_canvas_context(self, canvas_type, creates_boundary, mocker, db_session):
            '''Cover canvas context segmentation (lines 350-500)'''
            # Mock CanvasAudit
            mock_canvas_audit = mocker.patch('core.episode_segmentation_service.CanvasAudit')
            mock_canvas_audit.query.filter.return_value.all.return_value = [
                Mock(canvas_type=canvas_type, created_at=datetime.now(timezone.utc))
            ]

            from core.episode_segmentation_service import EpisodeSegmentationService
            service = EpisodeSegmentationService(db_session)
            segments = service.segment_by_canvas_context(episode_id="ep1")

            canvas_segments = [s for s in segments if s.get('trigger') == 'canvas']
            if creates_boundary:
                assert len(canvas_segments) > 0

        @pytest.mark.parametrize("agent_status,creates_boundary", [
            ("completed", True),     # Completed execution creates boundary
            ("failed", True),        # Failed execution creates boundary
            ("running", False),      # Running execution does not create boundary
        ])
        def test_segment_by_task_completion(self, agent_status, creates_boundary, mocker, db_session):
            '''Cover task completion detection (lines 500-600)'''
            # Mock AgentExecution
            mock_agent_exec = mocker.patch('core.episode_segmentation_service.AgentExecution')
            mock_agent_exec.query.filter.return_value.all.return_value = [
                Mock(status=agent_status, completed_at=datetime.now(timezone.utc))
            ]

            from core.episode_segmentation_service import EpisodeSegmentationService
            service = EpisodeSegmentationService(db_session)
            segments = service.segment_by_task_completion(episode_id="ep1")

            task_segments = [s for s in segments if s.get('trigger') == 'task_completion']
            if creates_boundary:
                assert len(task_segments) > 0
