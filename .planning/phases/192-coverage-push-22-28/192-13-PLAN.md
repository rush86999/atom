    from fastapi.testclient import TestClient
    from unittest.mock import Mock, patch

    class TestWorkflowRoutesCoverage:
        '''Coverage-driven tests for workflow_routes.py (0% -> 70%+ target)'''

        @pytest.fixture
        def test_client(self):
            '''FastAPI TestClient instance'''
            from api.workflow_routes import app
            return TestClient(app)

        @pytest.fixture
        def mock_auth_token(self):
            '''Mock authentication token'''
            return "Bearer test_token_123"

        @pytest.mark.parametrize("endpoint,method,expected_status", [
            ("/api/v1/workflows", "GET", 200),
            ("/api/v1/workflows", "POST", 201),
            ("/api/v1/workflows/wf-1", "GET", 200),
            ("/api/v1/workflows/wf-1", "PUT", 200),
            ("/api/v1/workflows/wf-1", "DELETE", 200),
        ])
        def test_workflow_crud_endpoints(self, endpoint, method, expected_status, test_client, mock_auth_token, mocker):
            '''Cover workflow CRUD endpoints (lines 50-200)'''
            # Mock database queries
            mock_db = mocker.patch('api.workflow_routes.db')
            mock_db.query.return_value.filter.return_value.first.return_value = Mock(
                id="wf-1",
                name="Test Workflow",
                definition={"nodes": [], "connections": []}
            )

            headers = {"Authorization": mock_auth_token}

            if method == "GET" and endpoint == "/api/v1/workflows":
                response = test_client.get(endpoint, headers=headers)
            elif method == "POST":
                response = test_client.post(
                    endpoint,
                    json={"name": "New Workflow", "definition": {}},
                    headers=headers
                )
            elif method == "GET":
                response = test_client.get(endpoint, headers=headers)
            elif method == "PUT":
                response = test_client.put(
                    endpoint,
                    json={"name": "Updated Workflow"},
                    headers=headers
                )
            else:  # DELETE
                response = test_client.delete(endpoint, headers=headers)

            assert response.status_code in [expected_status, 404]  # 404 acceptable if not found

        @pytest.mark.parametrize("action,expected_status", [
            ("start", 202),
            ("stop", 200),
            ("pause", 200),
            ("resume", 200),
        ])
        def test_workflow_execution_actions(self, action, expected_status, test_client, mock_auth_token, mocker):
            '''Cover workflow execution actions (lines 200-350)'''
            # Mock workflow engine
            mock_engine = mocker.patch('api.workflow_routes.WorkflowEngine')
            mock_engine.return_value.execute.return_value = Mock(
                execution_id="exec-1",
                status="running"
            )

            endpoint = f"/api/v1/workflows/wf-1/{action}"
            headers = {"Authorization": mock_auth_token}

            if action == "start":
                response = test_client.post(endpoint, headers=headers)
            else:
                response = test_client.post(endpoint, headers=headers)

            assert response.status_code in [expected_status, 404]

        @pytest.mark.parametrize("template_action,expected_status", [
            ("create", 201),
            ("list", 200),
            ("apply", 201),
        ])
        def test_workflow_template_operations(self, template_action, expected_status, test_client, mock_auth_token, mocker):
            '''Cover workflow template operations (lines 350-500)'''
            # Mock template system
            mock_templates = mocker.patch('api.workflow_routes.WorkflowTemplateSystem')

            if template_action == "create":
                endpoint = "/api/v1/workflows/templates"
                response = test_client.post(
                    endpoint,
                    json={"name": "Test Template", "definition": {}},
                    headers={"Authorization": mock_auth_token}
                )
            elif template_action == "list":
                endpoint = "/api/v1/workflows/templates"
                response = test_client.get(endpoint, headers={"Authorization": mock_auth_token})
            else:  # apply
                endpoint = "/api/v1/workflows/templates/tpl-1/apply"
                response = test_client.post(
                    endpoint,
                    json={"parameters": {}},
                    headers={"Authorization": mock_auth_token}
                )

            assert response.status_code in [expected_status, 404]

        @pytest.mark.parametrize("maturity,action,expected_status", [
            ("STUDENT", "create", 403),     # STUDENT blocked from creating
            ("INTERN", "create", 201),      # INTERN can create
            ("SUPERVISED", "execute", 202), # SUPERVISED can execute
            ("AUTONOMOUS", "delete", 200),  # AUTONOMOUS can delete
        ])
        def test_governance_enforcement(self, maturity, action, expected_status, test_client, mocker):
            '''Cover governance enforcement by maturity (lines 500-600)'''
            # Mock governance check
            mock_governance = mocker.patch('api.workflow_routes.check_governance')
            if maturity == "STUDENT" and action in ["create", "execute", "delete"]:
                mock_governance.return_value = {"allowed": False, "reason": "STUDENT blocked"}
            else:
                mock_governance.return_value = {"allowed": True}

            endpoint = f"/api/v1/workflows"
            headers = {"Authorization": f"Bearer {maturity}_token"}

            if action == "create":
                response = test_client.post(
                    endpoint,
                    json={"name": "Test", "definition": {}},
                    headers=headers
                )
            elif action == "execute":
                response = test_client.post("/api/v1/workflows/wf-1/start", headers=headers)
            else:  # delete
                response = test_client.delete("/api/v1/workflows/wf-1", headers=headers)

            assert response.status_code == expected_status

        @pytest.mark.parametrize("invalid_input,expected_error", [
            ({}, 400),                  # Empty body
            ({"name": ""}, 400),         # Empty name
            ({"definition": "invalid"}, 400),  # Invalid definition
        ])
        def test_validation_errors(self, invalid_input, expected_error, test_client, mock_auth_token):
            '''Cover validation errors (lines 150-200)'''
            endpoint = "/api/v1/workflows"
            headers = {"Authorization": mock_auth_token}

            response = test_client.post(endpoint, json=invalid_input, headers=headers)
            assert response.status_code == expected_error
