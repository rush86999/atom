    class TestWorkflowTemplateSystemCoverage:
        '''Coverage-driven tests for workflow_template_system.py (0% -> 75%+ target)'''

        @pytest.mark.parametrize("template_type,expected_structure", [
            ("sequential", {"has_start": True, "has_end": True, "linear": True}),
            ("parallel", {"has_start": True, "has_end": True, "branches": True}),
            ("conditional", {"has_start": True, "has_conditions": True}),
        ])
        def test_create_template(self, template_type, expected_structure, mocker, db_session):
            '''Cover template creation (lines 50-150)'''
            # Mock template storage
            mock_storage = mocker.patch('core.workflow_template_system.TemplateStorage')
            mock_storage.save.return_value = "template-123"

            from core.workflow_template_system import WorkflowTemplateSystem
            system = WorkflowTemplateSystem(db_session)
            template = system.create_template(
                name="test_template",
                template_type=template_type,
                definition={"nodes": [], "connections": []}
            )

            assert template.id is not None
            assert template.template_type == template_type

        @pytest.mark.parametrize("validation_check,should_pass", [
            ("has_nodes", True),          # Has required nodes
            ("has_connections", True),    # Has valid connections
            ("no_cycles", True),          # No circular dependencies
            ("all_nodes_connected", True), # All nodes reachable
        ])
        def test_validate_template(self, validation_check, should_pass, mocker, db_session):
            '''Cover template validation (lines 150-300)'''
            # Mock template with various structures
            if validation_check == "has_nodes":
                template = Mock(nodes=[Mock(id="node1")], connections=[])
            elif validation_check == "has_connections":
                template = Mock(
                    nodes=[Mock(id="node1"), Mock(id="node2")],
                    connections=[Mock(source="node1", target="node2")]
                )
            elif validation_check == "no_cycles":
                template = Mock(
                    nodes=[Mock(id="node1"), Mock(id="node2")],
                    connections=[Mock(source="node1", target="node2")]
                )
            else:
                template = Mock(
                    nodes=[Mock(id="node1"), Mock(id="node2"), Mock(id="node3")],
                    connections=[
                        Mock(source="node1", target="node2"),
                        Mock(source="node2", target="node3")
                    ]
                )

            from core.workflow_template_system import WorkflowTemplateSystem
            system = WorkflowTemplateSystem(db_session)
            result = system.validate_template(template, check=validation_check)

            assert result.valid == should_pass

        @pytest.mark.parametrize("param_count,param_types", [
            (0, []),                     # No parameters
            (1, ["string"]),              # Single string parameter
            (3, ["string", "int", "bool"]), # Multiple parameters
        ])
        def test_template_instantiation(self, param_count, param_types, mocker, db_session):
            '''Cover template instantiation (lines 300-450)'''
            # Mock template
            template = Mock(
                id="template-1",
                name="test_template",
                parameters=[
                    Mock(name=f"param_{i}", type=param_type, required=True, default=None)
                    for i, param_type in enumerate(param_types)
                ]
            )

            # Mock workflow engine
            mock_engine = mocker.patch('core.workflow_template_system.WorkflowEngine')

            from core.workflow_template_system import WorkflowTemplateSystem
            system = WorkflowTemplateSystem(db_session)

            params = {f"param_{i}": "test_value" for i in range(param_count)}
            workflow = system.instantiate_template(
                template_id="template-1",
                parameters=params
            )

            assert workflow is not None
            assert len(workflow.parameters) == param_count

        @pytest.mark.parametrize("substitution_pattern,expected_result", [
            ("simple", "value"),              # ${param} -> "value"
            ("nested", "nested_value"),       # ${parent.child} -> "nested_value"
            ("default", "default_value"),     # ${param:default} -> "default_value"
        ])
        def test_parameter_substitution(self, substitution_pattern, expected_result, mocker, db_session):
            '''Cover parameter substitution (lines 450-550)'''
            from core.workflow_template_system import WorkflowTemplateSystem
            system = WorkflowTemplateSystem(db_session)

            if substitution_pattern == "simple":
                template = "${param}"
                params = {"param": "value"}
            elif substitution_pattern == "nested":
                template = "${parent.child}"
                params = {"parent": {"child": "nested_value"}}
            else:
                template = "${param:default_value}"
                params = {}

            result = system.substitute_parameters(template, params)
            assert result == expected_result

        @pytest.mark.parametrize("invalid_template,error_type", [
            (None, TypeError),           # None template
            (Mock(nodes=[]), ValueError),  # Empty nodes
            (Mock(nodes=None), ValueError), # None nodes
        ])
        def test_template_validation_errors(self, invalid_template, error_type, db_session):

            system = WorkflowTemplateSystem(db_session)

            with pytest.raises(error_type):
                system.validate_template(invalid_template)
