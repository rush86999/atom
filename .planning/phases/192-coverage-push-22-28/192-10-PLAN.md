    ```python
    class TestConfigCoverage:
        '''Coverage-driven tests for config.py (0% -> 80%+ target)'''

        @pytest.mark.parametrize("env_var,value,expected_type", [
            ("PORT", "8000", int),
            ("DEBUG", "true", bool),
            ("LOG_LEVEL", "INFO", str),
            ("ALLOWED_HOSTS", "localhost,127.0.0.1", list),
        ])
        def test_load_env_var(self, env_var, value, expected_type, monkeypatch):
            '''Cover environment variable loading (lines 50-150)'''
            monkeypatch.setenv(env_var, value)

            from core.config import Config
            config = Config()

            actual_value = getattr(config, env_var.lower())
            assert isinstance(actual_value, expected_type)

        @pytest.mark.parametrize("env_content,expected_values", [
            ("PORT=8000\nDEBUG=true", {"port": 8000, "debug": True}),
            ("LOG_LEVEL=WARNING\nSECRET_KEY=test123", {"log_level": "WARNING", "secret_key": "test123"}),
            ("", {}),  # Empty .env
        ])
        def test_load_env_file(self, env_content, expected_values, tmp_path, monkeypatch):
            '''Cover .env file loading (lines 150-250)'''
            # Create temporary .env file
            env_file = tmp_path / ".env"
            env_file.write_text(env_content)

            monkeypatch.setenv("ENV_FILE", str(env_file))

            from core.config import Config
            config = Config()

            for key, expected_value in expected_values.items():
                actual_value = getattr(config, key)
                assert actual_value == expected_value

        @pytest.mark.parametrize("config_key,validation_rule", [
            ("port", "range_1_65535"),      # Port: 1-65535
            ("debug", "boolean"),            # Debug: boolean
            ("log_level", "in_levels"),      # Log level: valid levels
            ("database_url", "not_empty"),   # Database URL: required
        ])
        def test_validate_config(self, config_key, validation_rule):
            '''Cover config validation (lines 250-400)'''
            from core.config import Config

            if validation_rule == "range_1_65535":
                config = Config()
                config.port = 8000
                assert config.validate_field("port", config.port) is True
                config.port = 70000  # Invalid
                assert config.validate_field("port", config.port) is False

            elif validation_rule == "boolean":
                config = Config()
                config.debug = True
                assert config.validate_field("debug", config.debug) is True

        @pytest.mark.parametrize("field,default_value", [
            ("port", 8000),
            ("debug", False),
            ("log_level", "INFO"),
            ("workers", 1),
        ])
        def test_default_values(self, field, default_value):
            '''Cover default values (lines 400-500)'''
            from core.config import Config
            config = Config()

            actual_value = getattr(config, field)
            assert actual_value == default_value

        @pytest.mark.parametrize("override_source,override_key,override_value", [
            ("env", "PORT", "9000"),           # Environment override
            ("env_file", "DEBUG", "true"),     # .env file override
            ("runtime", "LOG_LEVEL", "DEBUG"), # Runtime override
        ])
        def test_config_overrides(self, override_source, override_key, override_value, monkeypatch):
            '''Cover config override priority (lines 500-600)'''
            if override_source == "env":
                monkeypatch.setenv(override_key, override_value)
            elif override_source == "env_file":
                # This would use a temp .env file
                pass

            from core.config import Config
            config = Config()

            if override_source == "env":
                assert getattr(config, override_key.lower()) == override_value

        @pytest.mark.parametrize("invalid_value,error_type", [
            ("PORT", "invalid", ValueError),     # Invalid int
            ("DEBUG", "not_bool", ValueError),   # Invalid bool
            ("PORT", "-1", ValueError),          # Out of range
            ("PORT", "70000", ValueError),       # Out of range
        ])
        def test_validation_errors(self, invalid_value, error_type, monkeypatch):
            '''Cover validation errors (lines 350-400)'''
            monkeypatch.setenv(*invalid_value)

            from core.config import Config
            with pytest.raises(error_type):
                Config()
    ```
