    ```python
    class TestBYOKHandlerCoverage:
        '''Coverage-driven tests for byok_handler.py (0% -> 70%+ target)

        Strategy: Mock all external LLM providers (OpenAI, Anthropic, DeepSeek, Gemini)
        to test routing logic without making actual API calls.
        '''

        @pytest.mark.parametrize("provider,expected_class", [
            ("openai", "OpenAIHandler"),
            ("anthropic", "AnthropicHandler"),
            ("deepseek", "DeepSeekHandler"),
            ("gemini", "GeminiHandler"),
        ])
        def test_provider_routing(self, provider, expected_class, mocker):
            '''Cover provider routing logic (lines 80-150)'''
            # Mock the provider class
            mock_handler = mocker.patch(f'core.llm.byok_handler.{expected_class}')
            mock_handler.return_value = Mock(stream=Mock(return_value=["token1", "token2"]))

            from core.llm.byok_handler import BYOKHandler
            handler = BYOKHandler(provider=provider)
            tokens = list(handler.stream("test prompt"))

            assert tokens == ["token1", "token2"]
            mock_handler.assert_called_once()

        @pytest.mark.parametrize("error_type,should_retry", [
            ("timeout", True),           # Retry on timeout
            ("rate_limit", True),        # Retry on rate limit
            ("auth_error", False),       # Don't retry auth errors
            ("invalid_request", False),  # Don't retry client errors
        ])
        def test_error_handling_with_retry(self, error_type, should_retry, mocker):
            '''Cover error handling and retry logic (lines 200-300)'''
            mock_handler = mocker.patch('core.llm.byok_handler.OpenAIHandler')

            if error_type == "timeout":
                mock_handler.side_effect = [TimeoutError(), Mock(stream=lambda: ["success"])]
            elif error_type == "rate_limit":
                mock_handler.side_effect = [Exception("429"), Mock(stream=lambda: ["success"])]
            else:
                mock_handler.side_effect = Exception(error_type)

            from core.llm.byok_handler import BYOKHandler
            handler = BYOKHandler(provider="openai", max_retries=2)

            if should_retry:
                tokens = list(handler.stream("test"))
                assert mock_handler.call_count == 2  # Initial + 1 retry
            else:
                with pytest.raises(Exception):
                    list(handler.stream("test"))

        @pytest.mark.parametrize("model,expected_cost_per_1k", [
            ("gpt-4", 0.03),
            ("gpt-3.5-turbo", 0.002),
            ("claude-3-opus", 0.015),
        ])
        def test_cost_tracking(self, model, expected_cost_per_1k, mocker):
            '''Cover cost tracking (lines 350-400)'''
            mock_handler = mocker.patch('core.llm.byok_handler.OpenAIHandler')
            mock_handler.return_value = Mock(stream=lambda: ["token"] * 1000)

            from core.llm.byok_handler import BYOKHandler
            handler = BYOKHandler(provider="openai", model=model)
            list(handler.stream("test"))

            assert handler.total_tokens == 1000
            assert handler.total_cost == expected_cost_per_1k

        @pytest.mark.parametrize("chunk_count,tokens_per_chunk", [
            (1, 10),    # Single chunk
            (5, 20),    # Multiple chunks
            (10, 50),   # Many chunks
        ])
        def test_token_streaming(self, chunk_count, tokens_per_chunk, mocker):
            '''Cover token streaming (lines 150-200)'''
            mock_handler = mocker.patch('core.llm.byok_handler.OpenAIHandler')

            chunks = [["token"] * tokens_per_chunk for _ in range(chunk_count)]
            mock_stream = Mock(return_value=iter(chunks))
            mock_handler.return_value = Mock(stream=mock_stream)

            from core.llm.byok_handler import BYOKHandler
            handler = BYOKHandler(provider="openai")
            tokens = list(handler.stream("test"))

            assert len(tokens) == chunk_count * tokens_per_chunk

        @pytest.mark.parametrize("invalid_input,error_type", [
            (None, TypeError),        # None provider
            ("", ValueError),         # Empty provider
            ("invalid_provider", ValueError),  # Unknown provider
        ])
        def test_initialization_validation(self, invalid_input, error_type):
            '''Cover initialization validation (lines 30-80)'''
            from core.llm.byok_handler import BYOKHandler
            with pytest.raises(error_type):
                BYOKHandler(provider=invalid_input)
    ```
