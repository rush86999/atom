    ```python
    class TestSkillRegistryServiceCoverage:
        '''Coverage-driven tests for skill_registry_service.py (0% -> 75%+ target)'''

        @pytest.mark.parametrize("skill_type,source_path", [
            ("builtin", "skills/builtin/"),
            ("community", "skills/community/"),
            ("custom", "skills/custom/"),
        ])
        def test_register_skill(self, skill_type, source_path, mocker, db_session):
            '''Cover skill registration (lines 50-150)'''
            # Mock skill file system
            mock_path = mocker.patch('core.skill_registry_service.Path')
            mock_path.return_value.exists.return_value = True
            mock_path.return_value.is_dir.return_value = True

            # Mock skill loader
            mock_load = mocker.patch('core.skill_registry_service.load_skill')
            mock_load.return_value = Mock(
                name="test_skill",
                version="1.0.0",
                description="Test skill",
                skill_type=skill_type
            )

            from core.skill_registry_service import SkillRegistryService
            registry = SkillRegistryService(db_session)
            skill = registry.register_skill(
                name="test_skill",
                source_path=source_path,
                skill_type=skill_type
            )

            assert skill.name == "test_skill"
            assert skill.skill_type == skill_type
            mock_load.assert_called_once()

        @pytest.mark.parametrize("query,expected_count", [
            ("test", 2),      # Matches 2 skills
            ("data", 1),      # Matches 1 skill
            ("nonexistent", 0),  # No matches
        ])
        def test_discover_skills(self, query, expected_count, mocker, db_session):
            '''Cover skill discovery (lines 150-250)'''
            # Mock skill database
            mock_query = mocker.patch('core.skill_registry_service.Skill.query')
            mock_query.return_value.filter.return_value.all.return_value = [
                Mock(name="test_skill_1", description="Test data processing"),
                Mock(name="test_skill_2", description="Test analytics"),
                Mock(name="other_skill", description="Other functionality"),
            ]

            from core.skill_registry_service import SkillRegistryService
            registry = SkillRegistryService(db_session)
            results = registry.discover_skills(query=query)

            assert len(results) == expected_count

        @pytest.mark.parametrize("validation_rule,should_pass", [
            ("has_metadata", True),        # Has required metadata
            ("valid_schema", True),        # Valid SKILL.md schema
            ("no_blocked_calls", True),    # No blocked API calls
            ("has_dependencies", True),    # Dependencies satisfied
        ])
        def test_validate_skill(self, validation_rule, should_pass, mocker, db_session):
            '''Cover skill validation (lines 250-400)'''
            # Mock skill data
            mock_skill = Mock(
                name="test_skill",
                metadata={"author": "test", "version": "1.0"},
                schema={"type": "object"},
                blocked_calls=[],
                dependencies=[]
            )

            from core.skill_registry_service import SkillRegistryService
            registry = SkillRegistryService(db_session)
            result = registry.validate_skill(mock_skill, rule=validation_rule)

            assert result.passed == should_pass

        @pytest.mark.parametrize("version_action,expected_result", [
            ("register", "1.0.0"),      # First registration
            ("update", "1.1.0"),        # Version update
            ("rollback", "1.0.0"),      # Rollback to previous
        ])
        def test_skill_versioning(self, version_action, expected_result, mocker, db_session):
            '''Cover skill versioning (lines 400-500)'''
            from core.skill_registry_service import SkillRegistryService
            registry = SkillRegistryService(db_session)

            if version_action == "register":
                result = registry.register_version(
                    skill_name="test_skill",
                    version="1.0.0"
                )
            elif version_action == "update":
                result = registry.register_version(
                    skill_name="test_skill",
                    version="1.1.0"
                )
            else:
                result = registry.rollback_version(
                    skill_name="test_skill",
                    target_version="1.0.0"
                )

            assert result.version == expected_result

        @pytest.mark.parametrize("invalid_skill,error_type", [
            (None, TypeError),           # None skill
            (Mock(name=""), ValueError),  # Empty name
            (Mock(name="x"*500), ValueError),  # Oversized name
        ])
        def test_registration_validation(self, invalid_skill, error_type, db_session):
            '''Cover registration validation (lines 100-150)'''
            from core.skill_registry_service import SkillRegistryService
            registry = SkillRegistryService(db_session)

            with pytest.raises(error_type):
                if invalid_skill is None:
                    registry.register_skill(name=None, source_path="test/")
                else:
                    registry.register_skill(
                        name=invalid_skill.name,
                        source_path="test/"
                    )
    ```
