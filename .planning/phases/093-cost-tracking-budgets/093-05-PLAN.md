     ```python
     def approve_spend_locked(
         self,
         db: Session,
         project_id: str,
         amount: Union[Decimal, str, float],
         description: str = None
     ) -> Dict[str, Any]:
         """
         Approve spend with pessimistic locking (SELECT FOR UPDATE) to prevent race conditions.

         Uses row-level locking to ensure only one transaction can check and update
         the budget at a time, preventing concurrent approvals from exceeding the limit.
         """
         with db.begin():
             # SELECT FOR UPDATE locks the Project row
             project = db.query(Project).filter(
                 Project.id == project_id
             ).with_for_update().first()

             if not project:
                 raise BudgetNotFoundError(f"Project {project_id} not found")

             amount_decimal = to_decimal(amount)

             # Check budget limit (atomic - no concurrent modifications possible)
             if project.actual_burn + amount_decimal > project.budget_amount:
                 raise InsufficientBudgetError(
                     requested=amount_decimal,
                     remaining=project.budget_amount - project.actual_burn,
                     budget_id=project_id
                 )

             # Atomic update (still within lock)
             project.actual_burn += amount_decimal

             # Update budget status
             utilization = (project.actual_burn / project.budget_amount) * 100
             if utilization >= 100:
                 project.budget_status = BudgetStatus.OVER_BUDGET
             elif utilization >= project.project.getattr(project, 'pause_threshold_pct', 90):
                 project.budget_status = BudgetStatus.AT_RISK
             else:
                 project.budget_status = BudgetStatus.ON_TRACK

             db.flush()

             # Create transaction record
             transaction = Transaction(
                 project_id=project_id,
                 amount=amount_decimal,
                 category="spend",
                 description=description or f"Budget spend: {amount_decimal}"
             )
             db.add(transaction)

         return {
             "status": "approved",
             "project_id": project_id,
             "amount": amount_decimal,
             "remaining": project.budget_amount - project.actual_burn,
             "budget_status": project.budget_status.value
         }
     ```

  2. Add `approve_spend_with_retry()` method using optimistic locking:
     - Uses version_id_col pattern if available
     - Retries up to 3 times on StaleDataError
     - Falls back to pessimistic locking if version column not available

  3. Add timeout handling:
     - Set lock timeout (e.g., 5 seconds)
     - Raise TimeoutError if lock cannot be acquired

  4. Update docstrings to document locking behavior

  5. Add logging for lock acquisition/release (useful for debugging)

  IMPORTANT: Pessimistic locking is the default. Optimistic locking is opt-in for low-contention scenarios.
