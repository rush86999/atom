4. Deadlock detection pattern:
```python
import asyncio
from concurrent.futures import ThreadPoolExecutor, as_completed

async def run_concurrent_operations(operations, timeout_seconds=60):
    """Run operations concurrently with deadlock detection via timeout."""
    try:
        async with asyncio.timeout(timeout_seconds):
            results = await asyncio.gather(*operations, return_exceptions=True)
            # Check for actual exceptions (not timeout)
            failures = [r for r in results if isinstance(r, Exception) and not isinstance(r, asyncio.TimeoutError)]
            assert len(failures) == 0, f"Operations failed: {failures}"
            return True
    except asyncio.TimeoutError:
        pytest.fail(f"Deadlock detected: operations did not complete within {timeout_seconds}s")
```

5. Race condition detection pattern:
```python
def test_increment_no_race():
    cache = GovernanceCache()
    key = "race_test_key"
    cache.set(key, 0)

    def increment_worker():
        for _ in range(100):
            current = cache.get(key)
            cache.set(key, current + 1)

    with ThreadPoolExecutor(max_workers=50) as executor:
        futures = [executor.submit(increment_worker) for _ in range(50)]
        for future in as_completed(futures, timeout=30):
            future.result()  # Raises exceptions if any

    final_value = cache.get(key)
    expected = 100 * 50  # 100 increments * 50 workers
    assert final_value == expected, f"Race condition detected: got {final_value}, expected {expected}"
```
